57  /**
58   * This plain provider class can be registered with a JPA RestfulServer
59   * to provide the <code>$binary-access-read</code> and <code>$binary-access-write</code>
60   * operations that can be used to access attachment data as a raw binary.
61   */
62  public class BinaryAccessProvider {
63  
64  	private static final Logger ourLog = LoggerFactory.getLogger(BinaryAccessProvider.class);
65  	@Autowired
66  	private FhirContext myCtx;
67  	@Autowired
68  	private DaoRegistry myDaoRegistry;
69  	@Autowired(required = false)
70  	private IBinaryStorageSvc myBinaryStorageSvc;
71  
72  	/**
73  	 * $binary-access-read
74  	 */
75  	@Operation(name = JpaConstants.OPERATION_BINARY_ACCESS_READ, global = true, manualResponse = true, idempotent = true)
76  	public void binaryAccessRead(
77  		@IdParam IIdType theResourceId,
78  		@OperationParam(name = "path", min = 1, max = 1) IPrimitiveType<String> thePath,
79  		ServletRequestDetails theRequestDetails,
80  		HttpServletRequest theServletRequest,
81  		HttpServletResponse theServletResponse) throws IOException {
82  
83  		String path = validateResourceTypeAndPath(theResourceId, thePath);
84  		IFhirResourceDao dao = getDaoForRequest(theResourceId);
85  		IBaseResource resource = dao.read(theResourceId, theRequestDetails, false);
86  
87  		IBinaryTarget target = findAttachmentForRequest(resource, path, theRequestDetails);
88  
89  		Optional<? extends IBaseExtension<?, ?>> attachmentId = target
90  			.getTarget()
91  			.getExtension()
92  			.stream()
93  			.filter(t -> JpaConstants.EXT_EXTERNALIZED_BINARY_ID.equals(t.getUrl()))
94  			.findFirst();
95  
96  		if (attachmentId.isPresent()) {
97  
98  			@SuppressWarnings("unchecked")
99  			IPrimitiveType<String> value = (IPrimitiveType<String>) attachmentId.get().getValue();
100 			String blobId = value.getValueAsString();
101 
102 			IBinaryStorageSvc.StoredDetails blobDetails = myBinaryStorageSvc.fetchBlobDetails(theResourceId, blobId);
103 			if (blobDetails == null) {
104 				String msg = myCtx.getLocalizer().getMessage(BinaryAccessProvider.class, "unknownBlobId");
105 				throw new InvalidRequestException(msg);
106 			}
107 
108 			theServletResponse.setStatus(200);
109 			theServletResponse.setContentType(blobDetails.getContentType());
110 			if (blobDetails.getBytes() <= Integer.MAX_VALUE) {
111 				theServletResponse.setContentLength((int) blobDetails.getBytes());
112 			}
113 
114 			RestfulServer server = theRequestDetails.getServer();
115 			server.addHeadersToResponse(theServletResponse);
116 
117 			theServletResponse.addHeader(Constants.HEADER_CACHE_CONTROL, Constants.CACHE_CONTROL_PRIVATE);
118 			theServletResponse.addHeader(Constants.HEADER_ETAG, '"' + blobDetails.getHash() + '"');
119 			theServletResponse.addHeader(Constants.HEADER_LAST_MODIFIED, DateUtils.formatDate(blobDetails.getPublished()));
120 
121 			myBinaryStorageSvc.writeBlob(theResourceId, blobId, theServletResponse.getOutputStream());
122 			theServletResponse.getOutputStream().close();
123 
124 		} else {
125 
126 			String contentType = target.getContentType();
127 			contentType = StringUtils.defaultIfBlank(contentType, Constants.CT_OCTET_STREAM);
128 
129 			byte[] data = target.getData();
130 			if (data == null) {
131 				String msg = myCtx.getLocalizer().getMessage(BinaryAccessProvider.class, "noAttachmentDataPresent", sanitizeUrlPart(theResourceId), sanitizeUrlPart(thePath));
132 				throw new InvalidRequestException(msg);
133 			}
134 
135 			theServletResponse.setStatus(200);
136 			theServletResponse.setContentType(contentType);
137 			theServletResponse.setContentLength(data.length);
138 
139 			RestfulServer server = theRequestDetails.getServer();
140 			server.addHeadersToResponse(theServletResponse);
141 
142 			theServletResponse.getOutputStream().write(data);
143 			theServletResponse.getOutputStream().close();
144 
145 		}
146 	}

148 	/**
149 	 * $binary-access-write
150 	 */
151 	@SuppressWarnings("unchecked")
152 	@Operation(name = JpaConstants.OPERATION_BINARY_ACCESS_WRITE, global = true, manualRequest = true, idempotent = false)
153 	public IBaseResource binaryAccessWrite(
154 		@IdParam IIdType theResourceId,
155 		@OperationParam(name = "path", min = 1, max = 1) IPrimitiveType<String> thePath,
156 		ServletRequestDetails theRequestDetails,
157 		HttpServletRequest theServletRequest,
158 		HttpServletResponse theServletResponse) throws IOException {
159 
160 		String path = validateResourceTypeAndPath(theResourceId, thePath);
161 		IFhirResourceDao dao = getDaoForRequest(theResourceId);
162 		IBaseResource resource = dao.read(theResourceId, theRequestDetails, false);
163 
164 		IBinaryTarget target = findAttachmentForRequest(resource, path, theRequestDetails);
165 
166 		String requestContentType = theServletRequest.getContentType();
167 		if (isBlank(requestContentType)) {
168 			throw new InvalidRequestException("No content-target supplied");
169 		}
170 		if (EncodingEnum.forContentTypeStrict(requestContentType) != null) {
171 			throw new InvalidRequestException("This operation is for binary content, got: " + requestContentType);
172 		}
173 
174 		long size = theServletRequest.getContentLength();
175 		ourLog.trace("Request specified content length: {}", size);
176 
177 		String blobId = null;
178 
179 		if (size > 0) {
180 			if (myBinaryStorageSvc != null) {
181 				if (myBinaryStorageSvc.shouldStoreBlob(size, theResourceId, requestContentType)) {
182 					IBinaryStorageSvc.StoredDetails storedDetails = myBinaryStorageSvc.storeBlob(theResourceId, requestContentType, theRequestDetails.getInputStream());
183 					size = storedDetails.getBytes();
184 					blobId = storedDetails.getBlobId();
185 					Validate.notBlank(blobId, "BinaryStorageSvc returned a null blob ID"); // should not happen
186 				}
187 			}
188 		}
189 
190 		if (blobId == null) {
191 			byte[] bytes = IOUtils.toByteArray(theRequestDetails.getInputStream());
192 			size = bytes.length;
193 			target.setData(bytes);
194 		} else {
195 
196 			target
197 				.getTarget()
198 				.getExtension()
199 				.removeIf(t -> JpaConstants.EXT_EXTERNALIZED_BINARY_ID.equals(t.getUrl()));
200 			target.setData(null);
201 
202 			IBaseExtension<?, ?> ext = target.getTarget().addExtension();
203 			ext.setUrl(JpaConstants.EXT_EXTERNALIZED_BINARY_ID);
204 			IPrimitiveType<String> blobIdString = (IPrimitiveType<String>) myCtx.getElementDefinition("string").newInstance();
205 			blobIdString.setValueAsString(blobId);
206 			ext.setValue(blobIdString);
207 		}
208 
209 		target.setContentType(requestContentType);
210 		target.setSize(null);
211 		if (size <= Integer.MAX_VALUE) {
212 			target.setSize((int) size);
213 		}
214 
215 		DaoMethodOutcome outcome = dao.update(resource, theRequestDetails);
216 		return outcome.getResource();
217 	}

219 	@Nonnull
220 	private IBinaryTarget findAttachmentForRequest(IBaseResource theResource, String thePath, ServletRequestDetails theRequestDetails) {
221 		FhirContext ctx = theRequestDetails.getFhirContext();
222 
223 		Optional<IBase> type = ctx.newFluentPath().evaluateFirst(theResource, thePath, IBase.class);
224 		String resType = myCtx.getResourceDefinition(theResource).getName();
225 		if (!type.isPresent()) {
226 			String msg = myCtx.getLocalizer().getMessageSanitized(BinaryAccessProvider.class, "unknownPath", resType, thePath);
227 			throw new InvalidRequestException(msg);
228 		}
229 
230 		// Path is attachment
231 		BaseRuntimeElementDefinition<?> def = ctx.getElementDefinition(type.get().getClass());
232 		if (def.getName().equals("Attachment")) {
233 			ICompositeType attachment = (ICompositeType) type.get();
234 			return new IBinaryTarget() {
235 				@Override
236 				public void setSize(Integer theSize) {
237 					AttachmentUtil.setSize(myCtx, attachment, theSize);
238 				}
239 
240 				@Override
241 				public String getContentType() {
242 					return AttachmentUtil.getOrCreateContentType(myCtx, attachment).getValueAsString();
243 				}
244 
245 				@Override
246 				public byte[] getData() {
247 					IPrimitiveType<byte[]> dataDt = AttachmentUtil.getOrCreateData(theRequestDetails.getFhirContext(), attachment);
248 					return dataDt.getValue();
249 				}
250 
251 				@Override
252 				public IBaseHasExtensions getTarget() {
253 					return (IBaseHasExtensions) AttachmentUtil.getOrCreateData(theRequestDetails.getFhirContext(), attachment);
254 				}
255 
256 				@Override
257 				public void setContentType(String theContentType) {
258 					AttachmentUtil.setContentType(myCtx, attachment, theContentType);
259 				}
260 
261 
262 				@Override
263 				public void setData(byte[] theBytes) {
264 					AttachmentUtil.setData(theRequestDetails.getFhirContext(), attachment, theBytes);
265 				}
266 
267 
268 			};
269 		}
270 
271 		// Path is Binary
272 		if (def.getName().equals("Binary")) {
273 			IBaseBinary binary = (IBaseBinary) type.get();
274 			return new IBinaryTarget() {
275 				@Override
276 				public void setSize(Integer theSize) {
277 					// ignore
278 				}
279 
280 				@Override
281 				public String getContentType() {
282 					return binary.getContentType();
283 				}
284 
285 				@Override
286 				public byte[] getData() {
287 					return binary.getContent();
288 				}
289 
290 				@Override
291 				public IBaseHasExtensions getTarget() {
292 					return (IBaseHasExtensions) BinaryUtil.getOrCreateData(myCtx, binary);
293 				}
294 
295 				@Override
296 				public void setContentType(String theContentType) {
297 					binary.setContentType(theContentType);
298 				}
299 
300 
301 				@Override
302 				public void setData(byte[] theBytes) {
303 					binary.setContent(theBytes);
304 				}
305 
306 
307 			};
308 		}
309 
310 		String msg = myCtx.getLocalizer().getMessageSanitized(BinaryAccessProvider.class, "unknownType", resType, thePath, def.getName());
311 		throw new InvalidRequestException(msg);
312 
313 	}

315 	private String validateResourceTypeAndPath(@IdParam IIdType theResourceId, @OperationParam(name = "path", min = 1, max = 1) IPrimitiveType<String> thePath) {
316 		if (isBlank(theResourceId.getResourceType())) {
317 			throw new InvalidRequestException("No resource type specified");
318 		}
319 		if (isBlank(theResourceId.getIdPart())) {
320 			throw new InvalidRequestException("No ID specified");
321 		}
322 		if (thePath == null || isBlank(thePath.getValue())) {
323 			if ("Binary".equals(theResourceId.getResourceType())) {
324 				return "Binary";
325 			}
326 			throw new InvalidRequestException("No path specified");
327 		}
328 
329 		return thePath.getValue();
330 	}

332 	@Nonnull
333 	private IFhirResourceDao getDaoForRequest(@IdParam IIdType theResourceId) {
334 		String resourceType = theResourceId.getResourceType();
335 		IFhirResourceDao dao = myDaoRegistry.getResourceDao(resourceType);
336 		if (dao == null) {
337 			throw new InvalidRequestException("Unknown/unsupported resource type: " + sanitizeUrlPart(resourceType));
338 		}
339 		return dao;
340 	}

342 	/**
343 	 * Wraps an Attachment datatype or Binary resource, since they both
344 	 * hold binary content but don't look entirely similar
345 	 */
346 	private interface IBinaryTarget {
347 
348 		void setSize(Integer theSize);
349 
350 		String getContentType();
351 
352 		void setContentType(String theContentType);
353 
354 		byte[] getData();
355 
356 		void setData(byte[] theBytes);
357 
358 		IBaseHasExtensions getTarget();
359 
360 	}
