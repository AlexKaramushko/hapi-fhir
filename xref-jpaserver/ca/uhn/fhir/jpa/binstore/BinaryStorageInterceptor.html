91  	@Hook(Pointcut.STORAGE_PRESTORAGE_EXPUNGE_RESOURCE)
92  	public void expungeResource(AtomicInteger theCounter, IBaseResource theResource) {
93  
94  		List<? extends IBase> binaryElements = myCtx.newTerser().getAllPopulatedChildElementsOfType(theResource, myBinaryType);
95  
96  		List<String> attachmentIds = binaryElements
97  			.stream()
98  			.flatMap(t -> ((IBaseHasExtensions) t).getExtension().stream())
99  			.filter(t -> JpaConstants.EXT_EXTERNALIZED_BINARY_ID.equals(t.getUrl()))
100 			.map(t -> ((IPrimitiveType) t.getValue()).getValueAsString())
101 			.collect(Collectors.toList());
102 
103 		for (String next : attachmentIds) {
104 			myBinaryStorageSvc.expungeBlob(theResource.getIdElement(), next);
105 			theCounter.incrementAndGet();
106 
107 			ourLog.info("Deleting binary blob {} because resource {} is being expunged", next, theResource.getIdElement().getValue());
108 		}
109 
110 	}

123 	/**
124 	 * Don't allow clients to submit resources with binary storage attachments declared unless the ID was already in the
125 	 * resource. In other words, only HAPI itself may add a binary storage ID extension to a resource unless that
126 	 * extension was already present.
127 	 */
128 	private void blockIllegalExternalBinaryIds(IBaseResource thePreviousResource, IBaseResource theResource) {
129 		Set<String> existingBinaryIds = new HashSet<>();
130 		if (thePreviousResource != null) {
131 			List<? extends IPrimitiveType<byte[]>> base64fields = myCtx.newTerser().getAllPopulatedChildElementsOfType(thePreviousResource, myBinaryType);
132 			for (IPrimitiveType<byte[]> nextBase64 : base64fields) {
133 				if (nextBase64 instanceof IBaseHasExtensions) {
134 					((IBaseHasExtensions) nextBase64)
135 						.getExtension()
136 						.stream()
137 						.filter(t -> t.getUserData(JpaConstants.EXTENSION_EXT_SYSTEMDEFINED) == null)
138 						.filter(t -> EXT_EXTERNALIZED_BINARY_ID.equals(t.getUrl()))
139 						.map(t -> (IPrimitiveType) t.getValue())
140 						.map(t -> t.getValueAsString())
141 						.filter(t -> isNotBlank(t))
142 						.forEach(t -> existingBinaryIds.add(t));
143 				}
144 			}
145 		}
146 
147 		List<? extends IPrimitiveType<byte[]>> base64fields = myCtx.newTerser().getAllPopulatedChildElementsOfType(theResource, myBinaryType);
148 		for (IPrimitiveType<byte[]> nextBase64 : base64fields) {
149 			if (nextBase64 instanceof IBaseHasExtensions) {
150 				Optional<String> hasExternalizedBinaryReference = ((IBaseHasExtensions) nextBase64)
151 					.getExtension()
152 					.stream()
153 					.filter(t -> t.getUserData(JpaConstants.EXTENSION_EXT_SYSTEMDEFINED) == null)
154 					.filter(t -> t.getUrl().equals(EXT_EXTERNALIZED_BINARY_ID))
155 					.map(t->(IPrimitiveType) t.getValue())
156 					.map(t->t.getValueAsString())
157 					.filter(t->isNotBlank(t))
158 					.filter(t->{
159 						return !existingBinaryIds.contains(t);
160 					}).findFirst();
161 
162 				if (hasExternalizedBinaryReference.isPresent()) {
163 					String msg = myCtx.getLocalizer().getMessage(BaseHapiFhirDao.class, "externalizedBinaryStorageExtensionFoundInRequestBody", EXT_EXTERNALIZED_BINARY_ID, hasExternalizedBinaryReference.get());
164 					throw new InvalidRequestException(msg);
165 				}
166 			}
167 		}
168 
169 	}

171 	private void extractLargeBinaries(ServletRequestDetails theRequestDetails, IBaseResource theResource, Pointcut thePointcut) throws IOException {
172 		if (theRequestDetails == null) {
173 			// RequestDetails will only be null for internal HAPI events.  If externalization is required for them it will need to be done in a different way.
174 			return;
175 		}
176 		IIdType resourceId = theResource.getIdElement();
177 		if (!resourceId.hasResourceType() && resourceId.hasIdPart()) {
178 			String resourceType = myCtx.getResourceDefinition(theResource).getName();
179 			resourceId = new IdType(resourceType + "/" + resourceId.getIdPart());
180 		}
181 
182 		List<IBinaryTarget> attachments = recursivelyScanResourceForBinaryData(theResource);
183 		for (IBinaryTarget nextTarget : attachments) {
184 			byte[] data = nextTarget.getData();
185 			if (data != null && data.length > 0) {
186 
187 				long nextPayloadLength = data.length;
188 				String nextContentType = nextTarget.getContentType();
189 				boolean shouldStoreBlob = myBinaryStorageSvc.shouldStoreBlob(nextPayloadLength, resourceId, nextContentType);
190 				if (shouldStoreBlob) {
191 
192 					String newBlobId;
193 					if (resourceId.hasIdPart()) {
194 						ByteArrayInputStream inputStream = new ByteArrayInputStream(data);
195 						StoredDetails storedDetails = myBinaryStorageSvc.storeBlob(resourceId, null, nextContentType, inputStream);
196 						newBlobId = storedDetails.getBlobId();
197 					} else {
198 						assert thePointcut == Pointcut.STORAGE_PRESTORAGE_RESOURCE_CREATED : thePointcut.name();
199 						newBlobId = myBinaryStorageSvc.newBlobId();
200 						List<DeferredBinaryTarget> deferredBinaryTargets = getOrCreateDeferredBinaryStorageMap(theRequestDetails);
201 						DeferredBinaryTarget newDeferredBinaryTarget = new DeferredBinaryTarget(newBlobId, nextTarget, data);
202 						deferredBinaryTargets.add(newDeferredBinaryTarget);
203 					}
204 
205 					myBinaryAccessProvider.replaceDataWithExtension(nextTarget, newBlobId);
206 				}
207 			}
208 
209 		}
210 	}

212 	@Nonnull
213 	@SuppressWarnings("unchecked")
214 	private List<DeferredBinaryTarget> getOrCreateDeferredBinaryStorageMap(ServletRequestDetails theRequestDetails) {
215 		List<DeferredBinaryTarget> deferredBinaryTargets = (List<DeferredBinaryTarget>) theRequestDetails.getUserData().get(getDeferredListKey());
216 		if (deferredBinaryTargets == null) {
217 			deferredBinaryTargets = new ArrayList<>();
218 			theRequestDetails.getUserData().put(getDeferredListKey(), deferredBinaryTargets);
219 		}
220 		return deferredBinaryTargets;
221 	}

224 	@Hook(Pointcut.STORAGE_PRECOMMIT_RESOURCE_CREATED)
225 	public void storeLargeBinariesBeforeCreatePersistence(ServletRequestDetails theRequestDetails, IBaseResource theResource, Pointcut thePoincut) throws IOException {
226 		if (theRequestDetails == null) {
227 			return;
228 		}
229 		List<DeferredBinaryTarget> deferredBinaryTargets = (List<DeferredBinaryTarget>) theRequestDetails.getUserData().get(getDeferredListKey());
230 		if (deferredBinaryTargets != null) {
231 			IIdType resourceId = theResource.getIdElement();
232 			for (DeferredBinaryTarget next : deferredBinaryTargets) {
233 				String blobId = next.getBlobId();
234 				IBinaryTarget target = next.getBinaryTarget();
235 				InputStream dataStream = next.getDataStream();
236 				String contentType = target.getContentType();
237 				myBinaryStorageSvc.storeBlob(resourceId, blobId, contentType, dataStream);
238 			}
239 		}
240 	}

246 	@Hook(Pointcut.STORAGE_PRESHOW_RESOURCES)
247 	public void preShow(IPreResourceShowDetails theDetails) throws IOException {
248 		long unmarshalledByteCount = 0;
249 
250 		for (IBaseResource nextResource : theDetails) {
251 
252 			IIdType resourceId = nextResource.getIdElement();
253 			List<IBinaryTarget> attachments = recursivelyScanResourceForBinaryData(nextResource);
254 
255 			for (IBinaryTarget nextTarget : attachments) {
256 				Optional<String> attachmentId = nextTarget.getAttachmentId();
257 				if (attachmentId.isPresent()) {
258 
259 					StoredDetails blobDetails = myBinaryStorageSvc.fetchBlobDetails(resourceId, attachmentId.get());
260 					if (blobDetails == null) {
261 						String msg = myCtx.getLocalizer().getMessage(BinaryAccessProvider.class, "unknownBlobId");
262 						throw new InvalidRequestException(msg);
263 					}
264 
265 					if ((unmarshalledByteCount + blobDetails.getBytes()) < myAutoDeExternalizeMaximumBytes) {
266 
267 						byte[] bytes = myBinaryStorageSvc.fetchBlob(resourceId, attachmentId.get());
268 						nextTarget.setData(bytes);
269 						unmarshalledByteCount += blobDetails.getBytes();
270 					}
271 				}
272 			}
273 
274 		}
275 	}

277 	@Nonnull
278 	private List<IBinaryTarget> recursivelyScanResourceForBinaryData(IBaseResource theResource) {
279 		List<IBinaryTarget> binaryTargets = new ArrayList<>();
280 		myCtx.newTerser().visit(theResource, new IModelVisitor2() {
281 			@Override
282 			public boolean acceptElement(IBase theElement, List<IBase> theContainingElementPath, List<BaseRuntimeChildDefinition> theChildDefinitionPath, List<BaseRuntimeElementDefinition<?>> theElementDefinitionPath) {
283 
284 				if (theElement.getClass().equals(myBinaryType)) {
285 					IBase parent = theContainingElementPath.get(theContainingElementPath.size() - 2);
286 					Optional<IBinaryTarget> binaryTarget = myBinaryAccessProvider.toBinaryTarget(parent);
287 					if (binaryTarget.isPresent()) {
288 						binaryTargets.add(binaryTarget.get());
289 					}
290 				}
291 				return true;
292 			}
293 		});
294 		return binaryTargets;
295 	}

297 	private static class DeferredBinaryTarget {
298 		private final String myBlobId;
299 		private final IBinaryTarget myBinaryTarget;
300 		private final InputStream myDataStream;
301 
302 		private DeferredBinaryTarget(String theBlobId, IBinaryTarget theBinaryTarget, byte[] theData) {
303 			myBlobId = theBlobId;
304 			myBinaryTarget = theBinaryTarget;
305 			myDataStream = new ByteArrayInputStream(theData);
306 		}
307 
308 		String getBlobId() {
309 			return myBlobId;
310 		}
311 
312 		IBinaryTarget getBinaryTarget() {
313 			return myBinaryTarget;
314 		}
315 
316 		InputStream getDataStream() {
317 			return myDataStream;
318 		}
319 	}
