126 	@Override
127 	public UploadStatistics applyDeltaCodeSystemsAdd(String theSystem, CustomTerminologySet theAdditions) {
128 		ValidateUtil.isNotBlankOrThrowInvalidRequest(theSystem, "No system provided");
129 		validateDstu3OrNewer();
130 		theAdditions.validateNoCycleOrThrowInvalidRequest();
131 
132 		TermCodeSystem cs = myCodeSystemDao.findByCodeSystemUri(theSystem);
133 		if (cs == null) {
134 			CodeSystem codeSystemResource = new CodeSystem();
135 			codeSystemResource.setUrl(theSystem);
136 			codeSystemResource.setContent(CodeSystem.CodeSystemContentMode.NOTPRESENT);
137 			myTerminologyVersionAdapterSvc.createOrUpdateCodeSystem(codeSystemResource);
138 
139 			cs = myCodeSystemDao.findByCodeSystemUri(theSystem);
140 		}
141 
142 		TermCodeSystemVersion csv = cs.getCurrentVersion();
143 		Validate.notNull(csv);
144 
145 		CodeSystem codeSystem = myTerminologySvc.getCodeSystemFromContext(theSystem);
146 		if (codeSystem.getContent() != CodeSystem.CodeSystemContentMode.NOTPRESENT) {
147 			throw new InvalidRequestException("CodeSystem with url[" + Constants.codeSystemWithDefaultDescription(theSystem) + "] can not apply a delta - wrong content mode: " + codeSystem.getContent());
148 		}
149 
150 		Validate.notNull(cs);
151 		Validate.notNull(cs.getPid());
152 
153 		IIdType codeSystemId = cs.getResource().getIdDt();
154 
155 		// Load all concepts for the code system
156 		Map<String, Long> codeToConceptPid = new HashMap<>();
157 		{
158 			ourLog.info("Loading all concepts in CodeSystem versionPid[{}] and url[{}]", cs.getPid(), theSystem);
159 			StopWatch sw = new StopWatch();
160 			CriteriaBuilder criteriaBuilder = myEntityManager.getCriteriaBuilder();
161 			CriteriaQuery<TermConcept> query = criteriaBuilder.createQuery(TermConcept.class);
162 			Root<TermConcept> root = query.from(TermConcept.class);
163 			Predicate predicate = criteriaBuilder.equal(root.get("myCodeSystemVersionPid").as(Long.class), csv.getPid());
164 			query.where(predicate);
165 			TypedQuery<TermConcept> typedQuery = myEntityManager.createQuery(query.select(root));
166 			org.hibernate.query.Query<TermConcept> hibernateQuery = (org.hibernate.query.Query<TermConcept>) typedQuery;
167 			ScrollableResults scrollableResults = hibernateQuery.scroll(ScrollMode.FORWARD_ONLY);
168 			try (ScrollableResultsIterator<TermConcept> scrollableResultsIterator = new ScrollableResultsIterator<>(scrollableResults)) {
169 				while (scrollableResultsIterator.hasNext()) {
170 					TermConcept next = scrollableResultsIterator.next();
171 					codeToConceptPid.put(next.getCode(), next.getId());
172 				}
173 			}
174 			ourLog.info("Loaded {} concepts in {}", codeToConceptPid.size(), sw.toString());
175 		}
176 
177 		// Load all parent/child links
178 		ListMultimap<String, String> parentCodeToChildCodes = ArrayListMultimap.create();
179 		ListMultimap<String, String> childCodeToParentCodes = ArrayListMultimap.create();
180 		{
181 			ourLog.info("Loading all parent/child relationships in CodeSystem url[" + theSystem + "]");
182 			int count = 0;
183 			StopWatch sw = new StopWatch();
184 			CriteriaBuilder criteriaBuilder = myEntityManager.getCriteriaBuilder();
185 			CriteriaQuery<TermConceptParentChildLink> query = criteriaBuilder.createQuery(TermConceptParentChildLink.class);
186 			Root<TermConceptParentChildLink> root = query.from(TermConceptParentChildLink.class);
187 			Predicate predicate = criteriaBuilder.equal(root.get("myCodeSystemVersionPid").as(Long.class), csv.getPid());
188 			root.fetch("myChild");
189 			root.fetch("myParent");
190 			query.where(predicate);
191 			TypedQuery<TermConceptParentChildLink> typedQuery = myEntityManager.createQuery(query.select(root));
192 			org.hibernate.query.Query<TermConceptParentChildLink> hibernateQuery = (org.hibernate.query.Query<TermConceptParentChildLink>) typedQuery;
193 			ScrollableResults scrollableResults = hibernateQuery.scroll(ScrollMode.FORWARD_ONLY);
194 			try (ScrollableResultsIterator<TermConceptParentChildLink> scrollableResultsIterator = new ScrollableResultsIterator<>(scrollableResults)) {
195 				while (scrollableResultsIterator.hasNext()) {
196 					TermConceptParentChildLink next = scrollableResultsIterator.next();
197 					String parentCode = next.getParent().getCode();
198 					String childCode = next.getChild().getCode();
199 					parentCodeToChildCodes.put(parentCode, childCode);
200 					childCodeToParentCodes.put(childCode, parentCode);
201 					count++;
202 				}
203 			}
204 			ourLog.info("Loaded {} parent/child relationships in {}", count, sw.toString());
205 		}
206 
207 		// Account for root codes in the parent->child map
208 		for (String nextCode : codeToConceptPid.keySet()) {
209 			if (childCodeToParentCodes.get(nextCode).isEmpty()) {
210 				parentCodeToChildCodes.put("", nextCode);
211 			}
212 		}
213 
214 		UploadStatistics retVal = new UploadStatistics(codeSystemId);
215 
216 		// Add root concepts
217 		for (TermConcept nextRootConcept : theAdditions.getRootConcepts()) {
218 			List<String> parentCodes = Collections.emptyList();
219 			addConcept(csv, codeToConceptPid, parentCodes, nextRootConcept, parentCodeToChildCodes, retVal, true);
220 		}
221 
222 		// Add unanchored child concepts
223 		for (TermConcept nextUnanchoredChild : theAdditions.getUnanchoredChildConceptsToParentCodes().keySet()) {
224 			List<String> nextParentCodes = theAdditions.getUnanchoredChildConceptsToParentCodes().get(nextUnanchoredChild);
225 			addConcept(csv, codeToConceptPid, nextParentCodes, nextUnanchoredChild, parentCodeToChildCodes, retVal, true);
226 		}
227 
228 		return retVal;
229 	}

233 	public UploadStatistics applyDeltaCodeSystemsRemove(String theSystem, CustomTerminologySet theValue) {
234 		ValidateUtil.isNotBlankOrThrowInvalidRequest(theSystem, "No system provided");
235 		validateDstu3OrNewer();
236 
237 		TermCodeSystem cs = myCodeSystemDao.findByCodeSystemUri(theSystem);
238 		if (cs == null) {
239 			throw new InvalidRequestException("Unknown code system: " + theSystem);
240 		}
241 
242 		AtomicInteger removeCounter = new AtomicInteger(0);
243 
244 		for (TermConcept nextSuppliedConcept : theValue.getRootConcepts()) {
245 			Optional<TermConcept> conceptOpt = myTerminologySvc.findCode(theSystem, nextSuppliedConcept.getCode());
246 			if (conceptOpt.isPresent()) {
247 				TermConcept concept = conceptOpt.get();
248 				deleteConceptChildrenAndConcept(concept, removeCounter);
249 			}
250 		}
251 
252 		IIdType target = cs.getResource().getIdDt();
253 		return new UploadStatistics(removeCounter.get(), target);
254 	}

279 	@Override
280 	public int saveConcept(TermConcept theConcept) {
281 		int retVal = 0;
282 
283 		/*
284 		 * If the concept has an ID, we're reindexing, so there's no need to
285 		 * save parent concepts first (it's way too slow to do that)
286 		 */
287 		if (theConcept.getId() == null) {
288 			retVal += ensureParentsSaved(theConcept.getParents());
289 		}
290 
291 		if (theConcept.getId() == null || theConcept.getIndexStatus() == null) {
292 			retVal++;
293 			theConcept.setIndexStatus(BaseHapiFhirDao.INDEX_STATUS_INDEXED);
294 			theConcept.setUpdated(new Date());
295 			myConceptDao.save(theConcept);
296 
297 			for (TermConceptProperty next : theConcept.getProperties()) {
298 				myConceptPropertyDao.save(next);
299 			}
300 
301 			for (TermConceptDesignation next : theConcept.getDesignations()) {
302 				myConceptDesignationDao.save(next);
303 			}
304 		}
305 
306 		ourLog.trace("Saved {} and got PID {}", theConcept.getCode(), theConcept.getId());
307 		return retVal;
308 	}

310 	@Override
311 	@Transactional(propagation = Propagation.MANDATORY)
312 	public void storeNewCodeSystemVersionIfNeeded(CodeSystem theCodeSystem, ResourceTable theResourceEntity) {
313 		if (theCodeSystem != null && isNotBlank(theCodeSystem.getUrl())) {
314 			String codeSystemUrl = theCodeSystem.getUrl();
315 			if (theCodeSystem.getContent() == CodeSystem.CodeSystemContentMode.COMPLETE || theCodeSystem.getContent() == null || theCodeSystem.getContent() == CodeSystem.CodeSystemContentMode.NOTPRESENT) {
316 				ourLog.info("CodeSystem {} has a status of {}, going to store concepts in terminology tables", theResourceEntity.getIdDt().getValue(), theCodeSystem.getContentElement().getValueAsString());
317 
318 				Long codeSystemResourcePid = getCodeSystemResourcePid(theCodeSystem.getIdElement());
319 
320 				/*
321 				 * If this is a not-present codesystem, we don't want to store a new version if one
322 				 * already exists, since that will wipe out the existing concepts. We do create or update
323 				 * the TermCodeSystem table though, since that allows the DB to reject changes
324 				 * that would result in duplicate CodeSysten.url values.
325 				 */
326 				if (theCodeSystem.getContent() == CodeSystem.CodeSystemContentMode.NOTPRESENT) {
327 					TermCodeSystem codeSystem = myCodeSystemDao.findByCodeSystemUri(theCodeSystem.getUrl());
328 					if (codeSystem != null) {
329 						getOrCreateTermCodeSystem(codeSystemResourcePid, theCodeSystem.getUrl(), theCodeSystem.getUrl(), theResourceEntity);
330 						return;
331 					}
332 				}
333 
334 				TermCodeSystemVersion persCs = new TermCodeSystemVersion();
335 
336 				populateCodeSystemVersionProperties(persCs, theCodeSystem, theResourceEntity);
337 
338 				persCs.getConcepts().addAll(BaseTermReadSvcImpl.toPersistedConcepts(theCodeSystem.getConcept(), persCs));
339 				ourLog.debug("Code system has {} concepts", persCs.getConcepts().size());
340 				storeNewCodeSystemVersion(codeSystemResourcePid, codeSystemUrl, theCodeSystem.getName(), theCodeSystem.getVersion(), persCs, theResourceEntity);
341 			}
342 
343 		}
344 	}

367 	@Override
368 	@Transactional(propagation = Propagation.REQUIRED)
369 	public void storeNewCodeSystemVersion(Long theCodeSystemResourcePid, String theSystemUri, String theSystemName, String theSystemVersionId, TermCodeSystemVersion theCodeSystemVersion, ResourceTable theCodeSystemResourceTable) {
370 		ourLog.debug("Storing code system");
371 
372 		ValidateUtil.isTrueOrThrowInvalidRequest(theCodeSystemVersion.getResource() != null, "No resource supplied");
373 		ValidateUtil.isNotBlankOrThrowInvalidRequest(theSystemUri, "No system URI supplied");
374 
375 		// Grab the existing versions so we can delete them later
376 		List<TermCodeSystemVersion> existing = myCodeSystemVersionDao.findByCodeSystemResourcePid(theCodeSystemResourcePid);
377 
378 		/*
379 		 * For now we always delete old versions. At some point it would be nice to allow configuration to keep old versions.
380 		 */
381 
382 		for (TermCodeSystemVersion next : existing) {
383 			ourLog.info("Deleting old code system version {}", next.getPid());
384 			Long codeSystemVersionPid = next.getPid();
385 			deleteCodeSystemVersion(codeSystemVersionPid);
386 		}
387 
388 		ourLog.debug("Flushing...");
389 		myConceptDao.flush();
390 		ourLog.debug("Done flushing");
391 
392 		/*
393 		 * Do the upload
394 		 */
395 
396 		TermCodeSystem codeSystem = getOrCreateTermCodeSystem(theCodeSystemResourcePid, theSystemUri, theSystemName, theCodeSystemResourceTable);
397 
398 		theCodeSystemVersion.setCodeSystem(codeSystem);
399 
400 		theCodeSystemVersion.setCodeSystemDisplayName(theSystemName);
401 		theCodeSystemVersion.setCodeSystemVersionId(theSystemVersionId);
402 
403 		ourLog.debug("Validating all codes in CodeSystem for storage (this can take some time for large sets)");
404 
405 		// Validate the code system
406 		ArrayList<String> conceptsStack = new ArrayList<>();
407 		IdentityHashMap<TermConcept, Object> allConcepts = new IdentityHashMap<>();
408 		int totalCodeCount = 0;
409 		for (TermConcept next : theCodeSystemVersion.getConcepts()) {
410 			totalCodeCount += validateConceptForStorage(next, theCodeSystemVersion, conceptsStack, allConcepts);
411 		}
412 
413 		ourLog.debug("Saving version containing {} concepts", totalCodeCount);
414 
415 		TermCodeSystemVersion codeSystemVersion = myCodeSystemVersionDao.saveAndFlush(theCodeSystemVersion);
416 
417 		ourLog.debug("Saving code system");
418 
419 		codeSystem.setCurrentVersion(theCodeSystemVersion);
420 		codeSystem = myCodeSystemDao.saveAndFlush(codeSystem);
421 
422 		ourLog.debug("Setting CodeSystemVersion[{}] on {} concepts...", codeSystem.getPid(), totalCodeCount);
423 
424 		for (TermConcept next : theCodeSystemVersion.getConcepts()) {
425 			populateVersion(next, codeSystemVersion);
426 		}
427 
428 		ourLog.debug("Saving {} concepts...", totalCodeCount);
429 
430 		IdentityHashMap<TermConcept, Object> conceptsStack2 = new IdentityHashMap<>();
431 		for (TermConcept next : theCodeSystemVersion.getConcepts()) {
432 			persistChildren(next, codeSystemVersion, conceptsStack2, totalCodeCount);
433 		}
434 
435 		ourLog.debug("Done saving concepts, flushing to database");
436 
437 		myConceptDao.flush();
438 		myConceptParentChildLinkDao.flush();
439 
440 		if (myDeferredStorageSvc.isStorageQueueEmpty() == false) {
441 			ourLog.info("Note that some concept saving has been deferred");
442 		}
443 	}

445 	private void deleteCodeSystemVersion(final Long theCodeSystemVersionPid) {
446 		ourLog.info(" * Deleting code system version {}", theCodeSystemVersionPid);
447 
448 		PageRequest page1000 = PageRequest.of(0, 1000);
449 
450 		// Parent/Child links
451 		{
452 			String descriptor = "parent/child links";
453 			Supplier<Slice<Long>> loader = () -> myConceptParentChildLinkDao.findIdsByCodeSystemVersion(page1000, theCodeSystemVersionPid);
454 			Supplier<Integer> counter = () -> myConceptParentChildLinkDao.countByCodeSystemVersion(theCodeSystemVersionPid);
455 			doDelete(descriptor, loader, counter, myConceptParentChildLinkDao);
456 		}
457 
458 		// Properties
459 		{
460 			String descriptor = "concept properties";
461 			Supplier<Slice<Long>> loader = () -> myConceptPropertyDao.findIdsByCodeSystemVersion(page1000, theCodeSystemVersionPid);
462 			Supplier<Integer> counter = () -> myConceptPropertyDao.countByCodeSystemVersion(theCodeSystemVersionPid);
463 			doDelete(descriptor, loader, counter, myConceptPropertyDao);
464 		}
465 
466 		// Designations
467 		{
468 			String descriptor = "concept designations";
469 			Supplier<Slice<Long>> loader = () -> myConceptDesignationDao.findIdsByCodeSystemVersion(page1000, theCodeSystemVersionPid);
470 			Supplier<Integer> counter = () -> myConceptDesignationDao.countByCodeSystemVersion(theCodeSystemVersionPid);
471 			doDelete(descriptor, loader, counter, myConceptDesignationDao);
472 		}
473 
474 		// Concepts
475 		{
476 			String descriptor = "concepts";
477 			// For some reason, concepts are much slower to delete, so use a smaller batch size
478 			PageRequest page100 = PageRequest.of(0, 100);
479 			Supplier<Slice<Long>> loader = () -> myConceptDao.findIdsByCodeSystemVersion(page100, theCodeSystemVersionPid);
480 			Supplier<Integer> counter = () -> myConceptDao.countByCodeSystemVersion(theCodeSystemVersionPid);
481 			doDelete(descriptor, loader, counter, myConceptDao);
482 		}
483 
484 		Optional<TermCodeSystem> codeSystemOpt = myCodeSystemDao.findWithCodeSystemVersionAsCurrentVersion(theCodeSystemVersionPid);
485 		if (codeSystemOpt.isPresent()) {
486 			TermCodeSystem codeSystem = codeSystemOpt.get();
487 			ourLog.info(" * Removing code system version {} as current version of code system {}", theCodeSystemVersionPid, codeSystem.getPid());
488 			codeSystem.setCurrentVersion(null);
489 			myCodeSystemDao.save(codeSystem);
490 		}
491 
492 		ourLog.info(" * Deleting code system version");
493 		myCodeSystemVersionDao.deleteById(theCodeSystemVersionPid);
494 
495 	}

501 	private void addConcept(TermCodeSystemVersion theCsv, Map<String, Long> theCodeToConceptPid, Collection<String> theParentCodes, TermConcept theConceptToAdd, ListMultimap<String, String> theParentCodeToChildCodes, UploadStatistics theStatisticsTracker, boolean theForceResequence) {
502 		TermConcept nextConceptToAdd = theConceptToAdd;
503 
504 		String nextCodeToAdd = nextConceptToAdd.getCode();
505 		String parentDescription = "(root concept)";
506 		Set<TermConcept> parentConcepts = new HashSet<>();
507 		if (!theParentCodes.isEmpty()) {
508 			parentDescription = "[" + String.join(", ", theParentCodes) + "]";
509 			for (String nextParentCode : theParentCodes) {
510 				Long nextParentCodePid = theCodeToConceptPid.get(nextParentCode);
511 				if (nextParentCodePid == null) {
512 					throw new InvalidRequestException("Unable to add code \"" + nextCodeToAdd + "\" to unknown parent: " + nextParentCode);
513 				}
514 				parentConcepts.add(myConceptDao.getOne(nextParentCodePid));
515 			}
516 		}
517 
518 		ourLog.info("Saving concept {} with parent {}", theStatisticsTracker.getUpdatedConceptCount(), parentDescription);
519 
520 		if (theCodeToConceptPid.containsKey(nextCodeToAdd)) {
521 
522 			TermConcept existingCode = myConceptDao.getOne(theCodeToConceptPid.get(nextCodeToAdd));
523 			existingCode.setIndexStatus(null);
524 			existingCode.setDisplay(nextConceptToAdd.getDisplay());
525 			nextConceptToAdd = existingCode;
526 
527 		}
528 
529 		if (theConceptToAdd.getSequence() == null || theForceResequence) {
530 			// If this is a new code, give it a sequence number based on how many concepts the
531 			// parent already has (or the highest number, if the code has multiple parents)
532 			int sequence = 0;
533 			for (String nextParentCode : theParentCodes) {
534 				theParentCodeToChildCodes.put(nextParentCode, nextCodeToAdd);
535 				sequence = Math.max(sequence, theParentCodeToChildCodes.get(nextParentCode).size());
536 			}
537 			if (theParentCodes.isEmpty()) {
538 				theParentCodeToChildCodes.put("", nextCodeToAdd);
539 				sequence = Math.max(sequence, theParentCodeToChildCodes.get("").size());
540 			}
541 			nextConceptToAdd.setSequence(sequence);
542 		}
543 
544 
545 		// Drop any old parent-child links if they aren't explicitly specified in the
546 		// hierarchy being added
547 		for (Iterator<TermConceptParentChildLink> iter = nextConceptToAdd.getParents().iterator(); iter.hasNext(); ) {
548 			TermConceptParentChildLink nextLink = iter.next();
549 			String parentCode = nextLink.getParent().getCode();
550 			boolean shouldRemove = !theParentCodes.contains(parentCode);
551 			if (shouldRemove) {
552 				ourLog.info("Dropping existing parent/child link from {} -> {}", parentCode, nextCodeToAdd);
553 				myConceptParentChildLinkDao.delete(nextLink);
554 				iter.remove();
555 
556 				List<TermConceptParentChildLink> parentChildrenList = nextLink.getParent().getChildren();
557 				parentChildrenList.remove(nextLink);
558 			}
559 		}
560 
561 		nextConceptToAdd.setParentPids(null);
562 		nextConceptToAdd.setCodeSystemVersion(theCsv);
563 		nextConceptToAdd = myConceptDao.save(nextConceptToAdd);
564 
565 		Long nextConceptPid = nextConceptToAdd.getId();
566 		Validate.notNull(nextConceptPid);
567 		theCodeToConceptPid.put(nextCodeToAdd, nextConceptPid);
568 		theStatisticsTracker.incrementUpdatedConceptCount();
569 
570 		// Add link to new child to the parent if this link doesn't already exist (this will be the
571 		// case for concepts being added to an existing child concept, but won't be the case when
572 		// we're recursively adding children)
573 		for (TermConcept nextParentConcept : parentConcepts) {
574 			if (nextParentConcept.getChildren().stream().noneMatch(t -> t.getChild().getCode().equals(nextCodeToAdd))) {
575 				TermConceptParentChildLink parentLink = new TermConceptParentChildLink();
576 				parentLink.setParent(nextParentConcept);
577 				parentLink.setChild(nextConceptToAdd);
578 				parentLink.setCodeSystem(theCsv);
579 				parentLink.setRelationshipType(TermConceptParentChildLink.RelationshipTypeEnum.ISA);
580 				nextParentConcept.getChildren().add(parentLink);
581 				nextConceptToAdd.getParents().add(parentLink);
582 				myConceptParentChildLinkDao.save(parentLink);
583 			}
584 		}
585 
586 		// Save children recursively
587 		for (TermConceptParentChildLink nextChildConceptLink : nextConceptToAdd.getChildren()) {
588 
589 			TermConcept nextChild = nextChildConceptLink.getChild();
590 			Collection<String> parentCodes = nextChild.getParents().stream().map(t -> t.getParent().getCode()).collect(Collectors.toList());
591 			addConcept(theCsv, theCodeToConceptPid, parentCodes, nextChild, theParentCodeToChildCodes, theStatisticsTracker, false);
592 
593 			if (nextChildConceptLink.getId() == null) {
594 				nextChildConceptLink.setCodeSystem(theCsv);
595 				myConceptParentChildLinkDao.save(nextChildConceptLink);
596 			}
597 		}
598 
599 	}

609 	private void persistChildren(TermConcept theConcept, TermCodeSystemVersion theCodeSystem, IdentityHashMap<TermConcept, Object> theConceptsStack, int theTotalConcepts) {
610 		if (theConceptsStack.put(theConcept, PLACEHOLDER_OBJECT) != null) {
611 			return;
612 		}
613 
614 		if ((theConceptsStack.size() + 1) % 10000 == 0) {
615 			float pct = (float) theConceptsStack.size() / (float) theTotalConcepts;
616 			ourLog.info("Have processed {}/{} concepts ({}%)", theConceptsStack.size(), theTotalConcepts, (int) (pct * 100.0f));
617 		}
618 
619 		theConcept.setCodeSystemVersion(theCodeSystem);
620 		theConcept.setIndexStatus(BaseHapiFhirDao.INDEX_STATUS_INDEXED);
621 
622 		if (theConceptsStack.size() <= myDaoConfig.getDeferIndexingForCodesystemsOfSize()) {
623 			saveConcept(theConcept);
624 		} else {
625 			myDeferredStorageSvc.addConceptToStorageQueue(theConcept);
626 		}
627 
628 		for (TermConceptParentChildLink next : theConcept.getChildren()) {
629 			persistChildren(next.getChild(), theCodeSystem, theConceptsStack, theTotalConcepts);
630 		}
631 
632 		for (TermConceptParentChildLink next : theConcept.getChildren()) {
633 			if (theConceptsStack.size() <= myDaoConfig.getDeferIndexingForCodesystemsOfSize()) {
634 				saveConceptLink(next);
635 			} else {
636 				myDeferredStorageSvc.addConceptLinkToStorageQueue(next);
637 			}
638 		}
639 
640 	}

658 	private int ensureParentsSaved(Collection<TermConceptParentChildLink> theParents) {
659 		ourLog.trace("Checking {} parents", theParents.size());
660 		int retVal = 0;
661 
662 		for (TermConceptParentChildLink nextLink : theParents) {
663 			if (nextLink.getRelationshipType() == TermConceptParentChildLink.RelationshipTypeEnum.ISA) {
664 				TermConcept nextParent = nextLink.getParent();
665 				retVal += ensureParentsSaved(nextParent.getParents());
666 				if (nextParent.getId() == null) {
667 					nextParent.setUpdated(new Date());
668 					myConceptDao.saveAndFlush(nextParent);
669 					retVal++;
670 					ourLog.debug("Saved parent code {} and got id {}", nextParent.getCode(), nextParent.getId());
671 				}
672 			}
673 		}
674 
675 		return retVal;
676 	}

720 	private <T> void doDelete(String theDescriptor, Supplier<Slice<Long>> theLoader, Supplier<Integer> theCounter, JpaRepository<T, Long> theDao) {
721 		int count;
722 		ourLog.info(" * Deleting {}", theDescriptor);
723 		int totalCount = theCounter.get();
724 		StopWatch sw = new StopWatch();
725 		count = 0;
726 		while (true) {
727 			Slice<Long> link = theLoader.get();
728 			if (!link.hasContent()) {
729 				break;
730 			}
731 
732 			TransactionTemplate txTemplate = new TransactionTemplate(myTransactionManager);
733 			txTemplate.setPropagationBehavior(TransactionDefinition.PROPAGATION_REQUIRED);
734 			txTemplate.execute(t -> {
735 				link.forEach(id -> theDao.deleteById(id));
736 				return null;
737 			});
738 
739 			count += link.getNumberOfElements();
740 			ourLog.info(" * {} {} deleted - {}/sec - ETA: {}", count, theDescriptor, sw.formatThroughput(count, TimeUnit.SECONDS), sw.getEstimatedTimeRemaining(count, totalCount));
741 		}
742 		theDao.flush();
743 	}

746 	private int validateConceptForStorage(TermConcept theConcept, TermCodeSystemVersion theCodeSystem, ArrayList<String> theConceptsStack,
747 													  IdentityHashMap<TermConcept, Object> theAllConcepts) {
748 		ValidateUtil.isTrueOrThrowInvalidRequest(theConcept.getCodeSystemVersion() != null, "CodeSystemVersion is null");
749 		ValidateUtil.isTrueOrThrowInvalidRequest(theConcept.getCodeSystemVersion() == theCodeSystem, "CodeSystems are not equal");
750 		ValidateUtil.isNotBlankOrThrowInvalidRequest(theConcept.getCode(), "CodeSystem contains a code with no code value");
751 
752 		if (theConceptsStack.contains(theConcept.getCode())) {
753 			throw new InvalidRequestException("CodeSystem contains circular reference around code " + theConcept.getCode());
754 		}
755 		theConceptsStack.add(theConcept.getCode());
756 
757 		int retVal = 0;
758 		if (theAllConcepts.put(theConcept, theAllConcepts) == null) {
759 			if (theAllConcepts.size() % 1000 == 0) {
760 				ourLog.info("Have validated {} concepts", theAllConcepts.size());
761 			}
762 			retVal = 1;
763 		}
764 
765 		for (TermConceptParentChildLink next : theConcept.getChildren()) {
766 			next.setCodeSystem(theCodeSystem);
767 			retVal += validateConceptForStorage(next.getChild(), theCodeSystem, theConceptsStack, theAllConcepts);
768 		}
769 
770 		theConceptsStack.remove(theConceptsStack.size() - 1);
771 
772 		return retVal;
773 	}
