25  import ca.uhn.fhir.rest.server.exceptions.InternalErrorException;
26  import ca.uhn.fhir.util.UrlUtil;
27  import org.hl7.fhir.convertors.VersionConvertor_30_40;
28  import org.hl7.fhir.dstu3.model.CodeSystem;
29  import org.hl7.fhir.dstu3.model.ConceptMap;
30  import org.hl7.fhir.dstu3.model.ValueSet;

32  import org.hl7.fhir.instance.model.api.IIdType;

34  import org.springframework.context.ApplicationContext;

36  import org.springframework.context.event.ContextStartedEvent;
37  import org.springframework.context.event.EventListener;
38  
39  import static org.apache.commons.lang3.StringUtils.isBlank;
40  
41  public class TermVersionAdapterSvcDstu3 extends BaseTermVersionAdapterSvcImpl implements ITermVersionAdapterSvc {
42  
43  	private IFhirResourceDao<ConceptMap> myConceptMapResourceDao;
44  	private IFhirResourceDao<CodeSystem> myCodeSystemResourceDao;
45  	private IFhirResourceDao<ValueSet> myValueSetResourceDao;
46  
47  	@Autowired
48  	private ApplicationContext myAppCtx;
49  
50  	public TermVersionAdapterSvcDstu3() {
51  		super();
52  	}
53  
54  
55  	/**
56  	 * Initialize the beans that are used by this service.
57  	 *
58  	 * Note: There is a circular dependency here where the CodeSystem DAO
59  	 * needs terminology services, and the term services need the CodeSystem DAO.
60  	 * So we look these up in a refresh event instead of just autowiring them
61  	 * in order to avoid weird circular reference errors.
62  	 */
63  	@SuppressWarnings({"unchecked", "unused"})
64  	@EventListener
65  	public void start(ContextRefreshedEvent theEvent) {
66  		myCodeSystemResourceDao = (IFhirResourceDao<CodeSystem>) myAppCtx.getBean("myCodeSystemDaoDstu3");
67  		myValueSetResourceDao = (IFhirResourceDao<ValueSet>) myAppCtx.getBean("myValueSetDaoDstu3");
68  		myConceptMapResourceDao = (IFhirResourceDao<ConceptMap>) myAppCtx.getBean("myConceptMapDaoDstu3");
69  	}
70  
71  		@Override
72  	public IIdType createOrUpdateCodeSystem(org.hl7.fhir.r4.model.CodeSystem theCodeSystemResource) {
73  		CodeSystem resourceToStore;
74  		try {
75  			resourceToStore = VersionConvertor_30_40.convertCodeSystem(theCodeSystemResource);
76  		} catch (FHIRException e) {
77  			throw new InternalErrorException(e);
78  		}
79  		validateCodeSystemForStorage(theCodeSystemResource);
80  		if (isBlank(resourceToStore.getIdElement().getIdPart())) {
81  			String matchUrl = "CodeSystem?url=" + UrlUtil.escapeUrlParam(theCodeSystemResource.getUrl());
82  			return myCodeSystemResourceDao.update(resourceToStore, matchUrl).getId();
83  		} else {
84  			return myCodeSystemResourceDao.update(resourceToStore).getId();
85  		}
86  	}
87  
88  	@Override
89  	public void createOrUpdateConceptMap(org.hl7.fhir.r4.model.ConceptMap theConceptMap) {
90  		ConceptMap resourceToStore;
91  		try {
92  			resourceToStore = VersionConvertor_30_40.convertConceptMap(theConceptMap);
93  		} catch (FHIRException e) {
94  			throw new InternalErrorException(e);
95  		}
96  		if (isBlank(resourceToStore.getIdElement().getIdPart())) {
97  			String matchUrl = "ConceptMap?url=" + UrlUtil.escapeUrlParam(theConceptMap.getUrl());
98  			myConceptMapResourceDao.update(resourceToStore, matchUrl);
99  		} else {
100 			myConceptMapResourceDao.update(resourceToStore);
101 		}
102 	}
103 
104 	@Override
105 	public void createOrUpdateValueSet(org.hl7.fhir.r4.model.ValueSet theValueSet) {
106 		ValueSet valueSetDstu3;
107 		try {
108 			valueSetDstu3 = VersionConvertor_30_40.convertValueSet(theValueSet);
109 		} catch (FHIRException e) {
110 			throw new InternalErrorException(e);
111 		}
112 
113 		if (isBlank(valueSetDstu3.getIdElement().getIdPart())) {
114 			String matchUrl = "ValueSet?url=" + UrlUtil.escapeUrlParam(theValueSet.getUrl());
115 			myValueSetResourceDao.update(valueSetDstu3, matchUrl);
116 		} else {
117 			myValueSetResourceDao.update(valueSetDstu3);
118 		}
119 	}
120 
121 }
