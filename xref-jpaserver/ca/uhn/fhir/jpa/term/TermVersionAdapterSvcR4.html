23  import ca.uhn.fhir.jpa.dao.IFhirResourceDao;
24  import ca.uhn.fhir.jpa.term.api.ITermVersionAdapterSvc;
25  import ca.uhn.fhir.util.UrlUtil;
26  import org.hl7.fhir.instance.model.api.IIdType;
27  import org.hl7.fhir.r4.model.CodeSystem;
28  import org.hl7.fhir.r4.model.ConceptMap;
29  import org.hl7.fhir.r4.model.ValueSet;
30  import org.springframework.beans.factory.annotation.Autowired;
31  import org.springframework.beans.factory.annotation.Qualifier;
32  import org.springframework.context.ApplicationContext;
33  import org.springframework.context.event.ContextRefreshedEvent;
34  import org.springframework.context.event.EventListener;
35  
36  import javax.annotation.PostConstruct;
37  
38  import static org.apache.commons.lang3.StringUtils.isBlank;
39  
40  public class TermVersionAdapterSvcR4 extends BaseTermVersionAdapterSvcImpl implements ITermVersionAdapterSvc {
41  	private IFhirResourceDao<ConceptMap> myConceptMapResourceDao;
42  	private IFhirResourceDao<CodeSystem> myCodeSystemResourceDao;
43  	private IFhirResourceDao<ValueSet> myValueSetResourceDao;
44  
45  	@Autowired
46  	private ApplicationContext myAppCtx;
47  
48  	/**
49  	 * Initialize the beans that are used by this service.
50  	 *
51  	 * Note: There is a circular dependency here where the CodeSystem DAO
52  	 * needs terminology services, and the term services need the CodeSystem DAO.
53  	 * So we look these up in a refresh event instead of just autowiring them
54  	 * in order to avoid weird circular reference errors.
55  	 */
56  	@SuppressWarnings({"unchecked", "unused"})
57  	@EventListener
58  	public void start(ContextRefreshedEvent theEvent) {
59  		myCodeSystemResourceDao = (IFhirResourceDao<CodeSystem>) myAppCtx.getBean("myCodeSystemDaoR4");
60  		myValueSetResourceDao = (IFhirResourceDao<ValueSet>) myAppCtx.getBean("myValueSetDaoR4");
61  		myConceptMapResourceDao = (IFhirResourceDao<ConceptMap>) myAppCtx.getBean("myConceptMapDaoR4");
62  	}
63  
64  	@Override
65  	public IIdType createOrUpdateCodeSystem(org.hl7.fhir.r4.model.CodeSystem theCodeSystemResource) {
66  		validateCodeSystemForStorage(theCodeSystemResource);
67  		if (isBlank(theCodeSystemResource.getIdElement().getIdPart())) {
68  			String matchUrl = "CodeSystem?url=" + UrlUtil.escapeUrlParam(theCodeSystemResource.getUrl());
69  			return myCodeSystemResourceDao.update(theCodeSystemResource, matchUrl).getId();
70  		} else {
71  			return myCodeSystemResourceDao.update(theCodeSystemResource).getId();
72  		}
73  	}
74  
75  	@Override
76  	public void createOrUpdateConceptMap(org.hl7.fhir.r4.model.ConceptMap theConceptMap) {
77  		if (isBlank(theConceptMap.getIdElement().getIdPart())) {
78  			String matchUrl = "ConceptMap?url=" + UrlUtil.escapeUrlParam(theConceptMap.getUrl());
79  			myConceptMapResourceDao.update(theConceptMap, matchUrl);
80  		} else {
81  			myConceptMapResourceDao.update(theConceptMap);
82  		}
83  	}
84  
85  	@Override
86  	public void createOrUpdateValueSet(org.hl7.fhir.r4.model.ValueSet theValueSet) {
87  		if (isBlank(theValueSet.getIdElement().getIdPart())) {
88  			String matchUrl = "ValueSet?url=" + UrlUtil.escapeUrlParam(theValueSet.getUrl());
89  			myValueSetResourceDao.update(theValueSet, matchUrl);
90  		} else {
91  			myValueSetResourceDao.update(theValueSet);
92  		}
93  	}
94  
95  }
