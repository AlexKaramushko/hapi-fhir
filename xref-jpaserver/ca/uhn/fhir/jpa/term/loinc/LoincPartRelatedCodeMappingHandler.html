63  	public LoincPartRelatedCodeMappingHandler(Map<String, TermConcept> theCode2concept, List<ValueSet> theValueSets, List<ConceptMap> theConceptMaps, Properties theUploadProperties) {
64  		super(theCode2concept, theValueSets, theConceptMaps, theUploadProperties);
65  	}
66  
67  	@Override
68  	public void accept(CSVRecord theRecord) {
69  
70  		String partNumber = trim(theRecord.get("PartNumber"));
71  		String partName = trim(theRecord.get("PartName"));
72  		String partTypeName = trim(theRecord.get("PartTypeName"));
73  		String extCodeId = trim(theRecord.get("ExtCodeId"));
74  		// TODO: use hex code for ascii 160
75  		extCodeId = extCodeId.replace(" ", "");
76  		String extCodeDisplayName = trim(theRecord.get("ExtCodeDisplayName"));
77  		String extCodeSystem = trim(theRecord.get("ExtCodeSystem"));
78  		String mapType = trim(theRecord.get("MapType"));
79  		String contentOrigin = trim(theRecord.get("ContentOrigin"));
80  		String extCodeSystemVersion = trim(theRecord.get("ExtCodeSystemVersion"));
81  		String extCodeSystemCopyrightNotice = trim(theRecord.get("ExtCodeSystemCopyrightNotice"));
82  
83  		Enumerations.ConceptMapEquivalence equivalence;
84  		switch (trim(defaultString(mapType))) {
85  			case "":
86  			case "Exact":
87  				// 'equal' is more exact than 'equivalent' in the equivalence codes
88  				equivalence = Enumerations.ConceptMapEquivalence.EQUAL;
89  				break;
90  			case "LOINC broader":
91  				equivalence = Enumerations.ConceptMapEquivalence.NARROWER;
92  				break;
93  			case "LOINC narrower":
94  				equivalence = Enumerations.ConceptMapEquivalence.WIDER;
95  				break;
96  			default:
97  				throw new InternalErrorException("Unknown MapType '" + mapType + "' for PartNumber: " + partNumber);
98  		}
99  
100 		String loincPartMapId;
101 		String loincPartMapUri;
102 		String loincPartMapName;
103 		switch (extCodeSystem) {
104 			case ITermLoaderSvc.SCT_URI:
105 				loincPartMapId = LOINC_SCT_PART_MAP_ID;
106 				loincPartMapUri = LOINC_SCT_PART_MAP_URI;
107 				loincPartMapName = LOINC_SCT_PART_MAP_NAME;
108 				break;
109 			case "http://www.nlm.nih.gov/research/umls/rxnorm":
110 				loincPartMapId = LOINC_RXNORM_PART_MAP_ID;
111 				loincPartMapUri = LOINC_RXNORM_PART_MAP_URI;
112 				loincPartMapName = LOINC_RXNORM_PART_MAP_NAME;
113 				break;
114 			case "http://www.radlex.org":
115 				loincPartMapId = LOINC_PART_TO_RID_PART_MAP_ID;
116 				loincPartMapUri = LOINC_PART_TO_RID_PART_MAP_URI;
117 				loincPartMapName = LOINC_PART_TO_RID_PART_MAP_NAME;
118 				break;
119 			case "http://pubchem.ncbi.nlm.nih.gov":
120 				loincPartMapId = LOINC_PUBCHEM_PART_MAP_ID;
121 				loincPartMapUri = LOINC_PUBCHEM_PART_MAP_URI;
122 				loincPartMapName = LOINC_PUBCHEM_PART_MAP_NAME;
123 				break;
124 			default:
125 				loincPartMapId = extCodeSystem.replaceAll("[^a-zA-Z]", "");
126 				loincPartMapUri = extCodeSystem;
127 				loincPartMapName = "Unknown Mapping";
128 				break;
129 		}
130 
131 		addConceptMapEntry(
132 			new ConceptMapping()
133 				.setConceptMapId(loincPartMapId)
134 				.setConceptMapUri(loincPartMapUri)
135 				.setConceptMapName(loincPartMapName)
136 				.setSourceCodeSystem(ITermLoaderSvc.LOINC_URI)
137 				.setSourceCode(partNumber)
138 				.setSourceDisplay(partName)
139 				.setTargetCodeSystem(extCodeSystem)
140 				.setTargetCode(extCodeId)
141 				.setTargetDisplay(extCodeDisplayName)
142 				.setTargetCodeSystemVersion(extCodeSystemVersion)
143 				.setEquivalence(equivalence)
144 				.setCopyright(extCodeSystemCopyrightNotice),
145 			CM_COPYRIGHT
146 		);
147 
148 	}
