23  import ca.uhn.fhir.jpa.entity.TermCodeSystemVersion;
24  import ca.uhn.fhir.jpa.entity.TermConcept;
25  import ca.uhn.fhir.jpa.term.IHapiTerminologyLoaderSvc;
26  import ca.uhn.fhir.jpa.term.IRecordHandler;
27  import ca.uhn.fhir.jpa.term.TerminologyLoaderSvcImpl;
28  import ca.uhn.fhir.rest.server.exceptions.InternalErrorException;
29  import org.apache.commons.csv.CSVRecord;
30  import org.apache.commons.lang3.Validate;
31  import org.hl7.fhir.r4.model.CodeSystem;
32  import org.slf4j.Logger;
33  import org.slf4j.LoggerFactory;
34  
35  import java.util.Map;
36  
37  import static org.apache.commons.lang3.StringUtils.*;
38  
39  public class LoincHandler implements IRecordHandler {
40  
41  	private static final Logger ourLog = LoggerFactory.getLogger(LoincHandler.class);
42  	private final Map<String, TermConcept> myCode2Concept;
43  	private final TermCodeSystemVersion myCodeSystemVersion;
44  	private final Map<String, CodeSystem.PropertyType> myPropertyNames;
45  	private final Map<PartTypeAndPartName, String> myPartTypeAndPartNameToPartNumber;
46  
47  	public LoincHandler(TermCodeSystemVersion theCodeSystemVersion, Map<String, TermConcept> theCode2concept, Map<String, CodeSystem.PropertyType> thePropertyNames, Map<PartTypeAndPartName, String> thePartTypeAndPartNameToPartNumber) {
48  		myCodeSystemVersion = theCodeSystemVersion;
49  		myCode2Concept = theCode2concept;
50  		myPropertyNames = thePropertyNames;
51  		myPartTypeAndPartNameToPartNumber = thePartTypeAndPartNameToPartNumber;
52  	}
53  
54  	@Override
55  	public void accept(CSVRecord theRecord) {
56  		String code = trim(theRecord.get("LOINC_NUM"));
57  		if (isNotBlank(code)) {
58  			String longCommonName = trim(theRecord.get("LONG_COMMON_NAME"));
59  			String shortName = trim(theRecord.get("SHORTNAME"));
60  			String consumerName = trim(theRecord.get("CONSUMER_NAME"));
61  			String display = TerminologyLoaderSvcImpl.firstNonBlank(longCommonName, shortName, consumerName);
62  
63  			TermConcept concept = new TermConcept(myCodeSystemVersion, code);
64  			concept.setDisplay(display);
65  
66  			if (isNotBlank(shortName) && !display.equalsIgnoreCase(shortName)) {
67  				concept
68  					.addDesignation()
69  					.setUseDisplay("ShortName")
70  					.setValue(shortName);
71  			}
72  
73  			for (String nextPropertyName : myPropertyNames.keySet()) {
74  				if (!theRecord.toMap().containsKey(nextPropertyName)) {
75  					continue;
76  				}
77  
78  				CodeSystem.PropertyType nextPropertyType = myPropertyNames.get(nextPropertyName);
79  
80  				String nextPropertyValue = theRecord.get(nextPropertyName);
81  				if (isNotBlank(nextPropertyValue)) {
82  					nextPropertyValue = trim(nextPropertyValue);
83  
84  					switch (nextPropertyType) {
85  						case STRING:
86  							concept.addPropertyString(nextPropertyName, nextPropertyValue);
87  							break;
88  						case CODING:
89  							// FIXME: handle "Ser/Plas^Donor"
90  							String propertyValue = nextPropertyValue;
91  							if (nextPropertyName.equals("COMPONENT")) {
92  								if (propertyValue.contains("^")) {
93  									propertyValue = propertyValue.substring(0, propertyValue.indexOf("^"));
94  								} else if (propertyValue.contains("/")) {
95  									propertyValue = propertyValue.substring(0, propertyValue.indexOf("/"));
96  								}
97  							}
98  
99  							PartTypeAndPartName key = new PartTypeAndPartName(nextPropertyName, propertyValue);
100 							String partNumber = myPartTypeAndPartNameToPartNumber.get(key);
101 
102 							if (partNumber == null && nextPropertyName.equals("TIME_ASPCT")) {
103 								key = new PartTypeAndPartName("TIME", nextPropertyValue);
104 								partNumber = myPartTypeAndPartNameToPartNumber.get(key);
105 							}
106 							if (partNumber == null && nextPropertyName.equals("METHOD_TYP")) {
107 								key = new PartTypeAndPartName("METHOD", nextPropertyValue);
108 								partNumber = myPartTypeAndPartNameToPartNumber.get(key);
109 							}
110 							if (partNumber == null && nextPropertyName.equals("SCALE_TYP")) {
111 								key = new PartTypeAndPartName("SCALE", nextPropertyValue);
112 								partNumber = myPartTypeAndPartNameToPartNumber.get(key);
113 							}
114 
115 							if (partNumber == null && nextPropertyName.equals("SYSTEM") && nextPropertyValue.startsWith("^")) {
116 								continue;
117 							}
118 
119 							if (isNotBlank(partNumber)) {
120 								concept.addPropertyCoding(nextPropertyName, IHapiTerminologyLoaderSvc.LOINC_URI, partNumber, nextPropertyValue);
121 							} else {
122 								String msg = "Unable to find part code with TYPE[" + key.getPartType() + "] and NAME[" + nextPropertyValue + "] (using name " + propertyValue + ")";
123 								ourLog.warn(msg);
124 //								throw new InternalErrorException(msg);
125 							}
126 							break;
127 						case DECIMAL:
128 						case CODE:
129 						case INTEGER:
130 						case BOOLEAN:
131 						case DATETIME:
132 						case NULL:
133 							throw new InternalErrorException("Don't know how to handle LOINC property of type: " + nextPropertyType);
134 					}
135 
136 				}
137 			}
138 
139 			Validate.isTrue(!myCode2Concept.containsKey(code), "The code %s has appeared more than once", code);
140 			myCode2Concept.put(code, concept);
141 		}
142 	}
143 }
