144 		if (myManagedTxManager instanceof JpaTransactionManager) {
145 			JpaDialect jpaDialect = ((JpaTransactionManager) myManagedTxManager).getJpaDialect();
146 			if (jpaDialect instanceof HibernateJpaDialect) {

151 			ourLog.warn("JPA dialect does not support transaction isolation! This can have an impact on search performance.");
152 		}

156 	public void cancelAllActiveSearches() {
157 		for (SearchTask next : myIdToSearchTask.values()) {
158 			ourLog.info("Requesting immediate abort of search: {}", next.getSearch().getUuid());
159 			next.requestImmediateAbort();
160 			AsyncUtil.awaitLatchAndIgnoreInterrupt(next.getCompletionLatch(), 30, TimeUnit.SECONDS);
161 		}

174 	@Override
175 	@Transactional(propagation = Propagation.NEVER)
176 	public List<Long> getResources(final String theUuid, int theFrom, int theTo, @Nullable RequestDetails theRequestDetails) {
177 		TransactionTemplate txTemplate = new TransactionTemplate(myManagedTxManager);
178 		txTemplate.setPropagationBehavior(TransactionDefinition.PROPAGATION_REQUIRED);
179 
180 		// If we're actively searching right now, don't try to do anything until at least one batch has been
181 		// persisted in the DB
182 		SearchTask searchTask = myIdToSearchTask.get(theUuid);
183 		if (searchTask != null) {
184 			searchTask.awaitInitialSync();
185 		}
186 
187 		ourLog.trace("About to start looking for resources {}-{}", theFrom, theTo);
188 
189 		Search search;
190 		StopWatch sw = new StopWatch();
191 		while (true) {
192 
193 			if (myNeverUseLocalSearchForUnitTests == false) {
194 				if (searchTask != null) {
195 					ourLog.trace("Local search found");
196 					List<Long> resourcePids = searchTask.getResourcePids(theFrom, theTo);
197 					ourLog.trace("Local search returned {} pids, wanted {}-{} - Search: {}", resourcePids.size(), theFrom, theTo, searchTask.getSearch());
198 
199 					/*
200 					 * Generally, if a search task is open, the fastest possible thing is to just return its results. This
201 					 * will work most of the time, but can fail if the task hit a search threshold and the client is requesting
202 					 * results beyond that threashold. In that case, we'll keep going below, since that will trigger another
203 					 * task.
204 					 */
205 					if ((searchTask.getSearch().getNumFound() - searchTask.getSearch().getNumBlocked()) >= theTo || resourcePids.size() == (theTo - theFrom)) {
206 						return resourcePids;
207 					}
208 				}
209 			}
210 
211 			search = mySearchCacheSvc
212 				.fetchByUuid(theUuid)
213 				.orElseThrow(() -> newResourceGoneException(theUuid));
214 
215 			verifySearchHasntFailedOrThrowInternalErrorException(search);
216 			if (search.getStatus() == SearchStatusEnum.FINISHED) {
217 				ourLog.trace("Search entity marked as finished with {} results", search.getNumFound());
218 				break;
219 			}
220 			if (search.getNumFound() >= theTo) {
221 				ourLog.trace("Search entity has {} results so far", search.getNumFound());
222 				break;
223 			}
224 
225 			if (sw.getMillis() > myMaxMillisToWaitForRemoteResults) {
226 				ourLog.error("Search {} of type {} for {}{} timed out after {}ms", search.getId(), search.getSearchType(), search.getResourceType(), search.getSearchQueryString(), sw.getMillis());
227 				throw new InternalErrorException("Request timed out after " + sw.getMillis() + "ms");
228 			}
229 
230 			// If the search was saved in "pass complete mode" it's probably time to
231 			// start a new pass
232 			if (search.getStatus() == SearchStatusEnum.PASSCMPLET) {
233 				ourLog.trace("Going to try to start next search");
234 				Optional<Search> newSearch = mySearchCacheSvc.tryToMarkSearchAsInProgress(search);
235 				if (newSearch.isPresent()) {
236 					ourLog.trace("Launching new search");
237 					search = newSearch.get();
238 					String resourceType = search.getResourceType();
239 					SearchParameterMap params = search.getSearchParameterMap().orElseThrow(() -> new IllegalStateException("No map in PASSCOMPLET search"));
240 					IFhirResourceDao<?> resourceDao = myDaoRegistry.getResourceDao(resourceType);
241 					SearchContinuationTask task = new SearchContinuationTask(search, resourceDao, params, resourceType, theRequestDetails);
242 					myIdToSearchTask.put(search.getUuid(), task);
243 					myExecutor.submit(task);
244 				}
245 			}
246 
247 			AsyncUtil.sleep(500);
248 		}
249 
250 		ourLog.trace("Finished looping");
251 
252 		List<Long> pids = mySearchResultCacheSvc.fetchResultPids(search, theFrom, theTo);
253 		if (pids == null) {
254 			throw newResourceGoneException(theUuid);
255 		}
256 
257 		ourLog.trace("Fetched {} results", pids.size());
258 
259 		return pids;
260 	}

279 	@Override
280 	public IBundleProvider registerSearch(final IDao theCallingDao, final SearchParameterMap theParams, String theResourceType, CacheControlDirective theCacheControlDirective, RequestDetails theRequestDetails) {
281 		final String searchUuid = UUID.randomUUID().toString();
282 
283 		ourLog.debug("Registering new search {}", searchUuid);
284 
285 		Class<? extends IBaseResource> resourceTypeClass = myContext.getResourceDefinition(theResourceType).getImplementingClass();
286 		final ISearchBuilder sb = theCallingDao.newSearchBuilder();
287 		sb.setType(resourceTypeClass, theResourceType);
288 		sb.setFetchSize(mySyncSize);
289 
290 		final Integer loadSynchronousUpTo = getLoadSynchronousUpToOrNull(theCacheControlDirective);
291 
292 		if (theParams.isLoadSynchronous() || loadSynchronousUpTo != null) {
293 			ourLog.debug("Search {} is loading in synchronous mode", searchUuid);
294 			return executeQuery(theParams, theRequestDetails, searchUuid, sb, loadSynchronousUpTo);
295 		}
296 
297 		/*
298 		 * See if there are any cached searches whose results we can return
299 		 * instead
300 		 */
301 		boolean useCache = true;
302 		if (theCacheControlDirective != null && theCacheControlDirective.isNoCache() == true) {
303 			useCache = false;
304 		}
305 
306 		final String queryString = theParams.toNormalizedQueryString(myContext);
307 		if (theParams.getEverythingMode() == null) {
308 			if (myDaoConfig.getReuseCachedSearchResultsForMillis() != null && useCache) {
309 				IBundleProvider foundSearchProvider = findCachedQuery(theCallingDao, theParams, theResourceType, theRequestDetails, queryString);
310 				if (foundSearchProvider != null) {
311 					return foundSearchProvider;
312 				}
313 			}
314 		}
315 
316 		return submitSearch(theCallingDao, theParams, theResourceType, theRequestDetails, searchUuid, sb, queryString);
317 
318 	}

320 	@Override
321 	public Optional<Integer> getSearchTotal(String theUuid) {
322 		SearchTask task = myIdToSearchTask.get(theUuid);
323 		if (task != null) {
324 			return Optional.ofNullable(task.awaitInitialSync());
325 		}
326 
327 		/*
328 		 * In case there is no running search, if the total is listed as accurate we know one is coming
329 		 * so let's wait a bit for it to show up
330 		 */
331 		TransactionTemplate txTemplate = new TransactionTemplate(myManagedTxManager);
332 		txTemplate.setPropagationBehavior(TransactionDefinition.PROPAGATION_REQUIRED);
333 		Optional<Search> search = mySearchCacheSvc.fetchByUuid(theUuid);
334 		if (search.isPresent()) {
335 			Optional<SearchParameterMap> searchParameterMap = search.get().getSearchParameterMap();
336 			if (searchParameterMap.isPresent() && searchParameterMap.get().getSearchTotalMode() == SearchTotalModeEnum.ACCURATE) {
337 				for (int i = 0; i < 10; i++) {
338 					if (search.isPresent()) {
339 						verifySearchHasntFailedOrThrowInternalErrorException(search.get());
340 						if (search.get().getTotalCount() != null) {
341 							return Optional.of(search.get().getTotalCount());
342 						}
343 					}
344 					search = mySearchCacheSvc.fetchByUuid(theUuid);
345 				}
346 			}
347 		}
348 
349 		return Optional.empty();
350 	}

352 	@NotNull
353 	private IBundleProvider submitSearch(IDao theCallingDao, SearchParameterMap theParams, String theResourceType, RequestDetails theRequestDetails, String theSearchUuid, ISearchBuilder theSb, String theQueryString) {
354 		StopWatch w = new StopWatch();
355 		Search search = new Search();
356 		populateSearchEntity(theParams, theResourceType, theSearchUuid, theQueryString, search);
357 
358 		// Interceptor call: STORAGE_PRESEARCH_REGISTERED
359 		HookParams params = new HookParams()
360 			.add(ICachedSearchDetails.class, search)
361 			.add(RequestDetails.class, theRequestDetails)
362 			.addIfMatchesType(ServletRequestDetails.class, theRequestDetails);
363 		JpaInterceptorBroadcaster.doCallHooks(myInterceptorBroadcaster, theRequestDetails, Pointcut.STORAGE_PRESEARCH_REGISTERED, params);
364 
365 		SearchTask task = new SearchTask(search, theCallingDao, theParams, theResourceType, theRequestDetails);
366 		myIdToSearchTask.put(search.getUuid(), task);
367 		myExecutor.submit(task);
368 
369 		PersistedJpaSearchFirstPageBundleProvider retVal = new PersistedJpaSearchFirstPageBundleProvider(search, theCallingDao, task, theSb, myManagedTxManager, theRequestDetails);
370 		populateBundleProvider(retVal);
371 
372 		ourLog.debug("Search initial phase completed in {}ms", w.getMillis());
373 		return retVal;
374 	}

376 	@org.jetbrains.annotations.Nullable
377 	private IBundleProvider findCachedQuery(IDao theCallingDao, SearchParameterMap theParams, String theResourceType, RequestDetails theRequestDetails, String theQueryString) {
378 		TransactionTemplate txTemplate = new TransactionTemplate(myManagedTxManager);
379 		PersistedJpaBundleProvider foundSearchProvider = txTemplate.execute(t -> {
380 
381 			// Interceptor call: STORAGE_PRECHECK_FOR_CACHED_SEARCH
382 			HookParams params = new HookParams()
383 				.add(SearchParameterMap.class, theParams)
384 				.add(RequestDetails.class, theRequestDetails)
385 				.addIfMatchesType(ServletRequestDetails.class, theRequestDetails);
386 			Object outcome = JpaInterceptorBroadcaster.doCallHooksAndReturnObject(myInterceptorBroadcaster, theRequestDetails, Pointcut.STORAGE_PRECHECK_FOR_CACHED_SEARCH, params);
387 			if (Boolean.FALSE.equals(outcome)) {
388 				return null;
389 			}
390 
391 			// Check for a search matching the given hash
392 			Search searchToUse = findSearchToUseOrNull(theQueryString, theResourceType);
393 			if (searchToUse == null) {
394 				return null;
395 			}
396 
397 			ourLog.debug("Reusing search {} from cache", searchToUse.getUuid());
398 			// Interceptor call: JPA_PERFTRACE_SEARCH_REUSING_CACHED
399 			params = new HookParams()
400 				.add(SearchParameterMap.class, theParams)
401 				.add(RequestDetails.class, theRequestDetails)
402 				.addIfMatchesType(ServletRequestDetails.class, theRequestDetails);
403 			JpaInterceptorBroadcaster.doCallHooks(myInterceptorBroadcaster, theRequestDetails, Pointcut.JPA_PERFTRACE_SEARCH_REUSING_CACHED, params);
404 
405 			mySearchCacheSvc.updateSearchLastReturned(searchToUse, new Date());
406 
407 			PersistedJpaBundleProvider retVal = new PersistedJpaBundleProvider(theRequestDetails, searchToUse.getUuid(), theCallingDao);
408 			retVal.setCacheHit(true);
409 			populateBundleProvider(retVal);
410 
411 			return retVal;
412 		});
413 
414 		if (foundSearchProvider != null) {
415 			return foundSearchProvider;
416 		}
417 		return null;
418 	}

420 	@Nullable
421 	private Search findSearchToUseOrNull(String theQueryString, String theResourceType) {
422 		Search searchToUse = null;
423 
424 		// createdCutoff is in recent past
425 		final Instant createdCutoff = Instant.now().minus(myDaoConfig.getReuseCachedSearchResultsForMillis(), ChronoUnit.MILLIS);
426 		Collection<Search> candidates = mySearchCacheSvc.findCandidatesForReuse(theResourceType, theQueryString, theQueryString.hashCode(), Date.from(createdCutoff));
427 
428 		for (Search nextCandidateSearch : candidates) {
429 			// We should only reuse our search if it was created within the permitted window
430 			// Date.after() is unreliable.  Instant.isAfter() always works.
431 			if (theQueryString.equals(nextCandidateSearch.getSearchQueryString()) && nextCandidateSearch.getCreated().toInstant().isAfter(createdCutoff)) {
432 				searchToUse = nextCandidateSearch;
433 				break;
434 			}
435 		}
436 		return searchToUse;
437 	}

439 	private IBundleProvider executeQuery(SearchParameterMap theParams, RequestDetails theRequestDetails, String theSearchUuid, ISearchBuilder theSb, Integer theLoadSynchronousUpTo) {
440 		SearchRuntimeDetails searchRuntimeDetails = new SearchRuntimeDetails(theRequestDetails, theSearchUuid);
441 		searchRuntimeDetails.setLoadSynchronous(true);
442 
443 		// Execute the query and make sure we return distinct results
444 		TransactionTemplate txTemplate = new TransactionTemplate(myManagedTxManager);
445 		txTemplate.setPropagationBehavior(TransactionDefinition.PROPAGATION_REQUIRED);
446 		return txTemplate.execute(t -> {
447 
448 			// Load the results synchronously
449 			final List<Long> pids = new ArrayList<>();
450 
451 			try (IResultIterator resultIter = theSb.createQuery(theParams, searchRuntimeDetails, theRequestDetails)) {
452 				while (resultIter.hasNext()) {
453 					pids.add(resultIter.next());
454 					if (theLoadSynchronousUpTo != null && pids.size() >= theLoadSynchronousUpTo) {
455 						break;
456 					}
457 					if (theParams.getLoadSynchronousUpTo() != null && pids.size() >= theParams.getLoadSynchronousUpTo()) {
458 						break;
459 					}
460 				}
461 			} catch (IOException e) {
462 				ourLog.error("IO failure during database access", e);
463 				throw new InternalErrorException(e);
464 			}
465 
466 			JpaPreResourceAccessDetails accessDetails = new JpaPreResourceAccessDetails(pids, () -> theSb);
467 			HookParams params = new HookParams()
468 				.add(IPreResourceAccessDetails.class, accessDetails)
469 				.add(RequestDetails.class, theRequestDetails)
470 				.addIfMatchesType(ServletRequestDetails.class, theRequestDetails);
471 			JpaInterceptorBroadcaster.doCallHooks(myInterceptorBroadcaster, theRequestDetails, Pointcut.STORAGE_PREACCESS_RESOURCES, params);
472 
473 			for (int i = pids.size() - 1; i >= 0; i--) {
474 				if (accessDetails.isDontReturnResourceAtIndex(i)) {
475 					pids.remove(i);
476 				}
477 			}
478 
479 			/*
480 			 * For synchronous queries, we load all the includes right away
481 			 * since we're returning a static bundle with all the results
482 			 * pre-loaded. This is ok because syncronous requests are not
483 			 * expected to be paged
484 			 *
485 			 * On the other hand for async queries we load includes/revincludes
486 			 * individually for pages as we return them to clients
487 			 */
488 			final Set<Long> includedPids = new HashSet<>();
489 			includedPids.addAll(theSb.loadIncludes(myContext, myEntityManager, pids, theParams.getRevIncludes(), true, theParams.getLastUpdated(), "(synchronous)", theRequestDetails));
490 			includedPids.addAll(theSb.loadIncludes(myContext, myEntityManager, pids, theParams.getIncludes(), false, theParams.getLastUpdated(), "(synchronous)", theRequestDetails));
491 			List<Long> includedPidsList = new ArrayList<>(includedPids);
492 
493 			List<IBaseResource> resources = new ArrayList<>();
494 			theSb.loadResourcesByPid(pids, includedPidsList, resources, false, theRequestDetails);
495 
496 			// Hook: STORAGE_PRESHOW_RESOURCES
497 			InterceptorUtil.fireStoragePreshowResource(resources, theRequestDetails, myInterceptorBroadcaster);
498 
499 			return new SimpleBundleProvider(resources);
500 		});
501 	}

503 	@org.jetbrains.annotations.Nullable
504 	private Integer getLoadSynchronousUpToOrNull(CacheControlDirective theCacheControlDirective) {
505 		final Integer loadSynchronousUpTo;
506 		if (theCacheControlDirective != null && theCacheControlDirective.isNoStore()) {
507 			if (theCacheControlDirective.getMaxResults() != null) {
508 				loadSynchronousUpTo = theCacheControlDirective.getMaxResults();
509 				if (loadSynchronousUpTo > myDaoConfig.getCacheControlNoStoreMaxResultsUpperLimit()) {
510 					throw new InvalidRequestException(Constants.HEADER_CACHE_CONTROL + " header " + Constants.CACHE_CONTROL_MAX_RESULTS + " value must not exceed " + myDaoConfig.getCacheControlNoStoreMaxResultsUpperLimit());
511 				}
512 			} else {
513 				loadSynchronousUpTo = 100;
514 			}
515 		} else {
516 			loadSynchronousUpTo = null;
517 		}

582 	/**
583 	 * A search task is a Callable task that runs in
584 	 * a thread pool to handle an individual search. One instance
585 	 * is created for any requested search and runs from the
586 	 * beginning to the end of the search.
587 	 * <p>
588 	 * Understand:
589 	 * This class executes in its own thread separate from the
590 	 * web server client thread that made the request. We do that
591 	 * so that we can return to the client as soon as possible,
592 	 * but keep the search going in the background (and have
593 	 * the next page of results ready to go when the client asks).
594 	 */
595 	public class SearchTask implements Callable<Void> {
596 		private final SearchParameterMap myParams;
597 		private final IDao myCallingDao;
598 		private final String myResourceType;
599 		private final ArrayList<Long> mySyncedPids = new ArrayList<>();
600 		private final CountDownLatch myInitialCollectionLatch = new CountDownLatch(1);
601 		private final CountDownLatch myCompletionLatch;
602 		private final ArrayList<Long> myUnsyncedPids = new ArrayList<>();
603 		private final RequestDetails myRequest;
604 		private Search mySearch;
605 		private boolean myAbortRequested;
606 		private int myCountSavedTotal = 0;
607 		private int myCountSavedThisPass = 0;
608 		private int myCountBlockedThisPass = 0;
609 		private boolean myAdditionalPrefetchThresholdsRemaining;

667 		@Nonnull
668 		List<Long> getResourcePids(int theFromIndex, int theToIndex) {
669 			ourLog.debug("Requesting search PIDs from {}-{}", theFromIndex, theToIndex);
670 
671 			boolean keepWaiting;
672 			do {
673 				synchronized (mySyncedPids) {
674 					ourLog.trace("Search status is {}", mySearch.getStatus());
675 					boolean haveEnoughResults = mySyncedPids.size() >= theToIndex;
676 					if (!haveEnoughResults) {
677 						switch (mySearch.getStatus()) {
678 							case LOADING:
679 								keepWaiting = true;
680 								break;
681 							case PASSCMPLET:
682 								/*
683 								 * If we get here, it means that the user requested resources that crossed the
684 								 * current pre-fetch boundary. For example, if the prefetch threshold is 50 and the
685 								 * user has requested resources 0-60, then they would get 0-50 back but the search
686 								 * coordinator would then stop searching.SearchCoordinatorSvcImplTest
687 								 */
688 								keepWaiting = false;
689 								break;
690 							case FAILED:
691 							case FINISHED:
692 							default:
693 								keepWaiting = false;
694 								break;
695 						}
696 					} else {
697 						keepWaiting = false;
698 					}
699 				}
700 
701 				if (keepWaiting) {
702 					ourLog.info("Waiting as we only have {} results - Search status: {}", mySyncedPids.size(), mySearch.getStatus());
703 					AsyncUtil.sleep(500L);
704 				}
705 			} while (keepWaiting);
706 
707 			ourLog.debug("Proceeding, as we have {} results", mySyncedPids.size());
708 
709 			ArrayList<Long> retVal = new ArrayList<>();
710 			synchronized (mySyncedPids) {
711 				verifySearchHasntFailedOrThrowInternalErrorException(mySearch);
712 
713 				int toIndex = theToIndex;
714 				if (mySyncedPids.size() < toIndex) {
715 					toIndex = mySyncedPids.size();
716 				}
717 				for (int i = theFromIndex; i < toIndex; i++) {
718 					retVal.add(mySyncedPids.get(i));
719 				}
720 			}
721 
722 			ourLog.trace("Done syncing results - Wanted {}-{} and returning {} of {}", theFromIndex, theToIndex, retVal.size(), mySyncedPids.size());
723 
724 			return retVal;
725 		}

739 		private void saveUnsynced(final IResultIterator theResultIter) {
740 			TransactionTemplate txTemplate = new TransactionTemplate(myManagedTxManager);
741 			txTemplate.setPropagationBehavior(TransactionDefinition.PROPAGATION_REQUIRED);
742 			txTemplate.execute(new TransactionCallbackWithoutResult() {
743 				@Override
744 				protected void doInTransactionWithoutResult(@NotNull TransactionStatus theArg0) {
745 					if (mySearch.getId() == null) {
746 						doSaveSearch();
747 					}
748 
749 					ArrayList<Long> unsyncedPids = myUnsyncedPids;
750 					int countBlocked = 0;
751 
752 					// Interceptor call: STORAGE_PREACCESS_RESOURCES
753 					// This can be used to remove results from the search result details before
754 					// the user has a chance to know that they were in the results
755 					if (mySearchRuntimeDetails.getRequestDetails() != null && unsyncedPids.isEmpty() == false) {
756 						JpaPreResourceAccessDetails accessDetails = new JpaPreResourceAccessDetails(unsyncedPids, () -> newSearchBuilder());
757 						HookParams params = new HookParams()
758 							.add(IPreResourceAccessDetails.class, accessDetails)
759 							.add(RequestDetails.class, mySearchRuntimeDetails.getRequestDetails())
760 							.addIfMatchesType(ServletRequestDetails.class, mySearchRuntimeDetails.getRequestDetails());
761 						JpaInterceptorBroadcaster.doCallHooks(myInterceptorBroadcaster, myRequest, Pointcut.STORAGE_PREACCESS_RESOURCES, params);
762 
763 						for (int i = unsyncedPids.size() - 1; i >= 0; i--) {
764 							if (accessDetails.isDontReturnResourceAtIndex(i)) {
765 								unsyncedPids.remove(i);
766 								myCountBlockedThisPass++;
767 								myCountSavedTotal++;
768 								countBlocked++;
769 							}
770 						}
771 					}
772 
773 					// Actually store the results in the query cache storage
774 					myCountSavedTotal += unsyncedPids.size();
775 					myCountSavedThisPass += unsyncedPids.size();
776 					mySearchResultCacheSvc.storeResults(mySearch, mySyncedPids, unsyncedPids);
777 
778 					synchronized (mySyncedPids) {
779 						int numSyncedThisPass = unsyncedPids.size();
780 						ourLog.trace("Syncing {} search results - Have more: {}", numSyncedThisPass, theResultIter.hasNext());
781 						mySyncedPids.addAll(unsyncedPids);
782 						unsyncedPids.clear();
783 
784 						if (theResultIter.hasNext() == false) {
785 							int skippedCount = theResultIter.getSkippedCount();
786 							int totalFetched = skippedCount + myCountSavedThisPass + myCountBlockedThisPass;
787 							ourLog.trace("MaxToFetch[{}] SkippedCount[{}] CountSavedThisPass[{}] CountSavedThisTotal[{}] AdditionalPrefetchRemaining[{}]", myMaxResultsToFetch, skippedCount, myCountSavedThisPass, myCountSavedTotal, myAdditionalPrefetchThresholdsRemaining);
788 
789 							if (myMaxResultsToFetch != null && totalFetched < myMaxResultsToFetch) {
790 								ourLog.trace("Setting search status to FINISHED");
791 								mySearch.setStatus(SearchStatusEnum.FINISHED);
792 								mySearch.setTotalCount(myCountSavedTotal);
793 							} else if (myAdditionalPrefetchThresholdsRemaining) {
794 								ourLog.trace("Setting search status to PASSCMPLET");
795 								mySearch.setStatus(SearchStatusEnum.PASSCMPLET);
796 								mySearch.setSearchParameterMap(myParams);
797 							} else {
798 								ourLog.trace("Setting search status to FINISHED");
799 								mySearch.setStatus(SearchStatusEnum.FINISHED);
800 								mySearch.setTotalCount(myCountSavedTotal);
801 							}
802 						}
803 					}
804 
805 					mySearch.setNumFound(myCountSavedTotal);
806 					mySearch.setNumBlocked(mySearch.getNumBlocked() + countBlocked);
807 
808 					int numSynced;
809 					synchronized (mySyncedPids) {
810 						numSynced = mySyncedPids.size();
811 					}
812 
813 					if (myDaoConfig.getCountSearchResultsUpTo() == null ||
814 						myDaoConfig.getCountSearchResultsUpTo() <= 0 ||
815 						myDaoConfig.getCountSearchResultsUpTo() <= numSynced) {
816 						myInitialCollectionLatch.countDown();
817 					}
818 
819 					doSaveSearch();
820 
821 				}
822 			});
823 
824 		}

845 		/**
846 		 * This is the method which actually performs the search.
847 		 * It is called automatically by the thread pool.
848 		 */
849 		@Override
850 		public Void call() {
851 			StopWatch sw = new StopWatch();
852 
853 			try {
854 				// Create an initial search in the DB and give it an ID
855 				saveSearch();
856 
857 				TransactionTemplate txTemplate = new TransactionTemplate(myManagedTxManager);
858 				txTemplate.setPropagationBehavior(TransactionDefinition.PROPAGATION_REQUIRES_NEW);
859 
860 				if (myCustomIsolationSupported) {
861 					txTemplate.setIsolationLevel(TransactionDefinition.ISOLATION_READ_COMMITTED);
862 				}
863 
864 				txTemplate.execute(new TransactionCallbackWithoutResult() {
865 					@Override
866 					protected void doInTransactionWithoutResult(TransactionStatus theStatus) {
867 						doSearch();
868 					}
869 				});
870 
871 				mySearchRuntimeDetails.setSearchStatus(mySearch.getStatus());
872 				if (mySearch.getStatus() == SearchStatusEnum.FINISHED) {
873 					HookParams params = new HookParams()
874 						.add(RequestDetails.class, myRequest)
875 						.addIfMatchesType(ServletRequestDetails.class, myRequest)
876 						.add(SearchRuntimeDetails.class, mySearchRuntimeDetails);
877 					JpaInterceptorBroadcaster.doCallHooks(myInterceptorBroadcaster, myRequest, Pointcut.JPA_PERFTRACE_SEARCH_COMPLETE, params);
878 				} else {
879 					HookParams params = new HookParams()
880 						.add(RequestDetails.class, myRequest)
881 						.addIfMatchesType(ServletRequestDetails.class, myRequest)
882 						.add(SearchRuntimeDetails.class, mySearchRuntimeDetails);
883 					JpaInterceptorBroadcaster.doCallHooks(myInterceptorBroadcaster, myRequest, Pointcut.JPA_PERFTRACE_SEARCH_PASS_COMPLETE, params);
884 				}
885 
886 				ourLog.trace("Have completed search for [{}{}] and found {} resources in {}ms - Status is {}", mySearch.getResourceType(), mySearch.getSearchQueryString(), mySyncedPids.size(), sw.getMillis(), mySearch.getStatus());
887 
888 			} catch (Throwable t) {
889 
890 				/*
891 				 * Don't print a stack trace for client errors (i.e. requests that
892 				 * aren't valid because the client screwed up).. that's just noise
893 				 * in the logs and who needs that.
894 				 */
895 				boolean logged = false;
896 				if (t instanceof BaseServerResponseException) {
897 					BaseServerResponseException exception = (BaseServerResponseException) t;
898 					if (exception.getStatusCode() >= 400 && exception.getStatusCode() < 500) {
899 						logged = true;
900 						ourLog.warn("Failed during search due to invalid request: {}", t.toString());
901 					}
902 				}
903 
904 				if (!logged) {
905 					ourLog.error("Failed during search loading after {}ms", sw.getMillis(), t);
906 				}
907 				myUnsyncedPids.clear();
908 
909 				Throwable rootCause = ExceptionUtils.getRootCause(t);
910 				rootCause = defaultIfNull(rootCause, t);
911 
912 				String failureMessage = rootCause.getMessage();
913 
914 				int failureCode = InternalErrorException.STATUS_CODE;
915 				if (t instanceof BaseServerResponseException) {
916 					failureCode = ((BaseServerResponseException) t).getStatusCode();
917 				}
918 
919 				mySearch.setFailureMessage(failureMessage);
920 				mySearch.setFailureCode(failureCode);
921 				mySearch.setStatus(SearchStatusEnum.FAILED);
922 
923 				mySearchRuntimeDetails.setSearchStatus(mySearch.getStatus());
924 				HookParams params = new HookParams()
925 					.add(RequestDetails.class, myRequest)
926 					.addIfMatchesType(ServletRequestDetails.class, myRequest)
927 					.add(SearchRuntimeDetails.class, mySearchRuntimeDetails);
928 				JpaInterceptorBroadcaster.doCallHooks(myInterceptorBroadcaster, myRequest, Pointcut.JPA_PERFTRACE_SEARCH_FAILED, params);
929 
930 				saveSearch();
931 
932 			} finally {
933 
934 				myIdToSearchTask.remove(mySearch.getUuid());
935 				myInitialCollectionLatch.countDown();
936 				markComplete();
937 
938 			}
939 			return null;
940 		}

942 		private void doSaveSearch() {
943 
944 			Search newSearch = mySearchCacheSvc.save(mySearch);
945 
946 			// mySearchDao.save is not supposed to return null, but in unit tests
947 			// it can if the mock search dao isn't set up to handle that
948 			if (newSearch != null) {
949 				mySearch = newSearch;
950 			}
951 		}
952 
953 		/**
954 		 * This method actually creates the database query to perform the
955 		 * search, and starts it.
956 		 */
957 		private void doSearch() {
958 
959 			/*
960 			 * If the user has explicitly requested a _count, perform a
961 			 *
962 			 * SELECT COUNT(*) ....
963 			 *
964 			 * before doing anything else.
965 			 */
966 			boolean wantOnlyCount = SummaryEnum.COUNT.equals(myParams.getSummaryMode());
967 			boolean wantCount =
968 				wantOnlyCount ||
969 					SearchTotalModeEnum.ACCURATE.equals(myParams.getSearchTotalMode()) ||
970 					(myParams.getSearchTotalMode() == null && SearchTotalModeEnum.ACCURATE.equals(myDaoConfig.getDefaultTotalMode()));
971 			if (wantCount) {
972 				ourLog.trace("Performing count");
973 				ISearchBuilder sb = newSearchBuilder();
974 				Iterator<Long> countIterator = sb.createCountQuery(myParams, mySearch.getUuid(), myRequest);
975 				Long count = countIterator.next();
976 				ourLog.trace("Got count {}", count);
977 
978 				TransactionTemplate txTemplate = new TransactionTemplate(myManagedTxManager);
979 				txTemplate.execute(new TransactionCallbackWithoutResult() {
980 					@Override
981 					protected void doInTransactionWithoutResult(TransactionStatus theArg0) {
982 						mySearch.setTotalCount(count.intValue());
983 						if (wantOnlyCount) {
984 							mySearch.setStatus(SearchStatusEnum.FINISHED);
985 						}
986 						doSaveSearch();
987 					}
988 				});
989 				if (wantOnlyCount) {
990 					return;
991 				}
992 			}
993 
994 			ourLog.trace("Done count");
995 			ISearchBuilder sb = newSearchBuilder();
996 
997 			/*
998 			 * Figure out how many results we're actually going to fetch from the
999 			 * database in this pass. This calculation takes into consideration the
1000 			 * "pre-fetch thresholds" specified in DaoConfig#getSearchPreFetchThresholds()
1001 			 * as well as the value of the _count parameter.
1002 			 */
1003 			int currentlyLoaded = defaultIfNull(mySearch.getNumFound(), 0);
1004 			int minWanted = 0;
1005 			if (myParams.getCount() != null) {
1006 				minWanted = myParams.getCount();
1007 				minWanted = Math.max(minWanted, myPagingProvider.getMaximumPageSize());
1008 				minWanted += currentlyLoaded;
1009 			}
1010 
1011 			for (Iterator<Integer> iter = myDaoConfig.getSearchPreFetchThresholds().iterator(); iter.hasNext(); ) {
1012 				int next = iter.next();
1013 				if (next != -1 && next <= currentlyLoaded) {
1014 					continue;
1015 				}
1016 
1017 				if (next == -1) {
1018 					sb.setMaxResultsToFetch(null);
1019 				} else {
1020 					myMaxResultsToFetch = Math.max(next, minWanted);
1021 					sb.setMaxResultsToFetch(myMaxResultsToFetch);
1022 				}
1023 
1024 				if (iter.hasNext()) {
1025 					myAdditionalPrefetchThresholdsRemaining = true;
1026 				}
1027 
1028 				// If we get here's we've found an appropriate threshold
1029 				break;
1030 			}
1031 
1032 			/*
1033 			 * Provide any PID we loaded in previous search passes to the
1034 			 * SearchBuilder so that we don't get duplicates coming from running
1035 			 * the same query again.
1036 			 *
1037 			 * We could possibly accomplish this in a different way by using sorted
1038 			 * results in our SQL query and specifying an offset. I don't actually
1039 			 * know if that would be faster or not. At some point should test this
1040 			 * idea.
1041 			 */
1042 			if (myPreviouslyAddedResourcePids != null) {
1043 				sb.setPreviouslyAddedResourcePids(myPreviouslyAddedResourcePids);
1044 				mySyncedPids.addAll(myPreviouslyAddedResourcePids);
1045 			}
1046 
1047 			/*
1048 			 * Construct the SQL query we'll be sending to the database
1049 			 */
1050 			try (IResultIterator resultIterator = sb.createQuery(myParams, mySearchRuntimeDetails, myRequest)) {
1051 				assert (resultIterator != null);
1052 
1053 				/*
1054 				 * The following loop actually loads the PIDs of the resources
1055 				 * matching the search off of the disk and into memory. After
1056 				 * every X results, we commit to the HFJ_SEARCH table.
1057 				 */
1058 				int syncSize = mySyncSize;
1059 				while (resultIterator.hasNext()) {
1060 					myUnsyncedPids.add(resultIterator.next());
1061 
1062 					boolean shouldSync = myUnsyncedPids.size() >= syncSize;
1063 
1064 					if (myDaoConfig.getCountSearchResultsUpTo() != null &&
1065 						myDaoConfig.getCountSearchResultsUpTo() > 0 &&
1066 						myDaoConfig.getCountSearchResultsUpTo() < myUnsyncedPids.size()) {
1067 						shouldSync = false;
1068 					}
1069 
1070 					if (myUnsyncedPids.size() > 50000) {
1071 						shouldSync = true;
1072 					}
1073 
1074 					// If no abort was requested, bail out
1075 					Validate.isTrue(isNotAborted(), "Abort has been requested");
1076 
1077 					if (shouldSync) {
1078 						saveUnsynced(resultIterator);
1079 					}
1080 
1081 					if (myLoadingThrottleForUnitTests != null) {
1082 						AsyncUtil.sleep(myLoadingThrottleForUnitTests);
1083 					}
1084 
1085 				}
1086 
1087 				// If no abort was requested, bail out
1088 				Validate.isTrue(isNotAborted(), "Abort has been requested");
1089 
1090 				saveUnsynced(resultIterator);
1091 
1092 			} catch (IOException e) {
1093 				ourLog.error("IO failure during database access", e);
1094 				throw new InternalErrorException(e);
1095 			}
1096 		}
1097 	}
