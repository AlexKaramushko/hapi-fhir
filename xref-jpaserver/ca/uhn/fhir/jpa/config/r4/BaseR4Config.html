58  @Configuration
59  @EnableTransactionManagement
60  public class BaseR4Config extends BaseConfig {
61  
62  	@Override
63  	public FhirContext fhirContext() {
64  		return fhirContextR4();
65  	}
66  
67  	@Bean
68  	@Primary
69  	public FhirContext fhirContextR4() {
70  		FhirContext retVal = FhirContext.forR4();
71  
72  		// Don't strip versions in some places
73  		ParserOptions parserOptions = retVal.getParserOptions();
74  		parserOptions.setDontStripVersionsFromReferencesAtPaths("AuditEvent.entity.what");
75  
76  		return retVal;
77  	}
78  
79  	@Bean
80  	public TransactionProcessor.ITransactionProcessorVersionAdapter transactionProcessorVersionFacade() {
81  		return new TransactionProcessorVersionAdapterR4();
82  	}
83  
84  	@Bean
85  	public TransactionProcessor<Bundle, Bundle.BundleEntryComponent> transactionProcessor() {
86  		return new TransactionProcessor<>();
87  	}
88  
89  	@Bean(name = GRAPHQL_PROVIDER_NAME)
90  	@Lazy
91  	public GraphQLProvider graphQLProvider() {
92  		return new GraphQLProvider(fhirContextR4(), validationSupportChainR4(), graphqlStorageServices());
93  	}
94  
95  	@Bean(name = "myInstanceValidatorR4")
96  	@Lazy
97  	public IValidatorModule instanceValidatorR4() {
98  		FhirInstanceValidator val = new FhirInstanceValidator();
99  		IResourceValidator.BestPracticeWarningLevel level = IResourceValidator.BestPracticeWarningLevel.Warning;
100 		val.setBestPracticeWarningLevel(level);
101 		val.setValidationSupport(validationSupportChainR4());
102 		return val;
103 	}
104 
105 	@Bean
106 	public JpaValidationSupportChainR4 jpaValidationSupportChain() {
107 		return new JpaValidationSupportChainR4();
108 	}
109 
110 	@Bean(name = "myJpaValidationSupportR4", autowire = Autowire.BY_NAME)
111 	public ca.uhn.fhir.jpa.dao.r4.IJpaValidationSupportR4 jpaValidationSupportR4() {
112 		ca.uhn.fhir.jpa.dao.r4.JpaValidationSupportR4 retVal = new ca.uhn.fhir.jpa.dao.r4.JpaValidationSupportR4();
113 		return retVal;
114 	}
115 
116 	@Bean(name = "myResourceCountsCache")
117 	public ResourceCountCache resourceCountsCache() {
118 		ResourceCountCache retVal = new ResourceCountCache(() -> systemDaoR4().getResourceCounts());
119 		retVal.setCacheMillis(4 * DateUtils.MILLIS_PER_HOUR);
120 		return retVal;
121 	}
122

124 	public IFulltextSearchSvc searchDaoR4() {

126 		return searchDao;
127 	}

132 	}

135 	public ISearchParamRegistry searchParamRegistry() {
136 		return new SearchParamRegistryR4();
137 	}
138 
139 	@Bean(name = "mySystemDaoR4", autowire = Autowire.BY_NAME)
140 	public IFhirSystemDao<org.hl7.fhir.r4.model.Bundle, org.hl7.fhir.r4.model.Meta> systemDaoR4() {
141 		ca.uhn.fhir.jpa.dao.r4.FhirSystemDaoR4 retVal = new ca.uhn.fhir.jpa.dao.r4.FhirSystemDaoR4();
142 		return retVal;
143 	}
144 
145 	@Bean(name = "mySystemProviderR4")
146 	public ca.uhn.fhir.jpa.provider.r4.JpaSystemProviderR4 systemProviderR4() {
147 		ca.uhn.fhir.jpa.provider.r4.JpaSystemProviderR4 retVal = new ca.uhn.fhir.jpa.provider.r4.JpaSystemProviderR4();
