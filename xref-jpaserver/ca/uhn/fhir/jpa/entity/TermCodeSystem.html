23  import ca.uhn.fhir.jpa.model.entity.ResourceTable;
24  import ca.uhn.fhir.util.ValidateUtil;
25  import org.apache.commons.lang3.builder.ToStringBuilder;
26  import org.apache.commons.lang3.builder.ToStringStyle;
27  
28  import javax.annotation.Nonnull;
29  import javax.persistence.*;
30  import java.io.Serializable;
31  
32  import static org.apache.commons.lang3.StringUtils.left;
33  import static org.apache.commons.lang3.StringUtils.length;
34  
35  //@formatter:off
36  @Table(name = "TRM_CODESYSTEM", uniqueConstraints = {
37  	@UniqueConstraint(name = "IDX_CS_CODESYSTEM", columnNames = {"CODE_SYSTEM_URI"})
38  })
39  @Entity()
40  //@formatter:on
41  public class TermCodeSystem implements Serializable {
42  	private static final long serialVersionUID = 1L;
43  
44  	private static final int MAX_NAME_LENGTH = 200;
45  	public static final int MAX_URL_LENGTH = 200;
46  
47  	@Column(name = "CODE_SYSTEM_URI", nullable = false, length = MAX_URL_LENGTH)
48  	private String myCodeSystemUri;
49  
50  	@OneToOne()
51  	@JoinColumn(name = "CURRENT_VERSION_PID", referencedColumnName = "PID", nullable = true, foreignKey = @ForeignKey(name = "FK_TRMCODESYSTEM_CURVER"))
52  	private TermCodeSystemVersion myCurrentVersion;
53  	@Id()
54  	@SequenceGenerator(name = "SEQ_CODESYSTEM_PID", sequenceName = "SEQ_CODESYSTEM_PID")
55  	@GeneratedValue(strategy = GenerationType.AUTO, generator = "SEQ_CODESYSTEM_PID")
56  	@Column(name = "PID")
57  	private Long myPid;
58  	@OneToOne()
59  	@JoinColumn(name = "RES_ID", referencedColumnName = "RES_ID", nullable = false, updatable = false, foreignKey = @ForeignKey(name = "FK_TRMCODESYSTEM_RES"))
60  	private ResourceTable myResource;
61  	@Column(name = "RES_ID", insertable = false, updatable = false)
62  	private Long myResourcePid;
63  	@Column(name = "CS_NAME", nullable = true, length = MAX_NAME_LENGTH)
64  	private String myName;
65  
66  	/**
67  	 * Constructor
68  	 */
69  	public TermCodeSystem() {
70  		super();
71  	}
72  
73  	public String getCodeSystemUri() {
74  		return myCodeSystemUri;
75  	}
76  
77  	public String getName() {
78  		return myName;
79  	}
80  
81  	public TermCodeSystem setCodeSystemUri(@Nonnull String theCodeSystemUri) {
82  		ValidateUtil.isNotBlankOrThrowIllegalArgument(theCodeSystemUri, "theCodeSystemUri must not be null or empty");
83  		ValidateUtil.isNotTooLongOrThrowIllegalArgument(theCodeSystemUri, MAX_URL_LENGTH,
84  			"URI exceeds maximum length (" + MAX_URL_LENGTH + "): " + length(theCodeSystemUri));
85  		myCodeSystemUri = theCodeSystemUri;
86  		return this;
87  	}
88  
89  	public TermCodeSystemVersion getCurrentVersion() {
90  		return myCurrentVersion;
91  	}
92  
93  	public TermCodeSystem setCurrentVersion(TermCodeSystemVersion theCurrentVersion) {
94  		myCurrentVersion = theCurrentVersion;
95  		return this;
96  	}
97  
98  	public Long getPid() {
99  		return myPid;
100 	}
101 
102 	public ResourceTable getResource() {
103 		return myResource;
104 	}
105 
106 	public TermCodeSystem setName(String theName) {
107 		myName = left(theName, MAX_NAME_LENGTH);
108 		return this;
109 	}
110 
111 	public TermCodeSystem setResource(ResourceTable theResource) {
112 		myResource = theResource;
113 		return this;
114 	}
115 
116 	@Override
117 	public String toString() {
118 		return new ToStringBuilder(this, ToStringStyle.SHORT_PREFIX_STYLE)
119 			.append("codeSystemUri", myCodeSystemUri)
120 			.append("currentVersion", myCurrentVersion)
121 			.append("pid", myPid)
122 			.append("resourcePid", myResourcePid)
123 			.append("name", myName)
124 			.toString();
125 	}
126 }
