23  import ca.uhn.fhir.util.ValidateUtil;
24  
25  import javax.annotation.Nonnull;
26  import javax.persistence.*;
27  import java.io.Serializable;
28  
29  import static org.apache.commons.lang3.StringUtils.left;
30  import static org.apache.commons.lang3.StringUtils.length;
31  
32  @Entity
33  @Table(name = "TRM_CONCEPT_DESIG", uniqueConstraints = {
34  }, indexes = {
35  })
36  public class TermConceptDesignation implements Serializable {
37  	private static final long serialVersionUID = 1L;
38  
39  	public static final int MAX_LENGTH = 500;
40  	public static final int MAX_VAL_LENGTH = 2000;
41  
42  	@ManyToOne
43  	@JoinColumn(name = "CONCEPT_PID", referencedColumnName = "PID", foreignKey = @ForeignKey(name = "FK_CONCEPTDESIG_CONCEPT"))
44  	private TermConcept myConcept;
45  	@Id()
46  	@SequenceGenerator(name = "SEQ_CONCEPT_DESIG_PID", sequenceName = "SEQ_CONCEPT_DESIG_PID")
47  	@GeneratedValue(strategy = GenerationType.AUTO, generator = "SEQ_CONCEPT_DESIG_PID")
48  	@Column(name = "PID")
49  	private Long myId;
50  	@Column(name = "LANG", nullable = true, length = MAX_LENGTH)
51  	private String myLanguage;
52  	@Column(name = "USE_SYSTEM", nullable = true, length = MAX_LENGTH)
53  	private String myUseSystem;
54  	@Column(name = "USE_CODE", nullable = true, length = MAX_LENGTH)
55  	private String myUseCode;
56  	@Column(name = "USE_DISPLAY", nullable = true, length = MAX_LENGTH)
57  	private String myUseDisplay;
58  	@Column(name = "VAL", nullable = false, length = MAX_VAL_LENGTH)
59  	private String myValue;
60  	/**
61  	 * TODO: Make this non-null
62  	 *
63  	 * @since 3.5.0
64  	 */
65  	@ManyToOne
66  	@JoinColumn(name = "CS_VER_PID", nullable = true, referencedColumnName = "PID", foreignKey = @ForeignKey(name = "FK_CONCEPTDESIG_CSV"))
67  	private TermCodeSystemVersion myCodeSystemVersion;
68  
69  	public String getLanguage() {
70  		return myLanguage;
71  	}
72  
73  	public TermConceptDesignation setLanguage(String theLanguage) {
74  		ValidateUtil.isNotTooLongOrThrowIllegalArgument(theLanguage, MAX_LENGTH,
75  			"Language exceeds maximum length (" + MAX_LENGTH + "): " + length(theLanguage));
76  		myLanguage = theLanguage;
77  		return this;
78  	}
79  
80  	public String getUseCode() {
81  		return myUseCode;
82  	}
83  
84  	public TermConceptDesignation setUseCode(String theUseCode) {
85  		ValidateUtil.isNotTooLongOrThrowIllegalArgument(theUseCode, MAX_LENGTH,
86  			"Use code exceeds maximum length (" + MAX_LENGTH + "): " + length(theUseCode));
87  		myUseCode = theUseCode;
88  		return this;
89  	}
90  
91  	public String getUseDisplay() {
92  		return myUseDisplay;
93  	}
94  
95  	public TermConceptDesignation setUseDisplay(String theUseDisplay) {
96  		myUseDisplay = left(theUseDisplay, MAX_LENGTH);
97  		return this;
98  	}
99  
100 	public String getUseSystem() {
101 		return myUseSystem;
102 	}
103 
104 	public TermConceptDesignation setUseSystem(String theUseSystem) {
105 		ValidateUtil.isNotTooLongOrThrowIllegalArgument(theUseSystem, MAX_LENGTH,
106 			"Use system exceeds maximum length (" + MAX_LENGTH + "): " + length(theUseSystem));
107 		myUseSystem = theUseSystem;
108 		return this;
109 	}
110 
111 	public String getValue() {
112 		return myValue;
113 	}
114 
115 	public TermConceptDesignation setValue(@Nonnull String theValue) {
116 		ValidateUtil.isNotBlankOrThrowIllegalArgument(theValue, "theValue must not be null or empty");
117 		ValidateUtil.isNotTooLongOrThrowIllegalArgument(theValue, MAX_VAL_LENGTH,
118 			"Value exceeds maximum length (" + MAX_VAL_LENGTH + "): " + length(theValue));
119 		myValue = theValue;
120 		return this;
121 	}
122 
123 	public TermConceptDesignation setCodeSystemVersion(TermCodeSystemVersion theCodeSystemVersion) {
124 		myCodeSystemVersion = theCodeSystemVersion;
125 		return this;
126 	}
127 
128 	public TermConceptDesignation setConcept(TermConcept theConcept) {
129 		myConcept = theConcept;
130 		return this;
131 	}
