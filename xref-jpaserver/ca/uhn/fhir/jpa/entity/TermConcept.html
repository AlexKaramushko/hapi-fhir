332 	private void parentPids(TermConcept theNextConcept, Set<Long> theParentPids) {
333 		for (TermConceptParentChildLink nextParentLink : theNextConcept.getParents()) {
334 			TermConcept parent = nextParentLink.getParent();
335 			if (parent != null) {
336 				Long parentConceptId = parent.getId();
337 				Validate.notNull(parentConceptId);
338 				if (theParentPids.add(parentConceptId)) {
339 					parentPids(parent, theParentPids);
340 				}
341 			}
342 		}
343 	}

345 	@SuppressWarnings("unused")
346 	@PreUpdate
347 	@PrePersist
348 	public void prePersist() {
349 		if (myParentPids == null) {
350 			Set<Long> parentPids = new HashSet<>();
351 			TermConcept entity = this;
352 			parentPids(entity, parentPids);
353 			entity.setParentPids(parentPids);
354 
355 			ourLog.trace("Code {}/{} has parents {}", entity.getId(), entity.getCode(), entity.getParentPidsAsString());
356 		}
357 	}

359 	private void setParentPids(Set<Long> theParentPids) {
360 		StringBuilder b = new StringBuilder();
361 		for (Long next : theParentPids) {
362 			if (b.length() > 0) {
363 				b.append(' ');
364 			}
365 			b.append(next);
366 		}
367 
368 		if (b.length() == 0) {
369 			b.append("NONE");
370 		}
371 
372 		myParentPids = b.toString();
373 	}

391 	public List<IContextValidationSupport.BaseConceptProperty> toValidationProperties() {
392 		List<IContextValidationSupport.BaseConceptProperty> retVal = new ArrayList<>();
393 		for (TermConceptProperty next : getProperties()) {
394 			switch (next.getType()) {
395 				case STRING:
396 					retVal.add(new IContextValidationSupport.StringConceptProperty(next.getKey(), next.getValue()));
397 					break;
398 				case CODING:
399 					retVal.add(new IContextValidationSupport.CodingConceptProperty(next.getKey(), next.getCodeSystem(), next.getValue(), next.getDisplay()));
400 					break;
401 				default:
402 					throw new IllegalStateException("Don't know how to handle " + next.getType());
403 			}
404 		}
405 		return retVal;
406 	}
