206 	private void extractTagsHapi(IResource theResource, ResourceTable theEntity, Set<ResourceTag> allDefs) {
207 		TagList tagList = ResourceMetadataKeyEnum.TAG_LIST.get(theResource);
208 		if (tagList != null) {
209 			for (Tag next : tagList) {
210 				TagDefinition def = getTagOrNull(TagTypeEnum.TAG, next.getScheme(), next.getTerm(), next.getLabel());
211 				if (def != null) {
212 					ResourceTag tag = theEntity.addTag(def);
213 					allDefs.add(tag);
214 					theEntity.setHasTags(true);
215 				}
216 			}
217 		}
218 
219 		List<BaseCodingDt> securityLabels = ResourceMetadataKeyEnum.SECURITY_LABELS.get(theResource);
220 		if (securityLabels != null) {
221 			for (BaseCodingDt next : securityLabels) {
222 				TagDefinition def = getTagOrNull(TagTypeEnum.SECURITY_LABEL, next.getSystemElement().getValue(), next.getCodeElement().getValue(), next.getDisplayElement().getValue());
223 				if (def != null) {
224 					ResourceTag tag = theEntity.addTag(def);
225 					allDefs.add(tag);
226 					theEntity.setHasTags(true);
227 				}
228 			}
229 		}
230 
231 		List<IdDt> profiles = ResourceMetadataKeyEnum.PROFILES.get(theResource);
232 		if (profiles != null) {
233 			for (IIdType next : profiles) {
234 				TagDefinition def = getTagOrNull(TagTypeEnum.PROFILE, NS_JPA_PROFILE, next.getValue(), null);
235 				if (def != null) {
236 					ResourceTag tag = theEntity.addTag(def);
237 					allDefs.add(tag);
238 					theEntity.setHasTags(true);
239 				}
240 			}
241 		}

327 	protected TagDefinition getTagOrNull(TagTypeEnum theTagType, String theScheme, String theTerm, String theLabel) {
328 		if (isBlank(theScheme) && isBlank(theTerm) && isBlank(theLabel)) {
329 			return null;
330 		}
331 
332 		CriteriaBuilder builder = myEntityManager.getCriteriaBuilder();
333 		CriteriaQuery<TagDefinition> cq = builder.createQuery(TagDefinition.class);
334 		Root<TagDefinition> from = cq.from(TagDefinition.class);
335 
336 		if (isNotBlank(theScheme)) {
337 			cq.where(
338 				builder.and(
339 					builder.equal(from.get("myTagType"), theTagType),
340 					builder.equal(from.get("mySystem"), theScheme),
341 					builder.equal(from.get("myCode"), theTerm)));
342 		} else {
343 			cq.where(
344 				builder.and(
345 					builder.equal(from.get("myTagType"), theTagType),
346 					builder.isNull(from.get("mySystem")),
347 					builder.equal(from.get("myCode"), theTerm)));
348 		}
349 
350 		TypedQuery<TagDefinition> q = myEntityManager.createQuery(cq);
351 		try {
352 			return q.getSingleResult();
353 		} catch (NoResultException e) {
354 			TagDefinition retVal = new TagDefinition(theTagType, theScheme, theTerm, theLabel);
355 			myEntityManager.persist(retVal);
356 			return retVal;
357 		}
358 	}

386 				search.setTotalCount(myResourceHistoryTableDao.countForAllResourceTypes());

450 	 * Returns true if the resource has changed (either the contents or the tags)
451 	 */
452 	protected EncodedResource populateResourceIntoEntity(RequestDetails theRequest, IBaseResource theResource, ResourceTable theEntity, boolean theUpdateHash) {
453 		if (theEntity.getResourceType() == null) {
454 			theEntity.setResourceType(toResourceName(theResource));
455 		}
456 
457 		if (theResource != null) {
458 			List<BaseResourceReferenceDt> refs = myContext.newTerser().getAllPopulatedChildElementsOfType(theResource, BaseResourceReferenceDt.class);
459 			for (BaseResourceReferenceDt nextRef : refs) {
460 				if (nextRef.getReference().isEmpty() == false) {
461 					if (nextRef.getReference().hasVersionIdPart()) {
462 						nextRef.setReference(nextRef.getReference().toUnqualifiedVersionless());
463 					}
464 				}
465 			}
466 		}
467 
468 		byte[] bytes;
469 		ResourceEncodingEnum encoding;
470 		boolean changed = false;
471 
472 		if (theEntity.getDeleted() == null) {
473 
474 			encoding = myConfig.getResourceEncoding();
475 			Set<String> excludeElements = ResourceMetaParams.EXCLUDE_ELEMENTS_IN_ENCODED;
476 			theEntity.setFhirVersion(myContext.getVersion().getVersion());
477 
478 			bytes = encodeResource(theResource, encoding, excludeElements, myContext);
479 
480 			if (theUpdateHash) {
481 				HashFunction sha256 = Hashing.sha256();
482 				String hashSha256 = sha256.hashBytes(bytes).toString();
483 				if (hashSha256.equals(theEntity.getHashSha256()) == false) {
484 					changed = true;
485 				}
486 				theEntity.setHashSha256(hashSha256);
487 			}
488 
489 			Set<ResourceTag> allDefs = new HashSet<>();
490 			Set<ResourceTag> allTagsOld = getAllTagDefinitions(theEntity);
491 
492 			if (theResource instanceof IResource) {
493 				extractTagsHapi((IResource) theResource, theEntity, allDefs);
494 			} else {
495 				extractTagsRi((IAnyResource) theResource, theEntity, allDefs);
496 			}
497 
498 			RuntimeResourceDefinition def = myContext.getResourceDefinition(theResource);
499 			if (def.isStandardType() == false) {
500 				String profile = def.getResourceProfile("");
501 				if (isNotBlank(profile)) {
502 					TagDefinition profileDef = getTagOrNull(TagTypeEnum.PROFILE, NS_JPA_PROFILE, profile, null);
503 					if (def != null) {
504 						ResourceTag tag = theEntity.addTag(profileDef);
505 						allDefs.add(tag);
506 						theEntity.setHasTags(true);
507 					}
508 				}
509 			}
510 
511 			Set<ResourceTag> allTagsNew = getAllTagDefinitions(theEntity);
512 			Set<TagDefinition> allDefsPresent = new HashSet<>();
513 			allTagsNew.forEach(tag -> {
514 
515 				// Don't keep duplicate tags
516 				if (!allDefsPresent.add(tag.getTag())) {
517 					theEntity.getTags().remove(tag);
518 				}
519 
520 				// Drop any tags that have been removed
521 				if (!allDefs.contains(tag)) {
522 					if (shouldDroppedTagBeRemovedOnUpdate(theRequest, tag)) {
523 						theEntity.getTags().remove(tag);
524 					}
525 				}
526 
527 			});
528 
529 			if (!allTagsOld.equals(allTagsNew)) {
530 				changed = true;
531 			}
532 			theEntity.setHasTags(!allTagsNew.isEmpty());
533 
534 		} else {
535 			theEntity.setHashSha256(null);
536 			bytes = null;
537 			encoding = ResourceEncodingEnum.DEL;
538 		}
539 
540 		if (changed == false) {
541 			if (theEntity.getId() == null) {
542 				changed = true;
543 			} else {
544 				ResourceHistoryTable currentHistoryVersion = theEntity.getCurrentVersionEntity();
545 				if (currentHistoryVersion == null) {
546 					currentHistoryVersion = myResourceHistoryTableDao.findForIdAndVersionAndFetchProvenance(theEntity.getId(), theEntity.getVersion());
547 				}
548 				if (currentHistoryVersion == null || currentHistoryVersion.getResource() == null) {
549 					changed = true;
550 				} else {
