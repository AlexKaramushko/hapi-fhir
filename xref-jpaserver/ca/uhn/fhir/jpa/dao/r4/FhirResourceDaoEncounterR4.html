41  public class FhirResourceDaoEncounterR4 extends BaseHapiFhirResourceDao<Encounter> implements IFhirResourceDaoEncounter<Encounter> {
42  
43  	@Override
44  	public IBundleProvider encounterInstanceEverything(HttpServletRequest theServletRequest, IIdType theId, IPrimitiveType<Integer> theCount, DateRangeParam theLastUpdated, SortSpec theSort) {
45  		SearchParameterMap paramMap = new SearchParameterMap();
46  		if (theCount != null) {
47  			paramMap.setCount(theCount.getValue());
48  		}
49  
50  //		paramMap.setRevIncludes(Collections.singleton(IResource.INCLUDE_ALL.asRecursive()));
51  		paramMap.setIncludes(Collections.singleton(IResource.INCLUDE_ALL.asRecursive()));
52  		paramMap.setEverythingMode(theId != null ? EverythingModeEnum.ENCOUNTER_INSTANCE : EverythingModeEnum.ENCOUNTER_TYPE);
53  		paramMap.setSort(theSort);
54  		paramMap.setLastUpdated(theLastUpdated);
55  		if (theId != null) {
56  			paramMap.add("_id", new StringParam(theId.getIdPart()));
57  		}
58  		IBundleProvider retVal = search(paramMap);
59  		return retVal;
60  	}
61  
62  	@Override
63  	public IBundleProvider encounterTypeEverything(HttpServletRequest theServletRequest, IPrimitiveType<Integer> theCount, DateRangeParam theLastUpdated, SortSpec theSort) {
64  		return encounterInstanceEverything(theServletRequest, null, theCount, theLastUpdated, theSort);
65  	}
