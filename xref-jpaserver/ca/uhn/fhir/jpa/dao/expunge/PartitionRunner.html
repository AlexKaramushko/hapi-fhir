23  import ca.uhn.fhir.jpa.dao.DaoConfig;
24  import ca.uhn.fhir.rest.server.exceptions.InternalErrorException;
25  import ca.uhn.fhir.util.StopWatch;
26  import com.google.common.collect.Lists;
27  import org.apache.commons.lang3.concurrent.BasicThreadFactory;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  import org.springframework.beans.factory.annotation.Autowired;
31  import org.springframework.data.domain.Slice;
32  import org.springframework.stereotype.Service;
33  
34  import java.util.ArrayList;
35  import java.util.List;
36  import java.util.concurrent.*;
37  import java.util.function.Consumer;
38  
39  @Service
40  public class PartitionRunner {
41  	private static final Logger ourLog = LoggerFactory.getLogger(ExpungeService.class);
42  	private static final int MAX_POOL_SIZE = 1000;
43  
44  	private final DaoConfig myDaoConfig;
45  
46  	@Autowired
47  	public PartitionRunner(DaoConfig theDaoConfig) {
48  		myDaoConfig = theDaoConfig;
49  	}
50  
51  	void runInPartitionedThreads(Slice<Long> theResourceIds, Consumer<List<Long>> partitionConsumer) {
52  
53  		List<Callable<Void>> callableTasks = buildCallableTasks(theResourceIds, partitionConsumer);
54  		if (callableTasks.size() == 0) {
55  			return;
56  		}
57  
58  		ExecutorService executorService = buildExecutor(callableTasks.size());
59  		try {
60  			List<Future<Void>> futures = executorService.invokeAll(callableTasks);
61  			// wait for all the threads to finish
62  			for (Future<Void> future : futures) {
63  				future.get();
64  			}
65  		} catch (InterruptedException e) {
66  			ourLog.error("Interrupted while expunging.", e);
67  			Thread.currentThread().interrupt();
68  		} catch (ExecutionException e) {
69  			ourLog.error("Error while expunging.", e);
70  			throw new InternalErrorException(e);
71  		} finally {
72  			executorService.shutdown();
73  		}
74  	}
75  
76  	private List<Callable<Void>> buildCallableTasks(Slice<Long> theResourceIds, Consumer<List<Long>> partitionConsumer) {
77  		List<Callable<Void>> retval = new ArrayList<>();
78  
79  		List<List<Long>> partitions = Lists.partition(theResourceIds.getContent(), myDaoConfig.getExpungeBatchSize());
80  
81  		for (List<Long> nextPartition : partitions) {
82  			Callable<Void> callableTask = () -> {
83  				ourLog.info("Expunging any search results pointing to {} resources", nextPartition.size());
84  				partitionConsumer.accept(nextPartition);
85  				return null;
86  			};
87  			retval.add(callableTask);
88  		}
89  
90  		return retval;
91  	}
92  
93  
94  	private ExecutorService buildExecutor(int numberOfTasks) {
95  		int threadCount = Math.min(numberOfTasks, myDaoConfig.getExpungeThreadCount());
96  		assert (threadCount > 0);
97  
98  		ourLog.info("Expunging with {} threads", threadCount);
99  		LinkedBlockingQueue<Runnable> executorQueue = new LinkedBlockingQueue<>(MAX_POOL_SIZE);
100 		BasicThreadFactory threadFactory = new BasicThreadFactory.Builder()
101 			.namingPattern("expunge-%d")
102 			.daemon(false)
103 			.priority(Thread.NORM_PRIORITY)
104 			.build();
105 		RejectedExecutionHandler rejectedExecutionHandler = (theRunnable, theExecutor) -> {
106 			ourLog.info("Note: Expunge executor queue is full ({} elements), waiting for a slot to become available!", executorQueue.size());
107 			StopWatch sw = new StopWatch();
108 			try {
109 				executorQueue.put(theRunnable);
110 			} catch (InterruptedException e) {
111 				throw new RejectedExecutionException("Task " + theRunnable.toString() +
112 					" rejected from " + e.toString());
113 			}
114 			ourLog.info("Slot become available after {}ms", sw.getMillis());
115 		};
116 		return new ThreadPoolExecutor(
117 			threadCount,
118 			MAX_POOL_SIZE,
119 			0L,
120 			TimeUnit.MILLISECONDS,
121 			executorQueue,
122 			threadFactory,
123 			rejectedExecutionHandler);
124 	}
125 }
