74  	private void addCompose(String theFilter, ValueSet theValueSetToPopulate, ValueSet theSourceValueSet, CodeSystemConcept theConcept) {
75  		if (isBlank(theFilter)) {
76  			addCompose(theValueSetToPopulate, theSourceValueSet.getCodeSystem().getSystem(), theConcept.getCode(), theConcept.getDisplay());
77  		} else {
78  			String filter = theFilter.toLowerCase();
79  			if (theConcept.getDisplay().toLowerCase().contains(filter) || theConcept.getCode().toLowerCase().contains(filter)) {
80  				addCompose(theValueSetToPopulate, theSourceValueSet.getCodeSystem().getSystem(), theConcept.getCode(), theConcept.getDisplay());
81  			}
82  		}
83  		for (CodeSystemConcept nextChild : theConcept.getConcept()) {
84  			addCompose(theFilter, theValueSetToPopulate, theSourceValueSet, nextChild);
85  		}
86  	}

88  	private void addCompose(ValueSet retVal, String theSystem, String theCode, String theDisplay) {
89  		if (isBlank(theCode)) {
90  			return;
91  		}
92  		ExpansionContains contains = retVal.getExpansion().addContains();
93  		contains.setSystem(theSystem);
94  		contains.setCode(theCode);
95  		contains.setDisplay(theDisplay);
96  	}

109 	@Override
110 	public ValueSet expand(ValueSet source, String theFilter) {
111 		ValueSet retVal = new ValueSet();
112 		retVal.setDate(DateTimeDt.withCurrentTime());
113 
114 		/*
115 		 * Add composed concepts
116 		 */
117 
118 		for (ComposeInclude nextInclude : source.getCompose().getInclude()) {
119 			for (ComposeIncludeConcept next : nextInclude.getConcept()) {
120 				if (isBlank(theFilter)) {
121 					addCompose(retVal, nextInclude.getSystem(), next.getCode(), next.getDisplay());
122 				} else {
123 					String filter = theFilter.toLowerCase();
124 					if (next.getDisplay().toLowerCase().contains(filter) || next.getCode().toLowerCase().contains(filter)) {
125 						addCompose(retVal, nextInclude.getSystem(), next.getCode(), next.getDisplay());
126 					}
127 				}
128 			}
129 		}
130 
131 		/*
132 		 * Add defined concepts
133 		 */
134 
135 		for (CodeSystemConcept next : source.getCodeSystem().getConcept()) {
136 			addCompose(theFilter, retVal, source, next);
137 		}
138 
139 		return retVal;
140 	}

148 	public ValueSet expandByIdentifier(String theUri, String theFilter) {
149 		if (isBlank(theUri)) {
150 			throw new InvalidRequestException("URI must not be blank or missing");
151 		}
152 		ValueSet source;
153 
154 		org.hl7.fhir.dstu2.model.ValueSet defaultValueSet = myDefaultProfileValidationSupport.fetchResource(myRiCtx, org.hl7.fhir.dstu2.model.ValueSet.class, theUri);
155 		if (defaultValueSet != null) {
156 			source = getContext().newJsonParser().parseResource(ValueSet.class, myRiCtx.newJsonParser().encodeResourceToString(defaultValueSet));
157 		} else {
158 			SearchParameterMap params = new SearchParameterMap();
159 			params.setLoadSynchronousUpTo(1);
160 			params.add(ValueSet.SP_URL, new UriParam(theUri));
161 			IBundleProvider ids = search(params);
162 			if (ids.size() == 0) {
163 				throw new InvalidRequestException("Unknown ValueSet URI: " + theUri);
164 			}
165 			source = (ValueSet) ids.getResources(0, 1).get(0);
166 		}
167 
168 		return expand(source, theFilter);
169 	}

177 	public List<IIdType> findCodeSystemIdsContainingSystemAndCode(String theCode, String theSystem, RequestDetails theRequest) {
178 		if (theSystem != null && theSystem.startsWith("http://hl7.org/fhir/")) {
179 			return Collections.singletonList((IIdType) new IdDt(theSystem));
180 		}
181 
182 		List<IIdType> valueSetIds;
183 		Set<Long> ids = searchForIds(new SearchParameterMap(ValueSet.SP_CODE, new TokenParam(theSystem, theCode)), theRequest);
184 		valueSetIds = new ArrayList<IIdType>();
185 		for (Long next : ids) {
186 			valueSetIds.add(new IdDt("ValueSet", next));
187 		}
188 		return valueSetIds;
189 	}

191 	private ValueSet loadValueSetForExpansion(IIdType theId, RequestDetails theRequest) {
192 		if (theId.getValue().startsWith("http://hl7.org/fhir/")) {
193 			org.hl7.fhir.dstu2.model.ValueSet valueSet = myValidationSupport.fetchResource(myRiCtx, org.hl7.fhir.dstu2.model.ValueSet.class, theId.getValue());
194 			if (valueSet != null) {
195 				return getContext().newJsonParser().parseResource(ValueSet.class, myRiCtx.newJsonParser().encodeResourceToString(valueSet));
196 			}
197 		}
198 		BaseHasResource sourceEntity = readEntity(theId, theRequest);
199 		if (sourceEntity == null) {
200 			throw new ResourceNotFoundException(theId);
201 		}
202 		ValueSet source = (ValueSet) toResource(sourceEntity, false);
203 		return source;
204 	}

206 	private IContextValidationSupport.LookupCodeResult lookup(List<ExpansionContains> theContains, String theSystem, String theCode) {
207 		for (ExpansionContains nextCode : theContains) {
208 
209 			String system = nextCode.getSystem();
210 			String code = nextCode.getCode();
211 			if (theSystem.equals(system) && theCode.equals(code)) {
212 				IContextValidationSupport.LookupCodeResult retVal = new IContextValidationSupport.LookupCodeResult();
213 				retVal.setSearchedForCode(code);
214 				retVal.setSearchedForSystem(system);
215 				retVal.setFound(true);
216 				if (nextCode.getAbstract() != null) {
217 					retVal.setCodeIsAbstract(nextCode.getAbstract());
218 				}
219 				retVal.setCodeDisplay(nextCode.getDisplay());
220 				retVal.setCodeSystemVersion(nextCode.getVersion());
221 				retVal.setCodeSystemDisplayName("Unknown"); // TODO: implement
222 				return retVal;
223 			}
224 
225 		}
226 
227 		return null;
228 	}

230 	@Override
231 	public IContextValidationSupport.LookupCodeResult lookupCode(IPrimitiveType<String> theCode, IPrimitiveType<String> theSystem, CodingDt theCoding, RequestDetails theRequest) {
232 		boolean haveCoding = theCoding != null && isNotBlank(theCoding.getSystem()) && isNotBlank(theCoding.getCode());
233 		boolean haveCode = theCode != null && theCode.isEmpty() == false;
234 		boolean haveSystem = theSystem != null && theSystem.isEmpty() == false;
235 
236 		if (!haveCoding && !(haveSystem && haveCode)) {
237 			throw new InvalidRequestException("No code, coding, or codeableConcept provided to validate");
238 		}
239 		if (!multiXor(haveCoding, (haveSystem && haveCode)) || (haveSystem != haveCode)) {
240 			throw new InvalidRequestException("$lookup can only validate (system AND code) OR (coding.system AND coding.code)");
241 		}
242 
243 		String code;
244 		String system;
245 		if (haveCoding) {
246 			code = theCoding.getCode();
247 			system = theCoding.getSystem();
248 		} else {
249 			code = theCode.getValue();
250 			system = theSystem.getValue();
251 		}
252 
253 		List<IIdType> valueSetIds = findCodeSystemIdsContainingSystemAndCode(code, system, theRequest);
254 		for (IIdType nextId : valueSetIds) {
255 			ValueSet expansion = expand(nextId, null, theRequest);
256 			List<ExpansionContains> contains = expansion.getExpansion().getContains();
257 			IContextValidationSupport.LookupCodeResult result = lookup(contains, system, code);
258 			if (result != null) {
259 				return result;
260 			}
261 		}
262 
263 		IContextValidationSupport.LookupCodeResult retVal = new IContextValidationSupport.LookupCodeResult();
264 		retVal.setFound(false);
265 		retVal.setSearchedForCode(code);
266 		retVal.setSearchedForSystem(system);
267 		return retVal;
268 	}

292 	@Override
293 	public ca.uhn.fhir.jpa.dao.IFhirResourceDaoValueSet.ValidateCodeResult validateCode(IPrimitiveType<String> theValueSetIdentifier, IIdType theId, IPrimitiveType<String> theCode,
294 			IPrimitiveType<String> theSystem, IPrimitiveType<String> theDisplay, CodingDt theCoding, CodeableConceptDt theCodeableConcept, RequestDetails theRequest) {
295 		List<IIdType> valueSetIds;
296 
297 		boolean haveCodeableConcept = theCodeableConcept != null && theCodeableConcept.getCoding().size() > 0;
298 		boolean haveCoding = theCoding != null && theCoding.isEmpty() == false;
299 		boolean haveCode = theCode != null && theCode.isEmpty() == false;
300 
301 		if (!haveCodeableConcept && !haveCoding && !haveCode) {
302 			throw new InvalidRequestException("No code, coding, or codeableConcept provided to validate");
303 		}
304 		if (!multiXor(haveCodeableConcept, haveCoding, haveCode)) {
305 			throw new InvalidRequestException("$validate-code can only validate (system AND code) OR (coding) OR (codeableConcept)");
306 		}
307 
308 		boolean haveIdentifierParam = theValueSetIdentifier != null && theValueSetIdentifier.isEmpty() == false;
309 		if (theId != null) {
310 			valueSetIds = Collections.singletonList(theId);
311 		} else if (haveIdentifierParam) {
312 			Set<Long> ids = searchForIds(new SearchParameterMap(ValueSet.SP_IDENTIFIER, new TokenParam(null, theValueSetIdentifier.getValue())), theRequest);
313 			valueSetIds = new ArrayList<>();
314 			for (Long next : ids) {
315 				valueSetIds.add(new IdDt("ValueSet", next));
316 			}
317 		} else {
318 			if (theCode == null || theCode.isEmpty()) {
319 				throw new InvalidRequestException("Either ValueSet ID or ValueSet identifier or system and code must be provided. Unable to validate.");
320 			}
321 			String code = theCode.getValue();
322 			String system = toStringOrNull(theSystem);
323 			valueSetIds = findCodeSystemIdsContainingSystemAndCode(code, system, theRequest);
324 		}
325 
326 		for (IIdType nextId : valueSetIds) {
327 			ValueSet expansion = expand(nextId, null, theRequest);
328 			List<ExpansionContains> contains = expansion.getExpansion().getContains();
329 			ValidateCodeResult result = validateCodeIsInContains(contains, toStringOrNull(theSystem), toStringOrNull(theCode), theCoding, theCodeableConcept);
330 			if (result != null) {
331 				if (theDisplay != null && isNotBlank(theDisplay.getValue()) && isNotBlank(result.getDisplay())) {
332 					if (!theDisplay.getValue().equals(result.getDisplay())) {
333 						return new ValidateCodeResult(false, "Display for code does not match", result.getDisplay());
334 					}
335 				}
336 				return result;
337 			}
338 		}
339 
340 		return new ValidateCodeResult(false, "Code not found", null);
341 	}

343 	private ca.uhn.fhir.jpa.dao.IFhirResourceDaoValueSet.ValidateCodeResult validateCodeIsInContains(List<ExpansionContains> contains, String theSystem, String theCode, CodingDt theCoding,
344 			CodeableConceptDt theCodeableConcept) {
345 		for (ExpansionContains nextCode : contains) {
346 			ca.uhn.fhir.jpa.dao.IFhirResourceDaoValueSet.ValidateCodeResult result = validateCodeIsInContains(nextCode.getContains(), theSystem, theCode, theCoding, theCodeableConcept);
347 			if (result != null) {
348 				return result;
349 			}
350 
351 			String system = nextCode.getSystem();
352 			String code = nextCode.getCode();
353 
354 			if (isNotBlank(theCode)) {
355 				if (theCode.equals(code) && (isBlank(theSystem) || theSystem.equals(system))) {
356 					return new ValidateCodeResult(true, "Validation succeeded", nextCode.getDisplay());
357 				}
358 			} else if (theCoding != null) {
359 				if (StringUtils.equals(system, theCoding.getSystem()) && StringUtils.equals(code, theCoding.getCode())) {
360 					return new ValidateCodeResult(true, "Validation succeeded", nextCode.getDisplay());
361 				}
362 			} else {
363 				for (CodingDt next : theCodeableConcept.getCoding()) {
364 					if (StringUtils.equals(system, next.getSystem()) && StringUtils.equals(code, next.getCode())) {
365 						return new ValidateCodeResult(true, "Validation succeeded", nextCode.getDisplay());
366 					}
367 				}
368 			}
369 
370 		}
371 
372 		return null;
373 	}

375 	private static boolean multiXor(boolean... theValues) {
376 		int count = 0;
377 		for (int i = 0; i < theValues.length; i++) {
378 			if (theValues[i]) {
379 				count++;
380 			}
381 		}
382 		return count == 1;
383 	}
