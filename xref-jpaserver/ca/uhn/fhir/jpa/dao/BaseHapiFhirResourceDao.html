101 	public void addTag(IIdType theId, TagTypeEnum theTagType, String theScheme, String theTerm, String theLabel, RequestDetails theRequest) {
102 		StopWatch w = new StopWatch();
103 		BaseHasResource entity = readEntity(theId, theRequest);
104 		if (entity == null) {
105 			throw new ResourceNotFoundException(theId);
106 		}
107 
108 		for (BaseTag next : new ArrayList<>(entity.getTags())) {
109 			if (ObjectUtil.equals(next.getTag().getTagType(), theTagType) &&
110 				ObjectUtil.equals(next.getTag().getSystem(), theScheme) &&
111 				ObjectUtil.equals(next.getTag().getCode(), theTerm)) {
112 				return;
113 			}
114 		}
115 
116 		entity.setHasTags(true);
117 
118 		TagDefinition def = getTagOrNull(TagTypeEnum.TAG, theScheme, theTerm, theLabel);
119 		if (def != null) {
120 			BaseTag newEntity = entity.addTag(def);
121 			if (newEntity.getTagId() == null) {
122 				myEntityManager.persist(newEntity);
123 				myEntityManager.merge(entity);
124 			}
125 		}
126 
127 		ourLog.debug("Processed addTag {}/{} on {} in {}ms", theScheme, theTerm, theId, w.getMillisAndRestart());
128 	}

163 			theResource.setId(UUID.randomUUID().toString());
164 		}
165

192 	public DaoMethodOutcome delete(IIdType theId, DeleteConflictList theDeleteConflicts, RequestDetails theRequest) {
193 		if (theId == null || !theId.hasIdPart()) {
194 			throw new InvalidRequestException("Can not perform delete, no ID provided");
195 		}
196 		final ResourceTable entity = readEntityLatestVersion(theId, theRequest);
197 		if (theId.hasVersionIdPart() && Long.parseLong(theId.getVersionIdPart()) != entity.getVersion()) {
198 			throw new ResourceVersionConflictException("Trying to delete " + theId + " but this is not the current version");
199 		}
200 
201 		// Don't delete again if it's already deleted
202 		if (entity.getDeleted() != null) {
203 			DaoMethodOutcome outcome = new DaoMethodOutcome();
204 			outcome.setEntity(entity);
205 
206 			IIdType id = getContext().getVersion().newIdType();
207 			id.setValue(entity.getIdDt().getValue());
208 			outcome.setId(id);
209 
210 			IBaseOperationOutcome oo = OperationOutcomeUtil.newInstance(getContext());
211 			String message = getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "successfulDeletes", 1, 0);
212 			String severity = "information";
213 			String code = "informational";
214 			OperationOutcomeUtil.addIssue(getContext(), oo, severity, message, null, code);
215 			outcome.setOperationOutcome(oo);
216 
217 			return outcome;
218 		}
219 
220 		StopWatch w = new StopWatch();
221 
222 		T resourceToDelete = toResource(myResourceType, entity, null, false);
223 		theDeleteConflicts.setResourceIdMarkedForDeletion(theId);
224 
225 		// Notify IServerOperationInterceptors about pre-action call
226 		HookParams hook = new HookParams()
227 			.add(IBaseResource.class, resourceToDelete)
228 			.add(RequestDetails.class, theRequest)
229 			.addIfMatchesType(ServletRequestDetails.class, theRequest);
230 		doCallHooks(theRequest, Pointcut.STORAGE_PRESTORAGE_RESOURCE_DELETED, hook);
231 
232 		myDeleteConflictService.validateOkToDelete(theDeleteConflicts, entity, false, theRequest);
233 
234 		preDelete(resourceToDelete, entity);
235 
236 		// Notify interceptors
237 		if (theRequest != null) {
238 			ActionRequestDetails requestDetails = new ActionRequestDetails(theRequest, getContext(), theId.getResourceType(), theId);
239 			notifyInterceptors(RestOperationTypeEnum.DELETE, requestDetails);
240 		}
241 
242 		ResourceTable savedEntity = updateEntityForDelete(theRequest, entity);
243 		resourceToDelete.setId(entity.getIdDt());
244 
245 		// Notify JPA interceptors
246 		TransactionSynchronizationManager.registerSynchronization(new TransactionSynchronizationAdapter() {
247 			@Override
248 			public void beforeCommit(boolean readOnly) {
249 				HookParams hookParams = new HookParams()
250 					.add(IBaseResource.class, resourceToDelete)
251 					.add(RequestDetails.class, theRequest)
252 					.addIfMatchesType(ServletRequestDetails.class, theRequest);
253 				doCallHooks(theRequest, Pointcut.STORAGE_PRECOMMIT_RESOURCE_DELETED, hookParams);
254 			}
255 		});
256 
257 		DaoMethodOutcome outcome = toMethodOutcome(theRequest, savedEntity, resourceToDelete).setCreated(true);
258 
259 		IBaseOperationOutcome oo = OperationOutcomeUtil.newInstance(getContext());
260 		String message = getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "successfulDeletes", 1, w.getMillis());
261 		String severity = "information";
262 		String code = "informational";
263 		OperationOutcomeUtil.addIssue(getContext(), oo, severity, message, null, code);
264 		outcome.setOperationOutcome(oo);
265 
266 		return outcome;
267 	}

291 	public DeleteMethodOutcome deleteByUrl(String theUrl, DeleteConflictList deleteConflicts, RequestDetails theRequest) {
292 		StopWatch w = new StopWatch();
293 
294 		Set<Long> resourceIds = myMatchResourceUrlService.processMatchUrl(theUrl, myResourceType, theRequest);
295 		if (resourceIds.size() > 1) {
296 			if (myDaoConfig.isAllowMultipleDelete() == false) {
297 				throw new PreconditionFailedException(getContext().getLocalizer().getMessageSanitized(BaseHapiFhirDao.class, "transactionOperationWithMultipleMatchFailure", "DELETE", theUrl, resourceIds.size()));
298 			}
299 		}
300 
301 		List<ResourceTable> deletedResources = new ArrayList<>();
302 		for (Long pid : resourceIds) {
303 			ResourceTable entity = myEntityManager.find(ResourceTable.class, pid);
304 			deletedResources.add(entity);
305 
306 			T resourceToDelete = toResource(myResourceType, entity, null, false);
307 
308 			// Notify IServerOperationInterceptors about pre-action call
309 			HookParams hooks = new HookParams()
310 				.add(IBaseResource.class, resourceToDelete)
311 				.add(RequestDetails.class, theRequest)
312 				.addIfMatchesType(ServletRequestDetails.class, theRequest);
313 			doCallHooks(theRequest, Pointcut.STORAGE_PRESTORAGE_RESOURCE_DELETED, hooks);
314 
315 			myDeleteConflictService.validateOkToDelete(deleteConflicts, entity, false, theRequest);
316 
317 			// Notify interceptors
318 			IdDt idToDelete = entity.getIdDt();
319 			if (theRequest != null) {
320 				ActionRequestDetails requestDetails = new ActionRequestDetails(theRequest, idToDelete.getResourceType(), idToDelete);
321 				notifyInterceptors(RestOperationTypeEnum.DELETE, requestDetails);
322 			}
323 
324 			// Perform delete
325 
326 			updateEntityForDelete(theRequest, entity);
327 			resourceToDelete.setId(entity.getIdDt());
328 
329 			// Notify JPA interceptors
330 			TransactionSynchronizationManager.registerSynchronization(new TransactionSynchronizationAdapter() {
331 				@Override
332 				public void beforeCommit(boolean readOnly) {
333 					HookParams hookParams = new HookParams()
334 						.add(IBaseResource.class, resourceToDelete)
335 						.add(RequestDetails.class, theRequest)
336 						.addIfMatchesType(ServletRequestDetails.class, theRequest);
337 					doCallHooks(theRequest, Pointcut.STORAGE_PRECOMMIT_RESOURCE_DELETED, hookParams);
338 				}
339 			});
340 		}
341 
342 		IBaseOperationOutcome oo;
343 		if (deletedResources.isEmpty()) {
344 			oo = OperationOutcomeUtil.newInstance(getContext());
345 			String message = getContext().getLocalizer().getMessageSanitized(BaseHapiFhirResourceDao.class, "unableToDeleteNotFound", theUrl);
346 			String severity = "warning";
347 			String code = "not-found";
348 			OperationOutcomeUtil.addIssue(getContext(), oo, severity, message, null, code);
349 		} else {
350 			oo = OperationOutcomeUtil.newInstance(getContext());
351 			String message = getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "successfulDeletes", deletedResources.size(), w.getMillis());
352 			String severity = "information";
353 			String code = "informational";
354 			OperationOutcomeUtil.addIssue(getContext(), oo, severity, message, null, code);
355 		}
356 
357 		ourLog.debug("Processed delete on {} (matched {} resource(s)) in {}ms", theUrl, deletedResources.size(), w.getMillis());
358 
359 		DeleteMethodOutcome retVal = new DeleteMethodOutcome();
360 		retVal.setDeletedEntities(deletedResources);
361 		retVal.setOperationOutcome(oo);
362 		return retVal;
363 	}

383 	private DaoMethodOutcome doCreate(T theResource, String theIfNoneExist, boolean thePerformIndexing, Date theUpdateTime, RequestDetails theRequest) {
384 		StopWatch w = new StopWatch();
385 
386 		preProcessResourceForStorage(theResource);
387 
388 		ResourceTable entity = new ResourceTable();
389 		entity.setResourceType(toResourceName(theResource));
390 
391 		if (isNotBlank(theIfNoneExist)) {
392 			Set<Long> match = myMatchResourceUrlService.processMatchUrl(theIfNoneExist, myResourceType, theRequest);
393 			if (match.size() > 1) {
394 				String msg = getContext().getLocalizer().getMessageSanitized(BaseHapiFhirDao.class, "transactionOperationWithMultipleMatchFailure", "CREATE", theIfNoneExist, match.size());
395 				throw new PreconditionFailedException(msg);
396 			} else if (match.size() == 1) {
397 				Long pid = match.iterator().next();
398 				entity = myEntityManager.find(ResourceTable.class, pid);
399 				IBaseResource resource = toResource(entity, false);
400 				theResource.setId(resource.getIdElement().getValue());
401 				return toMethodOutcome(theRequest, entity, resource).setCreated(false);
402 			}
403 		}
404 
405 		boolean serverAssignedId;
406 		if (isNotBlank(theResource.getIdElement().getIdPart())) {
407 			switch (myDaoConfig.getResourceClientIdStrategy()) {
408 				case NOT_ALLOWED:
409 					throw new ResourceNotFoundException(
410 						getContext().getLocalizer().getMessageSanitized(BaseHapiFhirResourceDao.class, "failedToCreateWithClientAssignedIdNotAllowed", theResource.getIdElement().getIdPart()));
411 				case ALPHANUMERIC:
412 					if (theResource.getIdElement().isIdPartValidLong()) {
413 						throw new InvalidRequestException(
414 							getContext().getLocalizer().getMessageSanitized(BaseHapiFhirResourceDao.class, "failedToCreateWithClientAssignedNumericId", theResource.getIdElement().getIdPart()));
415 					}
416 					createForcedIdIfNeeded(entity, theResource.getIdElement(), false);
417 					break;
418 				case ANY:
419 					createForcedIdIfNeeded(entity, theResource.getIdElement(), true);
420 					break;
421 			}
422 			serverAssignedId = false;
423 		} else {
424 			serverAssignedId = true;
425 		}
426 
427 		// Notify interceptors
428 		if (theRequest != null) {
429 			ActionRequestDetails requestDetails = new ActionRequestDetails(theRequest, getContext(), theResource);
430 			notifyInterceptors(RestOperationTypeEnum.CREATE, requestDetails);
431 		}
432 
433 		// Notify JPA interceptors
434 		HookParams hookParams = new HookParams()
435 			.add(IBaseResource.class, theResource)
436 			.add(RequestDetails.class, theRequest)
437 			.addIfMatchesType(ServletRequestDetails.class, theRequest);
438 		doCallHooks(theRequest, Pointcut.STORAGE_PRESTORAGE_RESOURCE_CREATED, hookParams);
439 
440 		// Perform actual DB update
441 		ResourceTable updatedEntity = updateEntity(theRequest, theResource, entity, null, thePerformIndexing, thePerformIndexing, theUpdateTime, false, thePerformIndexing);
442 
443 		theResource.setId(entity.getIdDt());
444 		if (serverAssignedId) {
445 			switch (myDaoConfig.getResourceClientIdStrategy()) {
446 				case NOT_ALLOWED:
447 				case ALPHANUMERIC:
448 					break;
449 				case ANY:
450 					ForcedId forcedId = createForcedIdIfNeeded(updatedEntity, theResource.getIdElement(), true);
451 					if (forcedId != null) {
452 						myForcedIdDao.save(forcedId);
453 					}
454 					break;
455 			}
456 		}
457 
458 		/*
459 		 * If we aren't indexing (meaning we're probably executing a sub-operation within a transaction),
460 		 * we'll manually increase the version. This is important because we want the updated version number
461 		 * to be reflected in the resource shared with interceptors
462 		 */
463 		if (!thePerformIndexing) {
464 			incrementId(theResource, entity, theResource.getIdElement());
465 		}
466 
467 		// Update the version/last updated in the resource so that interceptors get
468 		// the correct version
469 		updateResourceMetadata(entity, theResource);
470 
471 		// Populate the PID in the resource so it is available to hooks
472 		addPidToResource(entity, theResource);
473 
474 		// Notify JPA interceptors
475 		if (!updatedEntity.isUnchangedInCurrentOperation()) {
476 			TransactionSynchronizationManager.registerSynchronization(new TransactionSynchronizationAdapter() {
477 				@Override
478 				public void beforeCommit(boolean readOnly) {
479 					HookParams hookParams = new HookParams()
480 						.add(IBaseResource.class, theResource)
481 						.add(RequestDetails.class, theRequest)
482 						.addIfMatchesType(ServletRequestDetails.class, theRequest);
483 					doCallHooks(theRequest, Pointcut.STORAGE_PRECOMMIT_RESOURCE_CREATED, hookParams);
484 				}
485 			});
486 		}
487 
488 		DaoMethodOutcome outcome = toMethodOutcome(theRequest, entity, theResource).setCreated(true);
489 		if (!thePerformIndexing) {
490 			outcome.setId(theResource.getIdElement());
491 		}
492 
493 		String msg = getContext().getLocalizer().getMessageSanitized(BaseHapiFhirResourceDao.class, "successfulCreate", outcome.getId(), w.getMillisAndRestart());
494 		outcome.setOperationOutcome(createInfoOperationOutcome(msg));
495 
496 		ourLog.debug(msg);
497 		return outcome;
498 	}

500 	private <MT extends IBaseMetaType> void doMetaAdd(MT theMetaAdd, BaseHasResource entity) {
501 		List<TagDefinition> tags = toTagList(theMetaAdd);
502 
503 		for (TagDefinition nextDef : tags) {
504 
505 			boolean hasTag = false;
506 			for (BaseTag next : new ArrayList<>(entity.getTags())) {
507 				if (ObjectUtil.equals(next.getTag().getTagType(), nextDef.getTagType()) &&
508 					ObjectUtil.equals(next.getTag().getSystem(), nextDef.getSystem()) &&
509 					ObjectUtil.equals(next.getTag().getCode(), nextDef.getCode())) {
510 					hasTag = true;
511 					break;
512 				}
513 			}
514 
515 			if (!hasTag) {
516 				entity.setHasTags(true);
517 
518 				TagDefinition def = getTagOrNull(nextDef.getTagType(), nextDef.getSystem(), nextDef.getCode(), nextDef.getDisplay());
519 				if (def != null) {
520 					BaseTag newEntity = entity.addTag(def);
521 					if (newEntity.getTagId() == null) {
522 						myEntityManager.persist(newEntity);
523 					}
524 				}
525 			}
526 		}
527 
528 		validateMetaCount(entity.getTags().size());
529 
530 		myEntityManager.merge(entity);
531 	}

533 	private <MT extends IBaseMetaType> void doMetaDelete(MT theMetaDel, BaseHasResource entity) {
534 		List<TagDefinition> tags = toTagList(theMetaDel);
535 
536 		for (TagDefinition nextDef : tags) {
537 			for (BaseTag next : new ArrayList<BaseTag>(entity.getTags())) {
538 				if (ObjectUtil.equals(next.getTag().getTagType(), nextDef.getTagType()) &&
539 					ObjectUtil.equals(next.getTag().getSystem(), nextDef.getSystem()) &&

1309 	@Override
1310 	public DaoMethodOutcome update(T theResource, String theMatchUrl, boolean thePerformIndexing, boolean theForceUpdateVersion, RequestDetails theRequest) {
1311 		if (theResource == null) {
1312 			String msg = getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "missingBody");
1313 			throw new InvalidRequestException(msg);
1314 		}
1315 
1316 		StopWatch w = new StopWatch();
1317 
1318 		preProcessResourceForStorage(theResource);
1319 
1320 		final ResourceTable entity;
1321 
1322 		IIdType resourceId;
1323 		if (isNotBlank(theMatchUrl)) {
1324 			Set<Long> match = myMatchResourceUrlService.processMatchUrl(theMatchUrl, myResourceType, theRequest);
1325 			if (match.size() > 1) {
1326 				String msg = getContext().getLocalizer().getMessageSanitized(BaseHapiFhirDao.class, "transactionOperationWithMultipleMatchFailure", "UPDATE", theMatchUrl, match.size());
1327 				throw new PreconditionFailedException(msg);
1328 			} else if (match.size() == 1) {
1329 				Long pid = match.iterator().next();
1330 				entity = myEntityManager.find(ResourceTable.class, pid);
1331 				resourceId = entity.getIdDt();
1332 			} else {
1333 				return create(theResource, null, thePerformIndexing, new Date(), theRequest);
1334 			}
1335 		} else {
1336 			/*
1337 			 * Note: resourceId will not be null or empty here, because we
1338 			 * check it and reject requests in
1339 			 * BaseOutcomeReturningMethodBindingWithResourceParam
1340 			 */
1341 			resourceId = theResource.getIdElement();
1342 
1343 			try {
1344 				entity = readEntityLatestVersion(resourceId, theRequest);
1345 			} catch (ResourceNotFoundException e) {
1346 				return doCreate(theResource, null, thePerformIndexing, new Date(), theRequest);
1347 			}
1348 		}
1349 
1350 		if (resourceId.hasVersionIdPart() && Long.parseLong(resourceId.getVersionIdPart()) != entity.getVersion()) {
1351 			throw new ResourceVersionConflictException("Trying to update " + resourceId + " but this is not the current version");
1352 		}
1353 
1354 		if (resourceId.hasResourceType() && !resourceId.getResourceType().equals(getResourceName())) {
1355 			throw new UnprocessableEntityException(
1356 				"Invalid resource ID[" + entity.getIdDt().toUnqualifiedVersionless() + "] of type[" + entity.getResourceType() + "] - Does not match expected [" + getResourceName() + "]");
1357 		}
1358 
1359 		IBaseResource oldResource = toResource(entity, false);
1360 
1361 		/*
1362 		 * Mark the entity as not deleted - This is also done in the actual updateInternal()
1363 		 * method later on so it usually doesn't matter whether we do it here, but in the
1364 		 * case of a transaction with multiple PUTs we don't get there until later so
1365 		 * having this here means that a transaction can have a reference in one
1366 		 * resource to another resource in the same transaction that is being
1367 		 * un-deleted by the transaction. Wacky use case, sure. But it's real.
1368 		 *
1369 		 * See SystemProviderR4Test#testTransactionReSavesPreviouslyDeletedResources
1370 		 * for a test that needs this.
1371 		 */
1372 		boolean wasDeleted = entity.getDeleted() != null;
1373 		entity.setDeleted(null);
1374 
1375 		/*
1376 		 * If we aren't indexing, that means we're doing this inside a transaction.
1377 		 * The transaction will do the actual storage to the database a bit later on,
1378 		 * after placeholder IDs have been replaced, by calling {@link #updateInternal}
1379 		 * directly. So we just bail now.
1380 		 */
1381 		if (!thePerformIndexing) {
1382 			theResource.setId(entity.getIdDt().getValue());
1383 			DaoMethodOutcome outcome = toMethodOutcome(theRequest, entity, theResource).setCreated(wasDeleted);
1384 			outcome.setPreviousResource(oldResource);
1385 			return outcome;
1386 		}
1387 
1388 		/*
1389 		 * Otherwise, we're not in a transaction
1390 		 */
1391 		ResourceTable savedEntity = updateInternal(theRequest, theResource, thePerformIndexing, theForceUpdateVersion, entity, resourceId, oldResource);
1392 		DaoMethodOutcome outcome = toMethodOutcome(theRequest, savedEntity, theResource).setCreated(wasDeleted);
1393 
1394 		if (!thePerformIndexing) {
1395 			outcome.setId(theResource.getIdElement());
1396 		}
1397 
1398 		String msg = getContext().getLocalizer().getMessageSanitized(BaseHapiFhirResourceDao.class, "successfulUpdate", outcome.getId(), w.getMillisAndRestart());
1399 		outcome.setOperationOutcome(createInfoOperationOutcome(msg));
1400 
1401 		ourLog.debug(msg);
1402 		return outcome;
1403 	}
