101 	public void addTag(IIdType theId, TagTypeEnum theTagType, String theScheme, String theTerm, String theLabel) {
102 		StopWatch w = new StopWatch();
103 		BaseHasResource entity = readEntity(theId);
104 		if (entity == null) {
105 			throw new ResourceNotFoundException(theId);
106 		}
107 
108 		for (BaseTag next : new ArrayList<>(entity.getTags())) {
109 			if (ObjectUtil.equals(next.getTag().getTagType(), theTagType) &&
110 				ObjectUtil.equals(next.getTag().getSystem(), theScheme) &&
111 				ObjectUtil.equals(next.getTag().getCode(), theTerm)) {
112 				return;
113 			}
114 		}
115 
116 		entity.setHasTags(true);
117 
118 		TagDefinition def = getTagOrNull(TagTypeEnum.TAG, theScheme, theTerm, theLabel);
119 		if (def != null) {
120 			BaseTag newEntity = entity.addTag(def);
121 			if (newEntity.getTagId() == null) {
122 				myEntityManager.persist(newEntity);
123 				myEntityManager.merge(entity);
124 			}
125 		}
126 
127 		ourLog.debug("Processed addTag {}/{} on {} in {}ms", theScheme, theTerm, theId, w.getMillisAndRestart());
128 	}

163 			theResource.setId(UUID.randomUUID().toString());
164 		}

190 	public DaoMethodOutcome delete(IIdType theId, List<DeleteConflict> theDeleteConflicts, RequestDetails theRequest) {
191 		if (theId == null || !theId.hasIdPart()) {
192 			throw new InvalidRequestException("Can not perform delete, no ID provided");
193 		}
194 		final ResourceTable entity = readEntityLatestVersion(theId);
195 		if (theId.hasVersionIdPart() && Long.parseLong(theId.getVersionIdPart()) != entity.getVersion()) {
196 			throw new ResourceVersionConflictException("Trying to delete " + theId + " but this is not the current version");
197 		}
198 
199 		// Don't delete again if it's already deleted
200 		if (entity.getDeleted() != null) {
201 			DaoMethodOutcome outcome = new DaoMethodOutcome();
202 			outcome.setEntity(entity);
203 
204 			IIdType id = getContext().getVersion().newIdType();
205 			id.setValue(entity.getIdDt().getValue());
206 			outcome.setId(id);
207 
208 			IBaseOperationOutcome oo = OperationOutcomeUtil.newInstance(getContext());
209 			String message = getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "successfulDeletes", 1, 0);
210 			String severity = "information";
211 			String code = "informational";
212 			OperationOutcomeUtil.addIssue(getContext(), oo, severity, message, null, code);
213 			outcome.setOperationOutcome(oo);
214 
215 			return outcome;
216 		}
217 
218 		StopWatch w = new StopWatch();
219 
220 		T resourceToDelete = toResource(myResourceType, entity, null, false);
221 
222 		// Notify IServerOperationInterceptors about pre-action call
223 		HookParams hook = new HookParams()
224 			.add(IBaseResource.class, resourceToDelete)
225 			.add(RequestDetails.class, theRequest)
226 			.addIfMatchesType(ServletRequestDetails.class, theRequest);
227 		myInterceptorBroadcaster.callHooks(Pointcut.STORAGE_PRESTORAGE_RESOURCE_DELETED, hook);
228 
229 		validateOkToDelete(theDeleteConflicts, entity, false);
230 
231 		preDelete(resourceToDelete, entity);
232 
233 		// Notify interceptors
234 		if (theRequest != null) {
235 			ActionRequestDetails requestDetails = new ActionRequestDetails(theRequest, getContext(), theId.getResourceType(), theId);
236 			notifyInterceptors(RestOperationTypeEnum.DELETE, requestDetails);
237 		}
238 
239 		ResourceTable savedEntity = updateEntityForDelete(theRequest, entity);
240 		resourceToDelete.setId(entity.getIdDt());
241 
242 		// Notify JPA interceptors
243 		TransactionSynchronizationManager.registerSynchronization(new TransactionSynchronizationAdapter() {
244 			@Override
245 			public void beforeCommit(boolean readOnly) {
246 				HookParams hookParams = new HookParams()
247 					.add(IBaseResource.class, resourceToDelete)
248 					.add(RequestDetails.class, theRequest)
249 					.addIfMatchesType(ServletRequestDetails.class, theRequest);
250 				myInterceptorBroadcaster.callHooks(Pointcut.STORAGE_PRECOMMIT_RESOURCE_DELETED, hookParams);
251 			}
252 		});
253 
254 		DaoMethodOutcome outcome = toMethodOutcome(theRequest, savedEntity, resourceToDelete).setCreated(true);
255 
256 		IBaseOperationOutcome oo = OperationOutcomeUtil.newInstance(getContext());
257 		String message = getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "successfulDeletes", 1, w.getMillis());
258 		String severity = "information";
259 		String code = "informational";
260 		OperationOutcomeUtil.addIssue(getContext(), oo, severity, message, null, code);
261 		outcome.setOperationOutcome(oo);
262 
263 		return outcome;
264 	}

284 	public DeleteMethodOutcome deleteByUrl(String theUrl, List<DeleteConflict> deleteConflicts, RequestDetails theRequest) {
285 		StopWatch w = new StopWatch();
286 
287 		Set<Long> resourceIds = myMatchResourceUrlService.processMatchUrl(theUrl, myResourceType);
288 		if (resourceIds.size() > 1) {
289 			if (myDaoConfig.isAllowMultipleDelete() == false) {
290 				throw new PreconditionFailedException(getContext().getLocalizer().getMessage(BaseHapiFhirDao.class, "transactionOperationWithMultipleMatchFailure", "DELETE", theUrl, resourceIds.size()));
291 			}
292 		}
293 
294 		List<ResourceTable> deletedResources = new ArrayList<>();
295 		for (Long pid : resourceIds) {
296 			ResourceTable entity = myEntityManager.find(ResourceTable.class, pid);
297 			deletedResources.add(entity);
298 
299 			T resourceToDelete = toResource(myResourceType, entity, null, false);
300 
301 			// Notify IServerOperationInterceptors about pre-action call
302 			HookParams hooks = new HookParams()
303 				.add(IBaseResource.class, resourceToDelete)
304 				.add(RequestDetails.class, theRequest)
305 				.addIfMatchesType(ServletRequestDetails.class, theRequest);
306 			myInterceptorBroadcaster.callHooks(Pointcut.STORAGE_PRESTORAGE_RESOURCE_DELETED, hooks);
307 
308 			validateOkToDelete(deleteConflicts, entity, false);
309 
310 			// Notify interceptors
311 			IdDt idToDelete = entity.getIdDt();
312 			if (theRequest != null) {
313 				ActionRequestDetails requestDetails = new ActionRequestDetails(theRequest, idToDelete.getResourceType(), idToDelete);
314 				notifyInterceptors(RestOperationTypeEnum.DELETE, requestDetails);
315 			}
316 
317 			// Perform delete
318 
319 			updateEntityForDelete(theRequest, entity);
320 			resourceToDelete.setId(entity.getIdDt());
321 
322 			// Notify JPA interceptors
323 			TransactionSynchronizationManager.registerSynchronization(new TransactionSynchronizationAdapter() {
324 				@Override
325 				public void beforeCommit(boolean readOnly) {
326 					HookParams hookParams = new HookParams()
327 						.add(IBaseResource.class, resourceToDelete)
328 						.add(RequestDetails.class, theRequest)
329 						.addIfMatchesType(ServletRequestDetails.class, theRequest);
330 					myInterceptorBroadcaster.callHooks(Pointcut.STORAGE_PRECOMMIT_RESOURCE_DELETED, hookParams);
331 				}
332 			});
333 		}
334 
335 		IBaseOperationOutcome oo;
336 		if (deletedResources.isEmpty()) {
337 			oo = OperationOutcomeUtil.newInstance(getContext());
338 			String message = getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "unableToDeleteNotFound", theUrl);
339 			String severity = "warning";
340 			String code = "not-found";
341 			OperationOutcomeUtil.addIssue(getContext(), oo, severity, message, null, code);
342 		} else {
343 			oo = OperationOutcomeUtil.newInstance(getContext());
344 			String message = getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "successfulDeletes", deletedResources.size(), w.getMillis());
345 			String severity = "information";
346 			String code = "informational";
347 			OperationOutcomeUtil.addIssue(getContext(), oo, severity, message, null, code);
348 		}
349 
350 		ourLog.debug("Processed delete on {} (matched {} resource(s)) in {}ms", theUrl, deletedResources.size(), w.getMillis());
351 
352 		DeleteMethodOutcome retVal = new DeleteMethodOutcome();
353 		retVal.setDeletedEntities(deletedResources);
354 		retVal.setOperationOutcome(oo);
355 		return retVal;
356 	}

376 	private DaoMethodOutcome doCreate(T theResource, String theIfNoneExist, boolean thePerformIndexing, Date theUpdateTime, RequestDetails theRequest) {
377 		StopWatch w = new StopWatch();
378 
379 		preProcessResourceForStorage(theResource);
380 
381 		ResourceTable entity = new ResourceTable();
382 		entity.setResourceType(toResourceName(theResource));
383 
384 		if (isNotBlank(theIfNoneExist)) {
385 			Set<Long> match = myMatchResourceUrlService.processMatchUrl(theIfNoneExist, myResourceType);
386 			if (match.size() > 1) {
387 				String msg = getContext().getLocalizer().getMessage(BaseHapiFhirDao.class, "transactionOperationWithMultipleMatchFailure", "CREATE", theIfNoneExist, match.size());
388 				throw new PreconditionFailedException(msg);
389 			} else if (match.size() == 1) {
390 				Long pid = match.iterator().next();
391 				entity = myEntityManager.find(ResourceTable.class, pid);
392 				IBaseResource resource = toResource(entity, false);
393 				theResource.setId(resource.getIdElement().getValue());
394 				return toMethodOutcome(theRequest, entity, resource).setCreated(false);
395 			}
396 		}
397 
398 		boolean serverAssignedId;
399 		if (isNotBlank(theResource.getIdElement().getIdPart())) {
400 			switch (myDaoConfig.getResourceClientIdStrategy()) {
401 				case NOT_ALLOWED:
402 					throw new ResourceNotFoundException(
403 						getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "failedToCreateWithClientAssignedIdNotAllowed", theResource.getIdElement().getIdPart()));
404 				case ALPHANUMERIC:
405 					if (theResource.getIdElement().isIdPartValidLong()) {
406 						throw new InvalidRequestException(
407 							getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "failedToCreateWithClientAssignedNumericId", theResource.getIdElement().getIdPart()));
408 					}
409 					createForcedIdIfNeeded(entity, theResource.getIdElement(), false);
410 					break;
411 				case ANY:
412 					createForcedIdIfNeeded(entity, theResource.getIdElement(), true);
413 					break;
414 			}
415 			serverAssignedId = false;
416 		} else {
417 			serverAssignedId = true;
418 		}
419 
420 		// Notify interceptors
421 		if (theRequest != null) {
422 			ActionRequestDetails requestDetails = new ActionRequestDetails(theRequest, getContext(), theResource);
423 			notifyInterceptors(RestOperationTypeEnum.CREATE, requestDetails);
424 		}
425 
426 		// Notify JPA interceptors
427 		HookParams hookParams = new HookParams()
428 			.add(IBaseResource.class, theResource)
429 			.add(RequestDetails.class, theRequest)
430 			.addIfMatchesType(ServletRequestDetails.class, theRequest);
431 		myInterceptorBroadcaster.callHooks(Pointcut.STORAGE_PRESTORAGE_RESOURCE_CREATED, hookParams);
432 
433 		// Perform actual DB update
434 		ResourceTable updatedEntity = updateEntity(theRequest, theResource, entity, null, thePerformIndexing, thePerformIndexing, theUpdateTime, false, thePerformIndexing);
435 
436 		theResource.setId(entity.getIdDt());
437 		if (serverAssignedId) {
438 			switch (myDaoConfig.getResourceClientIdStrategy()) {
439 				case NOT_ALLOWED:
440 				case ALPHANUMERIC:
441 					break;
442 				case ANY:
443 					ForcedId forcedId = createForcedIdIfNeeded(updatedEntity, theResource.getIdElement(), true);
444 					if (forcedId != null) {
445 						myForcedIdDao.save(forcedId);
446 					}
447 					break;
448 			}
449 		}
450 
451 		/*
452 		 * If we aren't indexing (meaning we're probably executing a sub-operation within a transaction),
453 		 * we'll manually increase the version. This is important because we want the updated version number
454 		 * to be reflected in the resource shared with interceptors
455 		 */
456 		if (!thePerformIndexing) {
457 			incrementId(theResource, entity, theResource.getIdElement());
458 		}
459 
460 		// Update the version/last updated in the resource so that interceptors get
461 		// the correct version
462 		updateResourceMetadata(entity, theResource);
463 
464 		// Notify JPA interceptors
465 		if (!updatedEntity.isUnchangedInCurrentOperation()) {
466 			TransactionSynchronizationManager.registerSynchronization(new TransactionSynchronizationAdapter() {
467 				@Override
468 				public void beforeCommit(boolean readOnly) {
469 					HookParams hookParams = new HookParams()
470 						.add(IBaseResource.class, theResource)
471 						.add(RequestDetails.class, theRequest)
472 						.addIfMatchesType(ServletRequestDetails.class, theRequest);
473 					myInterceptorBroadcaster.callHooks(Pointcut.STORAGE_PRECOMMIT_RESOURCE_CREATED, hookParams);
474 				}
475 			});
476 		}
477 
478 		DaoMethodOutcome outcome = toMethodOutcome(theRequest, entity, theResource).setCreated(true);
479 		if (!thePerformIndexing) {
480 			outcome.setId(theResource.getIdElement());
481 		}
482 
483 		String msg = getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "successfulCreate", outcome.getId(), w.getMillisAndRestart());
484 		outcome.setOperationOutcome(createInfoOperationOutcome(msg));
485 
486 		ourLog.debug(msg);
487 		return outcome;
488 	}

490 	private <MT extends IBaseMetaType> void doMetaAdd(MT theMetaAdd, BaseHasResource entity) {
491 		List<TagDefinition> tags = toTagList(theMetaAdd);
492 
493 		for (TagDefinition nextDef : tags) {
494 
495 			boolean hasTag = false;
496 			for (BaseTag next : new ArrayList<>(entity.getTags())) {
497 				if (ObjectUtil.equals(next.getTag().getTagType(), nextDef.getTagType()) &&
498 					ObjectUtil.equals(next.getTag().getSystem(), nextDef.getSystem()) &&
499 					ObjectUtil.equals(next.getTag().getCode(), nextDef.getCode())) {
500 					hasTag = true;
501 					break;
502 				}
503 			}
504 
505 			if (!hasTag) {
506 				entity.setHasTags(true);
507 
508 				TagDefinition def = getTagOrNull(nextDef.getTagType(), nextDef.getSystem(), nextDef.getCode(), nextDef.getDisplay());
509 				if (def != null) {
510 					BaseTag newEntity = entity.addTag(def);
511 					if (newEntity.getTagId() == null) {
512 						myEntityManager.persist(newEntity);
513 					}
514 				}
515 			}
516 		}
517 
518 		validateMetaCount(entity.getTags().size());
519 
520 		myEntityManager.merge(entity);
521 	}

1254 	@Override
1255 	public DaoMethodOutcome update(T theResource, String theMatchUrl, boolean thePerformIndexing, boolean theForceUpdateVersion, RequestDetails theRequestDetails) {
1256 		if (theResource == null) {
1257 			String msg = getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "missingBody");
1258 			throw new InvalidRequestException(msg);
1259 		}
1260 
1261 		StopWatch w = new StopWatch();
1262 
1263 		preProcessResourceForStorage(theResource);
1264 
1265 		final ResourceTable entity;
1266 
1267 		IIdType resourceId;
1268 		if (isNotBlank(theMatchUrl)) {
1269 			Set<Long> match = myMatchResourceUrlService.processMatchUrl(theMatchUrl, myResourceType);
1270 			if (match.size() > 1) {
1271 				String msg = getContext().getLocalizer().getMessage(BaseHapiFhirDao.class, "transactionOperationWithMultipleMatchFailure", "UPDATE", theMatchUrl, match.size());
1272 				throw new PreconditionFailedException(msg);
1273 			} else if (match.size() == 1) {
1274 				Long pid = match.iterator().next();
1275 				entity = myEntityManager.find(ResourceTable.class, pid);
1276 				resourceId = entity.getIdDt();
1277 			} else {
1278 				return create(theResource, null, thePerformIndexing, new Date(), theRequestDetails);
1279 			}
1280 		} else {
1281 			/*
1282 			 * Note: resourceId will not be null or empty here, because we
1283 			 * check it and reject requests in
1284 			 * BaseOutcomeReturningMethodBindingWithResourceParam
1285 			 */
1286 			resourceId = theResource.getIdElement();
1287 
1288 			try {
1289 				entity = readEntityLatestVersion(resourceId);
1290 			} catch (ResourceNotFoundException e) {
1291 				return doCreate(theResource, null, thePerformIndexing, new Date(), theRequestDetails);
1292 			}
1293 		}
1294 
1295 		if (resourceId.hasVersionIdPart() && Long.parseLong(resourceId.getVersionIdPart()) != entity.getVersion()) {
1296 			throw new ResourceVersionConflictException("Trying to update " + resourceId + " but this is not the current version");
1297 		}
1298 
1299 		if (resourceId.hasResourceType() && !resourceId.getResourceType().equals(getResourceName())) {
1300 			throw new UnprocessableEntityException(
1301 				"Invalid resource ID[" + entity.getIdDt().toUnqualifiedVersionless() + "] of type[" + entity.getResourceType() + "] - Does not match expected [" + getResourceName() + "]");
1302 		}
1303 
1304 		IBaseResource oldResource = toResource(entity, false);
1305 
1306 		/*
1307 		 * Mark the entity as not deleted - This is also done in the actual updateInternal()
1308 		 * method later on so it usually doesn't matter whether we do it here, but in the
1309 		 * case of a transaction with multiple PUTs we don't get there until later so
1310 		 * having this here means that a transaction can have a reference in one
1311 		 * resource to another resource in the same transaction that is being
1312 		 * un-deleted by the transaction. Wacky use case, sure. But it's real.
1313 		 *
1314 		 * See SystemProviderR4Test#testTransactionReSavesPreviouslyDeletedResources
1315 		 * for a test that needs this.
1316 		 */
1317 		boolean wasDeleted = entity.getDeleted() != null;
1318 		entity.setDeleted(null);
1319 
1320 		/*
1321 		 * If we aren't indexing, that means we're doing this inside a transaction.
1322 		 * The transaction will do the actual storage to the database a bit later on,
1323 		 * after placeholder IDs have been replaced, by calling {@link #updateInternal}
1324 		 * directly. So we just bail now.
1325 		 */
1326 		if (!thePerformIndexing) {
1327 			theResource.setId(entity.getIdDt().getValue());
1328 			DaoMethodOutcome outcome = toMethodOutcome(theRequestDetails, entity, theResource).setCreated(wasDeleted);
1329 			outcome.setPreviousResource(oldResource);
1330 			return outcome;
1331 		}
1332 
1333 		/*
1334 		 * Otherwise, we're not in a transaction
1335 		 */
1336 		ResourceTable savedEntity = updateInternal(theRequestDetails, theResource, thePerformIndexing, theForceUpdateVersion, entity, resourceId, oldResource);
1337 		DaoMethodOutcome outcome = toMethodOutcome(theRequestDetails, savedEntity, theResource).setCreated(wasDeleted);
1338 
1339 		if (!thePerformIndexing) {
1340 			outcome.setId(theResource.getIdElement());
1341 		}
1342 
1343 		String msg = getContext().getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "successfulUpdate", outcome.getId(), w.getMillisAndRestart());
1344 		outcome.setOperationOutcome(createInfoOperationOutcome(msg));
1345 
1346 		ourLog.debug(msg);
1347 		return outcome;
1348 	}
