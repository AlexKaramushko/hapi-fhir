109 	@Transactional(value = Transactional.TxType.NEVER)
110 	@Override
111 	public synchronized void buildExportFiles() {
112 
113 		Optional<BulkExportJobEntity> jobToProcessOpt = myTxTemplate.execute(t -> {
114 			Pageable page = PageRequest.of(0, 1);
115 			Slice<BulkExportJobEntity> submittedJobs = myBulkExportJobDao.findByStatus(page, BulkJobStatusEnum.SUBMITTED);
116 			if (submittedJobs.isEmpty()) {
117 				return Optional.empty();
118 			}
119 			return Optional.of(submittedJobs.getContent().get(0));
120 		});
121 
122 		if (!jobToProcessOpt.isPresent()) {
123 			return;
124 		}
125 
126 		String jobUuid = jobToProcessOpt.get().getJobId();
127 
128 		try {
129 			myTxTemplate.execute(t -> {
130 				processJob(jobUuid);
131 				return null;
132 			});
133 		} catch (Exception e) {
134 			ourLog.error("Failure while preparing bulk export extract", e);
135 			myTxTemplate.execute(t -> {
136 				Optional<BulkExportJobEntity> submittedJobs = myBulkExportJobDao.findByJobId(jobUuid);
137 				if (submittedJobs.isPresent()) {
138 					BulkExportJobEntity jobEntity = submittedJobs.get();
139 					jobEntity.setStatus(BulkJobStatusEnum.ERROR);
140 					jobEntity.setStatusMessage(e.getMessage());
141 					myBulkExportJobDao.save(jobEntity);
142 				}
143 				return null;
144 			});
145 		}
146 
147 	}

154 	@Transactional(value = Transactional.TxType.NEVER)
155 	@Override
156 	public void purgeExpiredFiles() {
157 		Optional<BulkExportJobEntity> jobToDelete = myTxTemplate.execute(t -> {
158 			Pageable page = PageRequest.of(0, 1);
159 			Slice<BulkExportJobEntity> submittedJobs = myBulkExportJobDao.findByExpiry(page, new Date());
160 			if (submittedJobs.isEmpty()) {
161 				return Optional.empty();
162 			}
163 			return Optional.of(submittedJobs.getContent().get(0));
164 		});
165 
166 		if (jobToDelete.isPresent()) {
167 
168 			ourLog.info("Deleting bulk export job: {}", jobToDelete.get());
169 
170 			myTxTemplate.execute(t -> {
171 
172 				BulkExportJobEntity job = myBulkExportJobDao.getOne(jobToDelete.get().getId());
173 				for (BulkExportCollectionEntity nextCollection : job.getCollections()) {
174 					for (BulkExportCollectionFileEntity nextFile : nextCollection.getFiles()) {
175 
176 						ourLog.info("Purging bulk data file: {}", nextFile.getResourceId());
177 						getBinaryDao().delete(toId(nextFile.getResourceId()));
178 						getBinaryDao().forceExpungeInExistingTransaction(toId(nextFile.getResourceId()), new ExpungeOptions().setExpungeDeletedResources(true).setExpungeOldVersions(true), null);
179 						myBulkExportCollectionFileDao.deleteByPid(nextFile.getId());
180 
181 					}
182 
183 					myBulkExportCollectionDao.deleteByPid(nextCollection.getId());
184 				}
185 
186 				ourLog.info("*** ABOUT TO DELETE");
187 				myBulkExportJobDao.deleteByPid(job.getId());
188 				return null;
189 			});
190 
191 			ourLog.info("Finished deleting bulk export job: {}", jobToDelete.get());
192 
193 		}
194 
195 	}

197 	private void processJob(String theJobUuid) {
198 
199 		Optional<BulkExportJobEntity> jobOpt = myBulkExportJobDao.findByJobId(theJobUuid);
200 		if (!jobOpt.isPresent()) {
201 			ourLog.info("Job appears to be deleted");
202 			return;
203 		}
204 
205 		StopWatch jobStopwatch = new StopWatch();
206 		AtomicInteger jobResourceCounter = new AtomicInteger();
207 
208 		BulkExportJobEntity job = jobOpt.get();
209 		ourLog.info("Bulk export starting generation for batch export job: {}", job);
210 
211 		for (BulkExportCollectionEntity nextCollection : job.getCollections()) {
212 
213 			String nextType = nextCollection.getResourceType();
214 			IFhirResourceDao dao = myDaoRegistry.getResourceDao(nextType);
215 
216 			ourLog.info("Bulk export assembling export of type {} for job {}", nextType, theJobUuid);
217 
218 			ISearchBuilder sb = dao.newSearchBuilder();
219 			Class<? extends IBaseResource> nextTypeClass = myContext.getResourceDefinition(nextType).getImplementingClass();
220 			sb.setType(nextTypeClass, nextType);
221 
222 			SearchParameterMap map = new SearchParameterMap();
223 			map.setLoadSynchronous(true);
224 			if (job.getSince() != null) {
225 				map.setLastUpdated(new DateRangeParam(job.getSince(), null));
226 			}
227 
228 			IResultIterator resultIterator = sb.createQuery(map, new SearchRuntimeDetails(null, theJobUuid), null);
229 			storeResultsToFiles(nextCollection, sb, resultIterator, jobResourceCounter, jobStopwatch);
230 
231 
232 		}
233 
234 		job.setStatus(BulkJobStatusEnum.COMPLETE);
235 		updateExpiry(job);
236 		myBulkExportJobDao.save(job);
237 
238 		ourLog.info("Bulk export completed job in {}: {}", jobStopwatch, job);
239 
240 	}

242 	private void storeResultsToFiles(BulkExportCollectionEntity theExportCollection, ISearchBuilder theSearchBuilder, IResultIterator theResultIterator, AtomicInteger theJobResourceCounter, StopWatch theJobStopwatch) {
243 
244 		try (IResultIterator query = theResultIterator) {
245 			if (!query.hasNext()) {
246 				return;
247 			}
248 
249 			AtomicInteger fileCounter = new AtomicInteger(0);
250 			ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
251 			OutputStreamWriter writer = new OutputStreamWriter(outputStream, Constants.CHARSET_UTF8);
252 			IParser parser = myContext.newJsonParser().setPrettyPrint(false);
253 
254 			List<Long> pidsSpool = new ArrayList<>();
255 			List<IBaseResource> resourcesSpool = new ArrayList<>();
256 			while (query.hasNext()) {
257 				pidsSpool.add(query.next());
258 				fileCounter.incrementAndGet();
259 				theJobResourceCounter.incrementAndGet();
260 
261 				if (pidsSpool.size() >= 10 || !query.hasNext()) {
262 
263 					theSearchBuilder.loadResourcesByPid(pidsSpool, Collections.emptyList(), resourcesSpool, false, null);
264 
265 					for (IBaseResource nextFileResource : resourcesSpool) {
266 						parser.encodeResourceToWriter(nextFileResource, writer);
267 						writer.append("\n");
268 					}
269 
270 					pidsSpool.clear();
271 					resourcesSpool.clear();
272 
273 					if (outputStream.size() >= myFileMaxChars || !query.hasNext()) {
274 						Optional<IIdType> createdId = flushToFiles(theExportCollection, fileCounter, outputStream);
275 						createdId.ifPresent(theIIdType -> ourLog.info("Created resource {} for bulk export file containing {} resources of type {} - Total {} resources ({}/sec)", theIIdType.toUnqualifiedVersionless().getValue(), fileCounter.get(), theExportCollection.getResourceType(), theJobResourceCounter.get(), theJobStopwatch.formatThroughput(theJobResourceCounter.get(), TimeUnit.SECONDS)));
276 						fileCounter.set(0);
277 					}
278 
279 				}
280 			}
281 
282 		} catch (IOException e) {
283 			throw new InternalErrorException(e);
284 		}
285 	}

287 	private Optional<IIdType> flushToFiles(BulkExportCollectionEntity theCollection, AtomicInteger theCounter, ByteArrayOutputStream theOutputStream) {
288 		if (theOutputStream.size() > 0) {
289 			IBaseBinary binary = BinaryUtil.newBinary(myContext);
290 			binary.setContentType(Constants.CT_FHIR_NDJSON);
291 			binary.setContent(theOutputStream.toByteArray());
292 
293 			IIdType createdId = getBinaryDao().create(binary).getResource().getIdElement();
294 
295 			BulkExportCollectionFileEntity file = new BulkExportCollectionFileEntity();
296 			theCollection.getFiles().add(file);
297 			file.setCollection(theCollection);
298 			file.setResource(createdId.getIdPart());
299 			myBulkExportCollectionFileDao.saveAndFlush(file);
300 			theOutputStream.reset();
301 
302 			return Optional.of(createdId);
303 		}
304 
305 		return Optional.empty();
306 	}

314 	public void start() {
315 		ourLog.info("Bulk export service starting with refresh interval {}", StopWatch.formatMillis(REFRESH_INTERVAL));
316 		myTxTemplate = new TransactionTemplate(myTxManager);
317 
318 		ScheduledJobDefinition jobDetail = new ScheduledJobDefinition();
319 		jobDetail.setId(BulkDataExportSvcImpl.class.getName());
320 		jobDetail.setJobClass(BulkDataExportSvcImpl.SubmitJob.class);
321 		mySchedulerService.scheduleFixedDelay(REFRESH_INTERVAL, true, jobDetail);
322 	}

324 	@Transactional
325 	@Override
326 	public JobInfo submitJob(String theOutputFormat, Set<String> theResourceTypes, Date theSince, Set<String> theFilters) {
327 		String outputFormat = Constants.CT_FHIR_NDJSON;
328 		if (isNotBlank(theOutputFormat)) {
329 			outputFormat = theOutputFormat;
330 		}
331 		if (!Constants.CTS_NDJSON.contains(outputFormat)) {
332 			throw new InvalidRequestException("Invalid output format: " + theOutputFormat);
333 		}
334 
335 		StringBuilder requestBuilder = new StringBuilder();
336 		requestBuilder.append("/").append(JpaConstants.OPERATION_EXPORT);
337 		requestBuilder.append("?").append(JpaConstants.PARAM_EXPORT_OUTPUT_FORMAT).append("=").append(escapeUrlParam(outputFormat));
338 		Set<String> resourceTypes = theResourceTypes;
339 		if (resourceTypes != null) {
340 			requestBuilder.append("&").append(JpaConstants.PARAM_EXPORT_TYPE).append("=").append(String.join(",", resourceTypes));
341 		}
342 		Date since = theSince;
343 		if (since != null) {
344 			requestBuilder.append("&").append(JpaConstants.PARAM_EXPORT_SINCE).append("=").append(new InstantType(since).setTimeZoneZulu(true).getValueAsString());
345 		}
346 		if (theFilters != null && theFilters.size() > 0) {
347 			requestBuilder.append("&").append(JpaConstants.PARAM_EXPORT_TYPE).append("=").append(String.join(",", theFilters));
348 		}
349 		String request = requestBuilder.toString();
350 
351 		Date cutoff = DateUtils.addMilliseconds(new Date(), -myReuseBulkExportForMillis);
352 		Pageable page = PageRequest.of(0, 10);
353 		Slice<BulkExportJobEntity> existing = myBulkExportJobDao.findExistingJob(page, request, cutoff, BulkJobStatusEnum.ERROR);
354 		if (existing.isEmpty() == false) {
355 			return toSubmittedJobInfo(existing.iterator().next());
356 		}
357 
358 		if (resourceTypes != null && resourceTypes.contains("Binary")) {
359 			String msg = myContext.getLocalizer().getMessage(BulkDataExportSvcImpl.class, "onlyBinarySelected");
360 			throw new InvalidRequestException(msg);
361 		}
362 
363 		if (resourceTypes == null || resourceTypes.isEmpty()) {
364 			// This is probably not a useful default, but having the default be "download the whole
365 			// server" seems like a risky default too. We'll deal with that by having the default involve
366 			// only returning a small time span
367 			resourceTypes = myContext.getResourceNames();
368 			if (since == null) {
369 				since = DateUtils.addDays(new Date(), -1);
370 			}
371 		}
372 
373 		resourceTypes =
374 			resourceTypes
375 				.stream()
376 				.filter(t -> !"Binary".equals(t))
377 				.collect(Collectors.toSet());
378 
379 		BulkExportJobEntity job = new BulkExportJobEntity();
380 		job.setJobId(UUID.randomUUID().toString());
381 		job.setStatus(BulkJobStatusEnum.SUBMITTED);
382 		job.setSince(since);
383 		job.setCreated(new Date());
384 		job.setRequest(request);
385 
386 		updateExpiry(job);
387 		myBulkExportJobDao.save(job);
388 
389 		for (String nextType : resourceTypes) {
390 			if (!myDaoRegistry.isResourceTypeSupported(nextType)) {
391 				String msg = myContext.getLocalizer().getMessage(BulkDataExportSvcImpl.class, "unknownResourceType", nextType);
392 				throw new InvalidRequestException(msg);
393 			}
394 
395 			BulkExportCollectionEntity collection = new BulkExportCollectionEntity();
396 			collection.setJob(job);
397 			collection.setResourceType(nextType);
398 			job.getCollections().add(collection);
399 			myBulkExportCollectionDao.save(collection);
400 		}
401 
402 		ourLog.info("Bulk export job submitted: {}", job.toString());
403 
404 		return toSubmittedJobInfo(job);
405 	}

416 	@Transactional
417 	@Override
418 	public JobInfo getJobStatusOrThrowResourceNotFound(String theJobId) {
419 		BulkExportJobEntity job = myBulkExportJobDao
420 			.findByJobId(theJobId)
421 			.orElseThrow(() -> new ResourceNotFoundException(theJobId));
422 
423 		JobInfo retVal = new JobInfo();
424 		retVal.setJobId(theJobId);
425 		retVal.setStatus(job.getStatus());
426 		retVal.setStatus(job.getStatus());
427 		retVal.setStatusTime(job.getStatusTime());
428 		retVal.setStatusMessage(job.getStatusMessage());
429 		retVal.setRequest(job.getRequest());
430 
431 		if (job.getStatus() == BulkJobStatusEnum.COMPLETE) {
432 			for (BulkExportCollectionEntity nextCollection : job.getCollections()) {
433 				for (BulkExportCollectionFileEntity nextFile : nextCollection.getFiles()) {
434 					retVal.addFile()
435 						.setResourceType(nextCollection.getResourceType())
436 						.setResourceId(toQualifiedBinaryId(nextFile.getResourceId()));
437 				}
438 			}
439 		}
440 
441 		return retVal;
442 	}

459 	public synchronized void cancelAndPurgeAllJobs() {
460 		myTxTemplate.execute(t -> {
461 			ourLog.info("Deleting all files");
462 			myBulkExportCollectionFileDao.deleteAllFiles();
463 			ourLog.info("Deleting all collections");
464 			myBulkExportCollectionDao.deleteAllFiles();

467 			return null;
468 		});
