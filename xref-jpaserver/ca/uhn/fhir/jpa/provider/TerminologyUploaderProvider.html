86  	 * <code>
87  	 * $apply-codesystem-delta-add
88  	 * </code>
89  	 */
90  	@Operation(typeName="CodeSystem", name = JpaConstants.OPERATION_APPLY_CODESYSTEM_DELTA_ADD, idempotent = false, returnParameters = {
91  	})
92  	public IBaseParameters applyCodeSystemDeltaAdd(
93  		HttpServletRequest theServletRequest,
94  		@OperationParam(name = PARENT_CODE, min = 0, max = 1) IPrimitiveType<String> theParentCode,
95  		@OperationParam(name = VALUE, min = 1, max = 1) IBaseResource theValue,
96  		RequestDetails theRequestDetails
97  	) {
98  
99  		startRequest(theServletRequest);
100 		try {
101 
102 			CodeSystem value;
103 			if (theValue instanceof CodeSystem) {
104 				value = (CodeSystem) theValue;
105 			} else if (theValue instanceof org.hl7.fhir.dstu3.model.CodeSystem) {
106 				value = VersionConvertor_30_40.convertCodeSystem((org.hl7.fhir.dstu3.model.CodeSystem) theValue);
107 			} else {
108 				throw new InvalidRequestException("Value must be present and be a CodeSystem");
109 			}
110 
111 			String system = value.getUrl();
112 			String parentCode = theParentCode != null ? theParentCode.getValue() : null;
113 
114 			AtomicInteger counter = myTerminologySvc.applyDeltaCodesystemsAdd(system, parentCode, value);
115 
116 			IBaseParameters retVal = ParametersUtil.newInstance(myCtx);
117 			ParametersUtil.addParameterToParametersBoolean(myCtx, retVal, "success", true);
118 			ParametersUtil.addParameterToParametersInteger(myCtx, retVal, "addedConcepts", counter.get());
119 			return retVal;
120 
121 		} finally {
122 			endRequest(theServletRequest);
123 		}
124 
125 	}

169 	/**
170 	 * <code>
171 	 * $upload-external-codesystem
172 	 * </code>
173 	 */
174 	@Operation(typeName="CodeSystem", name = JpaConstants.OPERATION_UPLOAD_EXTERNAL_CODE_SYSTEM, idempotent = false, returnParameters = {
175 //		@OperationParam(name = "conceptCount", type = IntegerType.class, min = 1)
176 	})
177 	public IBaseParameters uploadExternalCodeSystem(
178 		HttpServletRequest theServletRequest,
179 		@OperationParam(name = "url", min = 1, typeName = "uri") IPrimitiveType<String> theCodeSystemUrl,
180 		@OperationParam(name = "contentMode", min = 0, typeName = "code") IPrimitiveType<String> theContentMode,
181 		@OperationParam(name = "localfile", min = 1, max = OperationParam.MAX_UNLIMITED, typeName = "string") List<IPrimitiveType<String>> theLocalFile,
182 		@OperationParam(name = "package", min = 0, max = OperationParam.MAX_UNLIMITED, typeName = "attachment") List<ICompositeType> thePackage,
183 		RequestDetails theRequestDetails
184 	) {
185 
186 		startRequest(theServletRequest);
187 
188 		if (theLocalFile == null || theLocalFile.size() == 0) {
189 			if (thePackage == null || thePackage.size() == 0) {
190 				throw new InvalidRequestException("No 'localfile' or 'package' parameter, or package had no data");
191 			}
192 			for (ICompositeType next : thePackage) {
193 				ValidateUtil.isTrueOrThrowInvalidRequest(myCtx.getElementDefinition(next.getClass()).getName().equals("Attachment"), "Package must be of type Attachment");
194 			}
195 		}
196 
197 		try {
198 			List<IHapiTerminologyLoaderSvc.FileDescriptor> localFiles = new ArrayList<>();
199 			if (theLocalFile != null && theLocalFile.size() > 0) {
200 				for (IPrimitiveType<String> nextLocalFile : theLocalFile) {
201 					if (isNotBlank(nextLocalFile.getValue())) {
202 						ourLog.info("Reading in local file: {}", nextLocalFile.getValue());
203 						File nextFile = new File(nextLocalFile.getValue());
204 						if (!nextFile.exists() || !nextFile.isFile()) {
205 							throw new InvalidRequestException("Unknown file: " + nextFile.getName());
206 						}
207 						localFiles.add(new IHapiTerminologyLoaderSvc.FileDescriptor() {
208 							@Override
209 							public String getFilename() {
210 								return nextFile.getAbsolutePath();
211 							}
212 
213 							@Override
214 							public InputStream getInputStream() {
215 								try {
216 									return new FileInputStream(nextFile);
217 								} catch (FileNotFoundException theE) {
218 									throw new InternalErrorException(theE);
219 								}
220 							}
221 						});
222 					}
223 				}
224 			}
225 
226 			if (thePackage != null) {
227 				for (ICompositeType nextPackage : thePackage) {
228 					final String url = AttachmentUtil.getOrCreateUrl(myCtx, nextPackage).getValueAsString();
229 
230 					if (isBlank(url)) {
231 						throw new UnprocessableEntityException("Package is missing mandatory url element");
232 					}
233 
234 					localFiles.add(new IHapiTerminologyLoaderSvc.FileDescriptor() {
235 						@Override
236 						public String getFilename() {
237 							return url;
238 						}
239 
240 						@Override
241 						public InputStream getInputStream() {
242 							byte[] data = AttachmentUtil.getOrCreateData(myCtx, nextPackage).getValue();
243 							return new ByteArrayInputStream(data);
244 						}
245 					});
246 				}
247 			}
248 
249 			String codeSystemUrl = theCodeSystemUrl != null ? theCodeSystemUrl.getValue() : null;
250 			codeSystemUrl = defaultString(codeSystemUrl);
251 
252 			String contentMode = theContentMode != null ? theContentMode.getValue() : null;
253 			UploadStatistics stats;
254 			if ("custom".equals(contentMode)) {
255 				stats = myTerminologyLoaderSvc.loadCustom(codeSystemUrl, localFiles, theRequestDetails);
256 			} else {
257 				switch (codeSystemUrl) {
258 					case IHapiTerminologyLoaderSvc.SCT_URI:
259 						stats = myTerminologyLoaderSvc.loadSnomedCt(localFiles, theRequestDetails);
260 						break;
261 					case IHapiTerminologyLoaderSvc.LOINC_URI:
262 						stats = myTerminologyLoaderSvc.loadLoinc(localFiles, theRequestDetails);
263 						break;
264 					case IHapiTerminologyLoaderSvc.IMGTHLA_URI:
265 						stats = myTerminologyLoaderSvc.loadImgthla(localFiles, theRequestDetails);
266 						break;
267 					default:
268 						throw new InvalidRequestException("Unknown URL: " + codeSystemUrl);
269 				}
270 			}
271 
272 			IBaseParameters retVal = ParametersUtil.newInstance(myCtx);
273 			ParametersUtil.addParameterToParametersBoolean(myCtx, retVal, "success", true);
274 			ParametersUtil.addParameterToParametersInteger(myCtx, retVal, CONCEPT_COUNT, stats.getConceptCount());
275 			ParametersUtil.addParameterToParametersReference(myCtx, retVal, TARGET, stats.getTarget().getValue());
276 
277 			return retVal;
278 		} finally {
279 			endRequest(theServletRequest);
280 		}
281 	}
