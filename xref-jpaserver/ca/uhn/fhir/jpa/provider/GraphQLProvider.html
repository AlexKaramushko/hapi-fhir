53  public class GraphQLProvider {
54  	private final Supplier<IGraphQLEngine> engineFactory;
55  	private final IGraphQLStorageServices myStorageServices;
56  	private Logger ourLog = LoggerFactory.getLogger(GraphQLProvider.class);
57  
58  	/**
59  	 * Constructor which uses a default context and validation support object
60  	 *
61  	 * @param theStorageServices The storage services (this object will be used to retrieve various resources as required by the GraphQL engine)
62  	 */
63  	public GraphQLProvider(IGraphQLStorageServices theStorageServices) {
64  		this(FhirContext.forR4(), null, theStorageServices);
65  	}
66  
67  	/**
68  	 * Constructor which uses the given worker context
69  	 *
70  	 * @param theFhirContext       The HAPI FHIR Context object
71  	 * @param theValidationSupport The HAPI Validation Support object, or null
72  	 * @param theStorageServices   The storage services (this object will be used to retrieve various resources as required by the GraphQL engine)
73  	 */
74  	public GraphQLProvider(@Nonnull FhirContext theFhirContext, @Nullable IContextValidationSupport theValidationSupport, @Nonnull IGraphQLStorageServices theStorageServices) {
75  		Validate.notNull(theFhirContext, "theFhirContext must not be null");
76  		Validate.notNull(theStorageServices, "theStorageServices must not be null");
77  
78  		switch (theFhirContext.getVersion().getVersion()) {
79  			case DSTU3: {
80  				IValidationSupport validationSupport = (IValidationSupport) theValidationSupport;
81  				validationSupport = ObjectUtils.defaultIfNull(validationSupport, new org.hl7.fhir.dstu3.hapi.ctx.DefaultProfileValidationSupport());
82  				org.hl7.fhir.dstu3.hapi.ctx.HapiWorkerContext workerContext = new org.hl7.fhir.dstu3.hapi.ctx.HapiWorkerContext(theFhirContext, validationSupport);
83  				engineFactory = () -> new org.hl7.fhir.dstu3.utils.GraphQLEngine(workerContext);
84  				break;
85  			}
86  			case R4: {
87  				org.hl7.fhir.r4.hapi.ctx.IValidationSupport validationSupport = (org.hl7.fhir.r4.hapi.ctx.IValidationSupport) theValidationSupport;
88  				validationSupport = ObjectUtils.defaultIfNull(validationSupport, new org.hl7.fhir.r4.hapi.ctx.DefaultProfileValidationSupport());
89  				org.hl7.fhir.r4.hapi.ctx.HapiWorkerContext workerContext = new org.hl7.fhir.r4.hapi.ctx.HapiWorkerContext(theFhirContext, validationSupport);
90  				engineFactory = () -> new org.hl7.fhir.r4.utils.GraphQLEngine(workerContext);
91  				break;
92  			}
93  			case R5: {
94  				org.hl7.fhir.r5.hapi.ctx.IValidationSupport validationSupport = (org.hl7.fhir.r5.hapi.ctx.IValidationSupport) theValidationSupport;
95  				validationSupport = ObjectUtils.defaultIfNull(validationSupport, new org.hl7.fhir.r5.hapi.ctx.DefaultProfileValidationSupport());
96  				org.hl7.fhir.r5.hapi.ctx.HapiWorkerContext workerContext = new org.hl7.fhir.r5.hapi.ctx.HapiWorkerContext(theFhirContext, validationSupport);
97  				engineFactory = () -> new org.hl7.fhir.r5.utils.GraphQLEngine(workerContext);
98  				break;
99  			}
100 			case DSTU2:
101 			case DSTU2_HL7ORG:
102 			case DSTU2_1:
103 			default: {
104 				throw new UnsupportedOperationException("GraphQL not supported for version: " + theFhirContext.getVersion().getVersion());
105 			}
106 		}
107 
108 		myStorageServices = theStorageServices;
109 	}
110 
111 	@GraphQL
112 	public String processGraphQlRequest(ServletRequestDetails theRequestDetails, @IdParam IIdType theId, @GraphQLQuery String theQuery) {
113 
114 		IGraphQLEngine engine = engineFactory.get();
115 		engine.setAppInfo(theRequestDetails);
116 		engine.setServices(myStorageServices);
117 		try {
118 			engine.setGraphQL(Parser.parse(theQuery));
119 		} catch (Exception theE) {
120 			throw new InvalidRequestException("Unable to parse GraphQL Expression: " + theE.toString());
121 		}
122 
123 		try {
124 
125 			if (theId != null) {
126 				IBaseResource focus = myStorageServices.lookup(theRequestDetails, theId.getResourceType(), theId.getIdPart());
127 				engine.setFocus(focus);
128 			}
129 			engine.execute();
130 
131 			StringBuilder outputBuilder = new StringBuilder();
132 			ObjectValue output = engine.getOutput();
133 			output.write(outputBuilder, 0, "\n");
134 
135 			return outputBuilder.toString();
136 
137 		} catch (Exception e) {
138 			StringBuilder b = new StringBuilder();
139 			b.append("Unable to execute GraphQL Expression: ");
140 			int statusCode = 500;
141 			if (e instanceof BaseServerResponseException) {
142 				b.append("HTTP ");
143 				statusCode = ((BaseServerResponseException) e).getStatusCode();
144 				b.append(statusCode);
145 				b.append(" ");
146 			} else {
147 				// This means it's a bug, so let's log
148 				ourLog.error("Failure during GraphQL processing", e);
149 			}
150 			b.append(e.getMessage());
151 			throw new UnclassifiedServerFailureException(statusCode, b.toString());
152 		}
153 	}
154 
155 	@Initialize
156 	public void initialize(RestfulServer theServer) {
157 		ourLog.trace("Initializing GraphQL provider");
158 		if (!theServer.getFhirContext().getVersion().getVersion().isEqualOrNewerThan(FhirVersionEnum.DSTU3)) {
159 			throw new ConfigurationException("Can not use " + getClass().getName() + " provider on server with FHIR " + theServer.getFhirContext().getVersion().getVersion().name() + " context");
160 		}
161 	}
