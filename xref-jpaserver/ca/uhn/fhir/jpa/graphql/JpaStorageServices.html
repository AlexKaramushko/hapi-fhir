51  	private IFhirResourceDao<? extends IBaseResource> getDao(String theResourceType) {
52  		RuntimeResourceDefinition typeDef = getContext().getResourceDefinition(theResourceType);
53  		return getDao(typeDef.getImplementingClass());
54  	}

56  	@Transactional(propagation = Propagation.NEVER)
57  	@Override
58  	public void listResources(Object theAppInfo, String theType, List<Argument> theSearchParams, List<Resource> theMatches) throws FHIRException {
59  
60  		RuntimeResourceDefinition typeDef = getContext().getResourceDefinition(theType);
61  		IFhirResourceDao<? extends IBaseResource> dao = getDao(typeDef.getImplementingClass());
62  
63  		SearchParameterMap params = new SearchParameterMap();
64  		params.setLoadSynchronousUpTo(500);
65  
66  		for (Argument nextArgument : theSearchParams) {
67  
68  			RuntimeSearchParam searchParam = mySearchParamRegistry.getSearchParamByName(typeDef, nextArgument.getName());
69  
70  			for (Value nextValue : nextArgument.getValues()) {
71  				String value = nextValue.getValue();
72  
73  				IQueryParameterType param = null;
74  				switch (searchParam.getParamType()) {
75  					case NUMBER:
76  						param = new NumberParam(value);
77  						break;
78  					case DATE:
79  						param = new DateParam(value);
80  						break;
81  					case STRING:
82  						param = new StringParam(value);
83  						break;
84  					case TOKEN:
85  						param = new TokenParam(null, value);
86  						break;
87  					case REFERENCE:
88  						param = new ReferenceParam(value);
89  						break;
90  					case COMPOSITE:
91  						throw new InvalidRequestException("Composite parameters are not yet supported in GraphQL");
92  					case QUANTITY:
93  						param = new QuantityParam(value);
94  						break;
95  					case URI:
96  						break;
97  					case HAS:
98  						break;
99  				}
100 
101 				params.add(nextArgument.getName(), param);
102 			}
103 		}
104 
105 		IBundleProvider response = dao.search(params);
106 		int size = response.size();
107 		if (response.preferredPageSize() != null && response.preferredPageSize() < size) {
108 			size = response.preferredPageSize();
109 		}
110 
111 		for (IBaseResource next : response.getResources(0, size)) {
112 			theMatches.add((Resource) next);
113 		}
114 
115 	}

117 	@Transactional(propagation = Propagation.REQUIRED)
118 	@Override
119 	public Resource lookup(Object theAppInfo, String theType, String theId) throws FHIRException {
120 		IIdType refId = getContext().getVersion().newIdType();
121 		refId.setValue(theType + "/" + theId);
122 		IFhirResourceDao<? extends IBaseResource> dao = getDao(theType);
123 		BaseHasResource id = dao.readEntity(refId);
124 
125 		return (Resource) toResource(id, false);
126 	}
