23  import java.util.Collections;
24  import java.util.List;
25  
26  import org.apache.commons.lang3.StringUtils;
27  
28  import ca.uhn.fhir.context.FhirContext;
29  import ca.uhn.fhir.model.api.IQueryParameterOr;
30  import ca.uhn.fhir.model.api.IQueryParameterType;
31  import ca.uhn.fhir.rest.api.QualifiedParamList;
32  import ca.uhn.fhir.rest.param.ParameterUtil;
33  import ca.uhn.fhir.rest.server.exceptions.InternalErrorException;
34  import ca.uhn.fhir.rest.server.exceptions.InvalidRequestException;
35  
36  public final class QueryParameterTypeBinder extends BaseBinder<IQueryParameterType> implements IParamBinder<IQueryParameterType> {
37  
38  	public QueryParameterTypeBinder(Class<? extends IQueryParameterType> theType, List<Class<? extends IQueryParameterType>> theCompositeTypes) {
39  		super(theType, theCompositeTypes);
40  	}
41  
42  	@SuppressWarnings("unchecked")
43  	@Override
44  	public List<IQueryParameterOr<?>> encode(FhirContext theContext, IQueryParameterType theValue) throws InternalErrorException {
45  		IQueryParameterType param = theValue;
46  		List<?> retVal = Collections.singletonList(ParameterUtil.singleton(param, null));
47  		return (List<IQueryParameterOr<?>>) retVal;
48  	}
49  
50  	@Override
51  	public IQueryParameterType parse(FhirContext theContext, String theParamName, List<QualifiedParamList> theParams) throws InternalErrorException, InvalidRequestException {
52  		String value = theParams.get(0).get(0);
53  		if (StringUtils.isBlank(value)) {
54  			return null;
55  		}
56  		
57  		IQueryParameterType dt = super.newInstance();
58  
59  		if (theParams.size() == 0 || theParams.get(0).size() == 0) {
60  			return dt;
61  		}
62  		if (theParams.size() > 1 || theParams.get(0).size() > 1) {
63  			throw new InvalidRequestException("Multiple values detected for non-repeatable parameter '" + theParamName + "'. This server is not configured to allow multiple (AND/OR) values for this param.");
64  		}
65  
66  		dt.setValueAsQueryToken(theContext, theParamName, theParams.get(0).getQualifier(), value);
67  		return dt;
68  	}
69  
70  }
