23  import org.slf4j.Logger;
24  import org.slf4j.event.Level;
25  
26  /**
27   * Utility to fill a glaring gap in SLF4j's API - The fact that you can't
28   * specify a log level at runtime.
29   *
30   * See here for a discussion:
31   * https://jira.qos.ch/browse/SLF4J-124
32   */
33  public class LogUtil {
34  
35  	public static void log(Logger theLogger, Level theLevel, String theMessage, Object... theArgs) {
36  		switch (theLevel) {
37  			case TRACE:
38  				theLogger.trace(theMessage, theArgs);
39  				break;
40  			case DEBUG:
41  				theLogger.debug(theMessage, theArgs);
42  				break;
43  			case INFO:
44  				theLogger.info(theMessage, theArgs);
45  				break;
46  			case WARN:
47  				theLogger.warn(theMessage, theArgs);
48  				break;
49  			case ERROR:
50  				theLogger.error(theMessage, theArgs);
51  				break;
52  		}
53  	}
54  
55  }
