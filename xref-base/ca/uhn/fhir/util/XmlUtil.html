1778 	/**
1779 	 * Parses an XML string into a set of StAX events
1780 	 */
1781 	public static List<XMLEvent> parse(String theValue) {
1782 		if (isBlank(theValue)) {
1783 			return Collections.emptyList();
1784 		}
1785 
1786 		String val = theValue.trim();
1787 		if (!val.startsWith("<")) {
1788 			val = XhtmlDt.DIV_OPEN_FIRST + val + "</div>";
1789 		}
1790 		boolean hasProcessingInstruction = val.startsWith("<?");
1791 		if (hasProcessingInstruction && val.endsWith("?>")) {
1792 			return null;
1793 		}
1794 
1795 
1796 		try {
1797 			ArrayList<XMLEvent> value = new ArrayList<>();
1798 			StringReader reader = new StringReader(val);
1799 			XMLEventReader er = XmlUtil.createXmlReader(reader);
1800 			boolean first = true;
1801 			while (er.hasNext()) {
1802 				XMLEvent next = er.nextEvent();
1803 				if (first) {
1804 					first = false;
1805 					continue;
1806 				}
1807 				if (er.hasNext()) {
1808 					// don't add the last event
1809 					value.add(next);
1810 				}
1811 			}
1812 			return value;
1813 
1814 		} catch (XMLStreamException e) {
1815 			throw new DataFormatException("String does not appear to be valid XML/XHTML (error is \"" + e.getMessage() + "\"): " + theValue, e);
1816 		} catch (FactoryConfigurationError e) {
1817 			throw new ConfigurationException(e);
1818 		}
1819 	}
