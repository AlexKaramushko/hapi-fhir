39  /**
40   * Writes an iterator over triples to N-Triples or Turtle
41   * in a streaming fashion, that is, without needing to hold
42   * the entire thing in memory.
43   * <p>
44   * Instances are single-use.
45   * <p>
46   * There doesn't seem to be a pre-packaged version of this
47   * functionality in Jena/ARQ that doesn't require a Graph or Model.
48   */
49  public class StreamingRDFWriter {
50  	private final OutputStream out;
51  	private final Iterator<Triple> triples;
52  	private int dedupWindowSize = 10000;
53  
54  	public StreamingRDFWriter(OutputStream out, Iterator<Triple> triples) {
55  		this.out = out;
56  		this.triples = triples;
57  	}
58  
59  	public void setDedupWindowSize(int newSize) {
60  		this.dedupWindowSize = newSize;
61  	}
62  
63  	public void writeNTriples() {
64  		StreamRDF writer = new WriterStreamRDFPlain(new IndentedWriter(out));
65  		if (dedupWindowSize > 0) {
66  			writer = new StreamRDFDedup(writer, dedupWindowSize);
67  		}
68  		writer.start();
69  		StreamOps.sendTriplesToStream(triples, writer);
70  		writer.finish();
71  	}
72  
73  	public void writeTurtle(String baseIRI, PrefixMapping prefixes, boolean writeBase) {
74  		// Auto-register RDF prefix so that rdf:type is displayed well
75  		// All other prefixes come from the query and should be as author intended
76  		prefixes = ensureRDFPrefix(prefixes);
77  
78  		if (writeBase) {
79  			// Jena's streaming Turtle writers don't output base even if it is provided,
80  			// so we write it directly.
81  			IndentedWriter w = new IndentedWriter(out);
82  			RiotLib.writeBase(w, baseIRI);
83  			w.flush();
84  		}
85  
86  		StreamRDF writer = new WriterStreamRDFBlocks(out);
87  		if (dedupWindowSize > 0) {
88  			writer = new StreamRDFDedup(writer, dedupWindowSize);
89  		}
90  		writer.start();
91  		writer.base(baseIRI);
92  		for (Entry<String, String> e : prefixes.getNsPrefixMap().entrySet()) {
93  			writer.prefix(e.getKey(), e.getValue());
94  		}
95  		StreamOps.sendTriplesToStream(triples, writer);
96  		writer.finish();
97  	}
98  
99  	private PrefixMapping ensureRDFPrefix(PrefixMapping prefixes) {
100 		// Some prefix already registered for the RDF namespace -- good enough
101 		if (prefixes.getNsURIPrefix(RDF.getURI()) != null) return prefixes;
102 		// rdf: is registered to something else -- give up
103 		if (prefixes.getNsPrefixURI("rdf") != null) return prefixes;
104 		// Register rdf:
105 		PrefixMapping newPrefixes = new PrefixMappingImpl();
106 		newPrefixes.setNsPrefixes(prefixes);
107 		newPrefixes.setNsPrefix("rdf", RDF.getURI());
108 		return newPrefixes;
109 	}
110 }
