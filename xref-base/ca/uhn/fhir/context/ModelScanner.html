92  	private void init(Map<Class<? extends IBase>, BaseRuntimeElementDefinition<?>> theExistingDefinitions, Set<Class<? extends IBase>> theTypesToScan) {
93  		if (theExistingDefinitions != null) {
94  			myClassToElementDefinitions.putAll(theExistingDefinitions);
95  		}
96  
97  		int startSize = myClassToElementDefinitions.size();
98  		long start = System.currentTimeMillis();
99  		Map<String, Class<? extends IBaseResource>> resourceTypes = myNameToResourceType;
100 
101 		Set<Class<? extends IBase>> typesToScan = theTypesToScan;
102 		myVersionTypes = scanVersionPropertyFile(typesToScan, resourceTypes, myVersion, myClassToElementDefinitions);
103 
104 		do {
105 			for (Class<? extends IBase> nextClass : typesToScan) {
106 				scan(nextClass);
107 			}
108 			myScanAlso.removeIf(theClass -> myClassToElementDefinitions.containsKey(theClass));
109 			typesToScan.clear();
110 			typesToScan.addAll(myScanAlso);
111 			myScanAlso.clear();
112 		} while (!typesToScan.isEmpty());
113 
114 		for (Entry<Class<? extends IBase>, BaseRuntimeElementDefinition<?>> nextEntry : myClassToElementDefinitions.entrySet()) {
115 			if (theExistingDefinitions != null && theExistingDefinitions.containsKey(nextEntry.getKey())) {
116 				continue;
117 			}
118 			BaseRuntimeElementDefinition<?> next = nextEntry.getValue();
119 
120 			boolean deferredSeal = false;
121 			if (myContext.getPerformanceOptions().contains(PerformanceOptionsEnum.DEFERRED_MODEL_SCANNING)) {
122 				if (next instanceof BaseRuntimeElementCompositeDefinition) {
123 					deferredSeal = true;
124 				}
125 			}
126 			if (!deferredSeal) {
127 				next.sealAndInitialize(myContext, myClassToElementDefinitions);
128 			}
129 		}
130 
131 		myRuntimeChildUndeclaredExtensionDefinition = new RuntimeChildUndeclaredExtensionDefinition();
132 		myRuntimeChildUndeclaredExtensionDefinition.sealAndInitialize(myContext, myClassToElementDefinitions);
133 
134 		long time = System.currentTimeMillis() - start;
135 		int size = myClassToElementDefinitions.size() - startSize;
136 		ourLog.debug("Done scanning FHIR library, found {} model entries in {}ms", size, time);
137 	}

143 	void scan(Class<? extends IBase> theClass) throws ConfigurationException {
144 		BaseRuntimeElementDefinition<?> existingDef = myClassToElementDefinitions.get(theClass);
145 		if (existingDef != null) {
146 			return;
147 		}
148 
149 		ResourceDef resourceDefinition = pullAnnotation(theClass, ResourceDef.class);
150 		if (resourceDefinition != null) {
151 			if (!IBaseResource.class.isAssignableFrom(theClass)) {
152 				throw new ConfigurationException(
153 					"Resource type contains a @" + ResourceDef.class.getSimpleName() + " annotation but does not implement " + IResource.class.getCanonicalName() + ": " + theClass.getCanonicalName());
154 			}
155 			@SuppressWarnings("unchecked")
156 			Class<? extends IBaseResource> resClass = (Class<? extends IBaseResource>) theClass;
157 			scanResource(resClass, resourceDefinition);
158 			return;
159 		}
160 
161 		DatatypeDef datatypeDefinition = pullAnnotation(theClass, DatatypeDef.class);
162 		if (datatypeDefinition != null) {
163 			if (ICompositeType.class.isAssignableFrom(theClass)) {
164 				@SuppressWarnings("unchecked")
165 				Class<? extends ICompositeType> resClass = (Class<? extends ICompositeType>) theClass;
166 				scanCompositeDatatype(resClass, datatypeDefinition);
167 			} else if (IPrimitiveType.class.isAssignableFrom(theClass)) {
168 				@SuppressWarnings({"unchecked"})
169 				Class<? extends IPrimitiveType<?>> resClass = (Class<? extends IPrimitiveType<?>>) theClass;
170 				scanPrimitiveDatatype(resClass, datatypeDefinition);
171 			}
172 
173 			return;
174 		}
175 
176 		Block blockDefinition = pullAnnotation(theClass, Block.class);
177 
178 		if (blockDefinition != null) {
179 			if (IResourceBlock.class.isAssignableFrom(theClass) || IBaseBackboneElement.class.isAssignableFrom(theClass) || IBaseDatatypeElement.class.isAssignableFrom(theClass)) {
180 				scanBlock(theClass);
181 			} else {
182 				throw new ConfigurationException(
183 					"Type contains a @" + Block.class.getSimpleName() + " annotation but does not implement " + IResourceBlock.class.getCanonicalName() + ": " + theClass.getCanonicalName());
184 			}
185 		}
186 
187 		if (blockDefinition == null
188 //Redundant	checking && datatypeDefinition == null && resourceDefinition == null
189 		) {
190 			throw new ConfigurationException("Resource class[" + theClass.getName() + "] does not contain any valid HAPI-FHIR annotations");
191 		}
192 	}

194 	private void scanBlock(Class<? extends IBase> theClass) {
195 		ourLog.debug("Scanning resource block class: {}", theClass.getName());
196 
197 		String resourceName = theClass.getCanonicalName();
198 
199 		// Just in case someone messes up when upgrading from DSTU2
200 		if (myContext.getVersion().getVersion().isEqualOrNewerThan(FhirVersionEnum.DSTU3)) {
201 			if (BaseIdentifiableElement.class.isAssignableFrom(theClass)) {
202 				throw new ConfigurationException("@Block class for version " + myContext.getVersion().getVersion().name() + " should not extend " + BaseIdentifiableElement.class.getSimpleName() + ": " + theClass.getName());
203 			}
204 		}
205 
206 		RuntimeResourceBlockDefinition blockDef = new RuntimeResourceBlockDefinition(resourceName, theClass, isStandardType(theClass), myContext, myClassToElementDefinitions);
207 		blockDef.populateScanAlso(myScanAlso);
208 
209 		myClassToElementDefinitions.put(theClass, blockDef);
210 	}

212 	private void scanCompositeDatatype(Class<? extends ICompositeType> theClass, DatatypeDef theDatatypeDefinition) {
213 		ourLog.debug("Scanning datatype class: {}", theClass.getName());
214 
215 		RuntimeCompositeDatatypeDefinition elementDef;
216 		if (theClass.equals(ExtensionDt.class)) {
217 			elementDef = new RuntimeExtensionDtDefinition(theDatatypeDefinition, theClass, true, myContext, myClassToElementDefinitions);
218 			// } else if (IBaseMetaType.class.isAssignableFrom(theClass)) {
219 			// resourceDef = new RuntimeMetaDefinition(theDatatypeDefinition, theClass, isStandardType(theClass));
220 		} else {
221 			elementDef = new RuntimeCompositeDatatypeDefinition(theDatatypeDefinition, theClass, isStandardType(theClass), myContext, myClassToElementDefinitions);
222 		}
223 		myClassToElementDefinitions.put(theClass, elementDef);
224 		myNameToElementDefinitions.put(elementDef.getName().toLowerCase(), elementDef);
225 
226 		/*
227 		 * See #423:
228 		 * If the type contains a field that has a custom type, we want to make
229 		 * sure that this type gets scanned as well
230 		 */
231 		elementDef.populateScanAlso(myScanAlso);
232 	}

234 	private String scanPrimitiveDatatype(Class<? extends IPrimitiveType<?>> theClass, DatatypeDef theDatatypeDefinition) {
235 		ourLog.debug("Scanning resource class: {}", theClass.getName());
236 
237 		String resourceName = theDatatypeDefinition.name();
238 		if (isBlank(resourceName)) {
239 			throw new ConfigurationException("Resource type @" + ResourceDef.class.getSimpleName() + " annotation contains no resource name: " + theClass.getCanonicalName());
240 		}
241 
242 		BaseRuntimeElementDefinition<?> elementDef;
243 		if (theClass.equals(XhtmlDt.class)) {
244 			@SuppressWarnings("unchecked")
245 			Class<XhtmlDt> clazz = (Class<XhtmlDt>) theClass;
246 			elementDef = new RuntimePrimitiveDatatypeNarrativeDefinition(resourceName, clazz, isStandardType(clazz));
247 		} else if (IBaseXhtml.class.isAssignableFrom(theClass)) {
248 			@SuppressWarnings("unchecked")
249 			Class<? extends IBaseXhtml> clazz = (Class<? extends IBaseXhtml>) theClass;
250 			elementDef = new RuntimePrimitiveDatatypeXhtmlHl7OrgDefinition(resourceName, clazz, isStandardType(clazz));
251 		} else if (IIdType.class.isAssignableFrom(theClass)) {
252 			elementDef = new RuntimeIdDatatypeDefinition(theDatatypeDefinition, theClass, isStandardType(theClass));
253 		} else {
254 			elementDef = new RuntimePrimitiveDatatypeDefinition(theDatatypeDefinition, theClass, isStandardType(theClass));
255 		}
256 		myClassToElementDefinitions.put(theClass, elementDef);
257 		if (!theDatatypeDefinition.isSpecialization()) {
258 			if (myVersion.isRi() && IDatatype.class.isAssignableFrom(theClass)) {
259 				ourLog.debug("Not adding non RI type {} to RI context", theClass);
260 			} else if (!myVersion.isRi() && !IDatatype.class.isAssignableFrom(theClass)) {
261 				ourLog.debug("Not adding RI type {} to non RI context", theClass);
262 			} else {
263 				myNameToElementDefinitions.put(resourceName, elementDef);
264 			}
265 		}
266 
267 		return resourceName;
268 	}

270 	private String scanResource(Class<? extends IBaseResource> theClass, ResourceDef resourceDefinition) {
271 		ourLog.debug("Scanning resource class: {}", theClass.getName());
272 
273 		boolean primaryNameProvider = true;
274 		String resourceName = resourceDefinition.name();
275 		if (isBlank(resourceName)) {
276 			Class<?> parent = theClass.getSuperclass();
277 			primaryNameProvider = false;
278 			while (parent.equals(Object.class) == false && isBlank(resourceName)) {
279 				ResourceDef nextDef = pullAnnotation(parent, ResourceDef.class);
280 				if (nextDef != null) {
281 					resourceName = nextDef.name();
282 				}
283 				parent = parent.getSuperclass();
284 			}
285 			if (isBlank(resourceName)) {
286 				throw new ConfigurationException("Resource type @" + ResourceDef.class.getSimpleName() + " annotation contains no resource name(): " + theClass.getCanonicalName()
287 					+ " - This is only allowed for types that extend other resource types ");
288 			}
289 		}
290 
291 		String resourceNameLowerCase = resourceName.toLowerCase();
292 		Class<? extends IBaseResource> builtInType = myNameToResourceType.get(resourceNameLowerCase);
293 		boolean standardType = builtInType != null && builtInType.equals(theClass) == true;
294 		if (primaryNameProvider) {
295 			if (builtInType != null && builtInType.equals(theClass) == false) {
296 				primaryNameProvider = false;
297 			}
298 		}
299 
300 		String resourceId = resourceDefinition.id();
301 		if (!isBlank(resourceId)) {
302 			if (myIdToResourceDefinition.containsKey(resourceId)) {
303 				throw new ConfigurationException("The following resource types have the same ID of '" + resourceId + "' - " + theClass.getCanonicalName() + " and "
304 					+ myIdToResourceDefinition.get(resourceId).getImplementingClass().getCanonicalName());
305 			}
306 		}
307 
308 		RuntimeResourceDefinition resourceDef = new RuntimeResourceDefinition(myContext, resourceName, theClass, resourceDefinition, standardType, myClassToElementDefinitions);
309 		myClassToElementDefinitions.put(theClass, resourceDef);
310 		if (primaryNameProvider) {
311 			if (resourceDef.getStructureVersion() == myVersion) {
312 				myNameToResourceDefinitions.put(resourceNameLowerCase, resourceDef);
313 			}
314 		}
315 
316 		myIdToResourceDefinition.put(resourceId, resourceDef);
317 
318 		scanResourceForSearchParams(theClass, resourceDef);
319 
320 		/*
321 		 * See #423:
322 		 * If the type contains a field that has a custom type, we want to make
323 		 * sure that this type gets scanned as well
324 		 */
325 		resourceDef.populateScanAlso(myScanAlso);
326 
327 		return resourceName;
328 	}

330 	private void scanResourceForSearchParams(Class<? extends IBaseResource> theClass, RuntimeResourceDefinition theResourceDef) {
331 
332 		Map<String, RuntimeSearchParam> nameToParam = new HashMap<>();
333 		Map<Field, SearchParamDefinition> compositeFields = new LinkedHashMap<>();
334 
335 		/*
336 		 * Make sure we pick up fields in interfaces too.. This ensures that we
337 		 * grab the _id field which generally gets picked up via interface
338 		 */
339 		Set<Field> fields = new HashSet<>(Arrays.asList(theClass.getFields()));
340 		Class<?> nextClass = theClass;
341 		do {
342 			for (Class<?> nextInterface : nextClass.getInterfaces()) {
343 				fields.addAll(Arrays.asList(nextInterface.getFields()));
344 			}
345 			nextClass = nextClass.getSuperclass();
346 		} while (nextClass.equals(Object.class) == false);
347 
348 		/*
349 		 * Now scan the fields for search params
350 		 */
351 		for (Field nextField : fields) {
352 			SearchParamDefinition searchParam = pullAnnotation(nextField, SearchParamDefinition.class);
353 			if (searchParam != null) {
354 				RestSearchParameterTypeEnum paramType = RestSearchParameterTypeEnum.forCode(searchParam.type().toLowerCase());
355 				if (paramType == null) {
356 					throw new ConfigurationException("Search param " + searchParam.name() + " has an invalid type: " + searchParam.type());
357 				}
358 				Set<String> providesMembershipInCompartments;
359 				providesMembershipInCompartments = new HashSet<>();
360 				for (Compartment next : searchParam.providesMembershipIn()) {
361 					if (paramType != RestSearchParameterTypeEnum.REFERENCE) {
362 						StringBuilder b = new StringBuilder();
363 						b.append("Search param ");
364 						b.append(searchParam.name());
365 						b.append(" on resource type ");
366 						b.append(theClass.getName());
367 						b.append(" provides compartment membership but is not of type 'reference'");
368 						ourLog.warn(b.toString());
369 						continue;
370 //						throw new ConfigurationException(b.toString());
371 					}
372 					providesMembershipInCompartments.add(next.name());
373 				}
374 
375 				if (paramType == RestSearchParameterTypeEnum.COMPOSITE) {
376 					compositeFields.put(nextField, searchParam);
377 					continue;
378 				}
379 
380 
381 				Collection<String> base = Collections.singletonList(theResourceDef.getName());
382 				RuntimeSearchParam param = new RuntimeSearchParam(null, null, searchParam.name(), searchParam.description(), searchParam.path(), paramType, null, providesMembershipInCompartments, toTargetList(searchParam.target()), RuntimeSearchParamStatusEnum.ACTIVE, base);
383 				theResourceDef.addSearchParam(param);
384 				nameToParam.put(param.getName(), param);
385 			}
386 		}
387 
388 		for (Entry<Field, SearchParamDefinition> nextEntry : compositeFields.entrySet()) {
389 			SearchParamDefinition searchParam = nextEntry.getValue();
390 
391 			List<RuntimeSearchParam> compositeOf = new ArrayList<>();
392 			for (String nextName : searchParam.compositeOf()) {
393 				RuntimeSearchParam param = nameToParam.get(nextName);
394 				if (param == null) {
395 					ourLog.warn("Search parameter {}.{} declares that it is a composite with compositeOf value '{}' but that is not a valid parametr name itself. Valid values are: {}",
396 						theResourceDef.getName(), searchParam.name(), nextName, nameToParam.keySet());
397 					continue;
398 				}
399 				compositeOf.add(param);
400 			}
401 
402 			RuntimeSearchParam param = new RuntimeSearchParam(null, null, searchParam.name(), searchParam.description(), searchParam.path(), RestSearchParameterTypeEnum.COMPOSITE, compositeOf, null, toTargetList(searchParam.target()), RuntimeSearchParamStatusEnum.ACTIVE);
403 			theResourceDef.addSearchParam(param);
404 		}
405 	}

407 	private Set<String> toTargetList(Class<? extends IBaseResource>[] theTarget) {
408 		HashSet<String> retVal = new HashSet<>();
409 
410 		for (Class<? extends IBaseResource> nextType : theTarget) {
411 			ResourceDef resourceDef = nextType.getAnnotation(ResourceDef.class);
412 			if (resourceDef != null) {
413 				retVal.add(resourceDef.name());
414 			}
415 		}
416 
417 		return retVal;
418 	}

420 	static Class<?> determineElementType(Field next) {
421 		Class<?> nextElementType = next.getType();
422 		if (List.class.equals(nextElementType)) {
423 			nextElementType = ReflectionUtil.getGenericCollectionTypeOfField(next);
424 		} else if (Collection.class.isAssignableFrom(nextElementType)) {
425 			throw new ConfigurationException("Field '" + next.getName() + "' in type '" + next.getClass().getCanonicalName() + "' is a Collection - Only java.util.List curently supported");
426 		}
427 		return nextElementType;
428 	}

431 	static IValueSetEnumBinder<Enum<?>> getBoundCodeBinder(Field theNext) {
432 		Class<?> bound = getGenericCollectionTypeOfCodedField(theNext);
433 		if (bound == null) {
434 			throw new ConfigurationException("Field '" + theNext + "' has no parameter for " + BoundCodeDt.class.getSimpleName() + " to determine enum type");
435 		}
436 
437 		String fieldName = "VALUESET_BINDER";
438 		try {
439 			Field bindingField = bound.getField(fieldName);
440 			return (IValueSetEnumBinder<Enum<?>>) bindingField.get(null);
441 		} catch (Exception e) {
442 			throw new ConfigurationException("Field '" + theNext + "' has type parameter " + bound.getCanonicalName() + " but this class has no valueset binding field (must have a field called " + fieldName + ")", e);
443 		}
444 	}

462 	private static Class<?> getGenericCollectionTypeOfCodedField(Field next) {
463 		Class<?> type;
464 		ParameterizedType collectionType = (ParameterizedType) next.getGenericType();
465 		Type firstArg = collectionType.getActualTypeArguments()[0];
466 		if (ParameterizedType.class.isAssignableFrom(firstArg.getClass())) {
467 			ParameterizedType pt = ((ParameterizedType) firstArg);
468 			firstArg = pt.getActualTypeArguments()[0];
469 			type = (Class<?>) firstArg;
470 		} else {
471 			type = (Class<?>) firstArg;
472 		}
473 		return type;
474 	}

476 	static Set<Class<? extends IBase>> scanVersionPropertyFile(Set<Class<? extends IBase>> theDatatypes, Map<String, Class<? extends IBaseResource>> theResourceTypes, FhirVersionEnum theVersion, Map<Class<? extends IBase>, BaseRuntimeElementDefinition<?>> theExistingElementDefinitions) {
477 		Set<Class<? extends IBase>> retVal = new HashSet<>();
478 
479 		try (InputStream str = theVersion.getVersionImplementation().getFhirVersionPropertiesFile()) {
480 			Properties prop = new Properties();
481 			prop.load(str);
482 			for (Entry<Object, Object> nextEntry : prop.entrySet()) {
483 				String nextKey = nextEntry.getKey().toString();
484 				String nextValue = nextEntry.getValue().toString();
485 
486 				if (nextKey.startsWith("datatype.")) {
487 					if (theDatatypes != null) {
488 						try {
489 							// Datatypes
490 
491 							@SuppressWarnings("unchecked")
492 							Class<? extends IBase> dtType = (Class<? extends IBase>) Class.forName(nextValue);
493 							if (theExistingElementDefinitions.containsKey(dtType)) {
494 								continue;
495 							}
496 							retVal.add(dtType);
497 
498 							if (IElement.class.isAssignableFrom(dtType)) {
499 								@SuppressWarnings("unchecked")
500 								Class<? extends IElement> nextClass = (Class<? extends IElement>) dtType;
501 								theDatatypes.add(nextClass);
502 							} else if (IBaseDatatype.class.isAssignableFrom(dtType)) {
503 								@SuppressWarnings("unchecked")
504 								Class<? extends IBaseDatatype> nextClass = (Class<? extends IBaseDatatype>) dtType;
505 								theDatatypes.add(nextClass);
506 							} else {
507 								ourLog.warn("Class is not assignable from " + IElement.class.getSimpleName() + " or " + IBaseDatatype.class.getSimpleName() + ": " + nextValue);
508 								continue;
509 							}
510 
511 						} catch (ClassNotFoundException e) {
512 							throw new ConfigurationException("Unknown class[" + nextValue + "] for data type definition: " + nextKey.substring("datatype.".length()), e);
513 						}
514 					}
515 				} else if (nextKey.startsWith("resource.")) {
516 					// Resources
517 					String resName = nextKey.substring("resource.".length()).toLowerCase();
518 					try {
519 						@SuppressWarnings("unchecked")
520 						Class<? extends IBaseResource> nextClass = (Class<? extends IBaseResource>) Class.forName(nextValue);
521 						if (theExistingElementDefinitions.containsKey(nextClass)) {
522 							continue;
523 						}
524 						if (!IBaseResource.class.isAssignableFrom(nextClass)) {
525 							throw new ConfigurationException("Class is not assignable from " + IBaseResource.class.getSimpleName() + ": " + nextValue);
526 						}
527 
528 						theResourceTypes.put(resName, nextClass);
529 					} catch (ClassNotFoundException e) {
530 						throw new ConfigurationException("Unknown class[" + nextValue + "] for resource definition: " + nextKey.substring("resource.".length()), e);
531 					}
532 				} else {
533 					throw new ConfigurationException("Unexpected property in version property file: " + nextKey + "=" + nextValue);
534 				}
535 			}
536 		} catch (IOException e) {
537 			throw new ConfigurationException("Failed to load model property file from classpath: " + "/ca/uhn/fhir/model/dstu/model.properties");
538 		}
539 
540 		return retVal;
541 	}
